92      assertEquals(60.0 * 60, Time.getRelativeTime("-1h"), 0);
93      assertEquals(60.0 * 60 * 24, Time.getRelativeTime("-1d"), 0);
94      assertEquals(60.0 * 60 * 24 * 7, Time.getRelativeTime("-1w"), 0);
95      assertEquals(60.0 * 60 * 24 * 30, Time.getRelativeTime("-1m"), 0);

99    @Test
100   public void when_mainCalled_then_convertTime() throws Exception {
101     Time.main(new String[0]);
102     assertTrue(outContent.toString().contains("-d2j [yyyymmddhhmmss] date to j2k"));
103     outContent.reset();
104 
105     String[] args = new String[2];
106 
107     args[0] = "-j2d";
108     args[1] = "5.01325203E8";
109     Time.main(args);
110     assertEquals("2015-11-20 21:00:03.000\n", outContent.toString());
111     outContent.reset();
112 
113     args[0] = "-j2e";
114     args[1] = "5.01325203E8";
115     Time.main(args);
116     assertEquals("1.448053203E9\n", outContent.toString());
117     outContent.reset();
118 
119     args[0] = "-d2j";
120     args[1] = INPUT_TIME_STRING;
121     Time.main(args);
122     assertEquals("5.01325203E8\n", outContent.toString());
123     outContent.reset();
124 
125     args[0] = "-e2d";
126     args[1] = "1448053203";
127     Time.main(args);
128     assertEquals("2015-11-20 21:00:03.000\n", outContent.toString());
129     outContent.reset();
130   }
