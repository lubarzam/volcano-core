6   package gov.usgs.volcanoes.core.args;
7   
8   import com.martiansoftware.jsap.JSAPException;
9   import com.martiansoftware.jsap.JSAPResult;
10  import com.martiansoftware.jsap.Parameter;
11  import com.martiansoftware.jsap.SimpleJSAP;
12  
13  /**
14   * An argument processor.
15   *
16   * @author Tom Parker
17   */
18  public class Args implements Arguments {
19    protected SimpleJSAP jsap;
20    protected JSAPResult jsapResult;
21  
22    /**
23     * Construct a Args instance.
24     *
25     * @param programName Name of the application use is running
26     * @param explanation Brief explanation of application use
27     * @param parameters Prepopulated list of JSAP Parameters
28     * @throws ArgumentException if things go wrong
29     */
30    public Args(String programName, String explanation, Parameter[] parameters)
31        throws ArgumentException {
32      try {
33        jsap = new SimpleJSAP(programName, explanation, parameters);
34      } catch (final JSAPException ex) {
35        throw new ArgumentException(ex);
36      }
37    }
38  
39    @Override
40    public Parameter getById(String id) {
41      return jsap.getByID(id);
42    }
43  
44    @Override
45    public boolean messagePrinted() {
46      return jsap.messagePrinted();
47    }
48  
49    /**
50     * Parse the command line.
51     *
52     * @param args The command line args
53     * @return the parsed command line
54     */
55    @Override
56    public JSAPResult parse(String[] args) {
57      jsapResult = jsap.parse(args);
58  
59      return jsapResult;
60    }
61  
62    /**
63     * Register parameter with JSAP.
64     * 
65     * @param parameter the Parameter to register
66     * @throws ArgumentException when parameter cannot be registered
67     */
68    @Override
69    public void registerParameter(Parameter parameter) throws ArgumentException {
70      try {
71        jsap.registerParameter(parameter);
72      } catch (final JSAPException ex) {
73        throw new ArgumentException(ex);
74      }
75    }
76  }
