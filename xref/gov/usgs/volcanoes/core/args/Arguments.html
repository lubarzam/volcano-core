6   package gov.usgs.volcanoes.core.args;
7   
8   import com.martiansoftware.jsap.JSAPResult;
9   import com.martiansoftware.jsap.Parameter;
10  
11  /**
12   * Decorator interface for command line arguments.
13   *
14   * @author Tom Parker
15   */
16  public interface Arguments {
17  
18    /**
19     * Find a parameter given its id.
20     *
21     * @param id The id to search for
22     * @return The value of the id
23     */
24    public Parameter getById(String id);
25  
26    /**
27     * Report whether JSAP displayed a message to the user. Typically means either an error or that
28     * the user requested help.
29     *
30     * @return True if JSAP printed a message while parsing arguments
31     */
32    public boolean messagePrinted();
33  
34    /**
35     * Parse a command line.
36     *
37     * @param args The arguments provided at program launch
38     * @return The parsed arguments
39     * @throws ArgumentException if anything goes wrong. Yes, pretty much anything.
40     */
41    public JSAPResult parse(String[] args) throws ArgumentException;
42  
43    /**
44     * Register a Parameter with JSAP.
45     *
46     * @param parameter The Parameter to register
47     * @throws ArgumentException if I cannot register the parameter
48     */
49    public void registerParameter(Parameter parameter) throws ArgumentException;
50  }
