6   package gov.usgs.volcanoes.core.args.decorator;

8   import com.martiansoftware.jsap.JSAP;
9   import com.martiansoftware.jsap.UnflaggedOption;
10  
11  import gov.usgs.volcanoes.core.args.ArgUtil;
12  import gov.usgs.volcanoes.core.args.ArgsDecorator;
13  import gov.usgs.volcanoes.core.args.ArgumentException;
14  import gov.usgs.volcanoes.core.args.Arguments;
15  
16  /**
17   * Gather a config filename from the command line.
18   *
19   * <P>The filename is taken from the first available unflagged option and placed in 
20   * "config-filename". Be careful of the order that Args are added.
21   *
22   * @author Tom Parker
23   */
24  public class ConfigFileArg extends ArgsDecorator {
25  
26    /**
27     * Constructs a ConfigFileArg and adds its parameter to the list.
28     *
29     * @param defaultFileName The value to assign if the parameter is not given on the command line
30     * @param nextArg The next Arguments object in the chain
31     * @throws ArgumentException if new Parameter cannot be created. Not sure why this would happen
32     */
33    public ConfigFileArg(final String defaultFileName, final Arguments nextArg)
34        throws ArgumentException {
35      super(nextArg);
36  
37      boolean isRequired = ArgUtil.isRequired(defaultFileName);
38      String helpString = "The config file name.";
39      if (!isRequired) {
40        helpString += " (default: " + defaultFileName + ")";
41      }
42  
43      registerParameter(new UnflaggedOption("config-filename", JSAP.STRING_PARSER, defaultFileName,
44          isRequired, JSAP.NOT_GREEDY, helpString));
45    }
46  }
