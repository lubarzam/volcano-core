6   package gov.usgs.volcanoes.core.args.decorator;
7   
8   import com.martiansoftware.jsap.FlaggedOption;
9   import com.martiansoftware.jsap.JSAP;
10  
11  import gov.usgs.volcanoes.core.args.ArgsDecorator;
12  import gov.usgs.volcanoes.core.args.ArgumentException;
13  import gov.usgs.volcanoes.core.args.Arguments;
14  import gov.usgs.volcanoes.core.args.parser.ScnlParser;
15  import gov.usgs.volcanoes.core.data.Scnl;
16  
17  
18  /**
19   * Gather a SCNL from the command line.
20   *
21   *
22   * @author Tom Parker
23   */
24  public class ScnlArg extends ArgsDecorator {
25  
26    /**
27     * Register arguments that define a SCNL.
28     *
29     * @param isRequired If true, this option must be provided
30     * @param nextArg The Argument object I'm wrapping
31     * @throws ArgumentException if parameters cannot be registered
32     */
33    public ScnlArg(boolean isRequired, Arguments nextArg) throws ArgumentException {
34      super(nextArg);
35  
36      String delimiter = Scnl.DELIMITER;
37      nextArg.registerParameter(
38          new FlaggedOption("channel", new ScnlParser(), JSAP.NO_DEFAULT, isRequired, 'c', "channel",
39              String.format("Channel as 'S%sC%sN%sL'\n", delimiter, delimiter, delimiter)));
40    }
41  }
