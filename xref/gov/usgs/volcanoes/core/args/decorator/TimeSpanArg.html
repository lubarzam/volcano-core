6   package gov.usgs.volcanoes.core.args.decorator;
7   
8   import com.martiansoftware.jsap.FlaggedOption;
9   import com.martiansoftware.jsap.JSAP;
10  import com.martiansoftware.jsap.StringParser;
11  
12  import gov.usgs.volcanoes.core.args.ArgsDecorator;
13  import gov.usgs.volcanoes.core.args.ArgumentException;
14  import gov.usgs.volcanoes.core.args.Arguments;
15  import gov.usgs.volcanoes.core.args.parser.TimeSpanParser;
16  
17  
18  /**
19   * Gather a date range from the command line.
20   *
21   * @author Tom Parker
22   */
23  public class TimeSpanArg extends ArgsDecorator {
24  
25    /**
26     * Register arguments that define a date range.
27     *
28     * @param dateFormat Format string suitable for feeding to SimpleDateFormat
29     * @param nextArg The Argument object I'm wrapping
30     * @throws ArgumentException if parameters cannot be registered
31     */
32    public TimeSpanArg(String dateFormat, boolean isRequired, Arguments nextArg)
33        throws ArgumentException {
34      super(nextArg);
35  
36  
37      final StringParser timeSpanParser = new TimeSpanParser(dateFormat);
38      nextArg.registerParameter(
39          new FlaggedOption("timeSpan", timeSpanParser, JSAP.NO_DEFAULT, isRequired, 't', "timeSpan",
40              String.format("Time span as %s-%s\n", dateFormat, dateFormat)));
41    }
42  }
