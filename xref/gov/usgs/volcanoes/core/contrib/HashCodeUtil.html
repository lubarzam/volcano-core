29  package gov.usgs.volcanoes.core.contrib;
30  
31  import java.lang.reflect.Array;
32  
33  /**
34   * Collected methods which allow easy implementation of <tt>hashCode</tt>.
35   *
36   * <p>Example use case:
37   * 
38   * <pre>
39   * public int hashCode() {
40   *   int result = HashCodeUtil.SEED;
41   *   // collect the contributions of various fields
42   *   result = HashCodeUtil.hash(result, fPrimitive);
43   *   result = HashCodeUtil.hash(result, fObject);
44   *   result = HashCodeUtil.hash(result, fArray);
45   *   return result;
46   * }
47   * </pre>
48   * 
49   * <p>Taken from http://www.javapractices.com/topic/TopicAction.do?Id=28
50   */
51  public final class HashCodeUtil {
52  
53    /**
54    * An initial value for a <tt>hashCode</tt>, to which is added contributions
55    * from fields. Using a non-zero value decreases collisons of <tt>hashCode</tt>
56    * values.
57    */
58    public static final int SEED = 23;
59  
60    /** booleans.  */
61    public static int hash(int aSeed, boolean aBoolean) {
62      log("boolean...");
63      return firstTerm(aSeed) + (aBoolean ? 1 : 0);
64    }
65  
66    /*** chars.  */
67    public static int hash(int aSeed, char aChar) {
68      log("char...");
69      return firstTerm(aSeed) + (int) aChar;
70    }
71  
72    /** ints.  */
73    public static int hash(int aSeed, int aInt) {
74      /*
75       * Implementation Note
76       * Note that byte and short are handled by this method, through
77       * implicit conversion.
78       */
79      log("int...");
80      return firstTerm(aSeed) + aInt;
81    }
82  
83    /** longs.  */
84    public static int hash(int aSeed, long aLong) {
85      log("long...");
86      return firstTerm(aSeed) + (int) (aLong ^ (aLong >>> 32));
87    }
88  
89    /** floats.  */
90    public static int hash(int aSeed, float aFloat) {
91      return hash(aSeed, Float.floatToIntBits(aFloat));
92    }
93  
94    /** doubles. */
95    public static int hash(int aSeed, double aDouble) {
96      return hash(aSeed, Double.doubleToLongBits(aDouble));
97    }
98  
99    /**
100   * <tt>aObject</tt> is a possibly-null object field, and possibly an array.
101   *
102   * If <tt>aObject</tt> is an array, then each element may be a primitive
103   * or a possibly-null object.
104   */
105   public static int hash(int aSeed, Object aObject) {
106     int result = aSeed;
107     if (aObject == null) {
108       result = hash(result, 0);
109     } else if (!isArray(aObject)) {
110       result = hash(result, aObject.hashCode());
111     } else {
112       int length = Array.getLength(aObject);
113       for (int idx = 0; idx < length; ++idx) {
114         Object item = Array.get(aObject, idx);
115         // if an item in the array references the array itself, prevent infinite looping
116         if (!(item == aObject))
117           // recursive call!
118           result = hash(result, item);
119       }
120     }
121     return result;
122   }
123 
124   // PRIVATE
125   private static final int fODD_PRIME_NUMBER = 37;
126 
127   private static int firstTerm(int aSeed) {
128     return fODD_PRIME_NUMBER * aSeed;
129   }
130 
131   private static boolean isArray(Object aObject) {
132     return aObject.getClass().isArray();
133   }
134 
135   private static void log(String aMessage) {
136     // System.out.println(aMessage);
137   }
138 }
