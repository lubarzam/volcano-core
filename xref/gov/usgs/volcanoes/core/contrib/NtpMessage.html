8   /**
9    * This class represents a NTP message, as specified in RFC 2030. The message
10   * format is compatible with all versions of NTP and SNTP.
11   *
12   * This class does not support the optional authentication protocol, and
13   * ignores the key ID and message digest fields.
14   * 
15   * For convenience, this class exposes message values as native Java types, not
16   * the NTP-specified data formats. For example, timestamps are
17   * stored as doubles (as opposed to the NTP unsigned 64-bit fixed point
18   * format).
19   * 
20   * However, the contructor NtpMessage(byte[]) and the method toByteArray()
21   * allow the import and export of the raw NTP message format.
22   * 
23   * 
24   * Usage example
25   * 
26   * // Send message
27   * DatagramSocket socket = new DatagramSocket();
28   * InetAddress address = InetAddress.getByName("ntp.cais.rnp.br");
29   * byte[] buf = new NtpMessage().toByteArray();
30   * DatagramPacket packet = new DatagramPacket(buf, buf.length, address, 123);
31   * socket.send(packet);
32   * 
33   * // Get response
34   * socket.receive(packet);
35   * System.out.println(msg.toString());
36   * 
37   *

39   *

41   * under the terms of the GNU General Public License as published by the Free

63     * This is a two-bit code warning of an impending leap second to be

68     * ----- -------
69     * 0 no warning

77    /**
78     * This value indicates the NTP/SNTP version number. The version number
79     * is 3 for Version 3 (IPv4 only) and 4 for Version 4 (IPv4, IPv6 and OSI).
80     * If necessary to distinguish between IPv4, IPv6 and OSI, the
81     * encapsulating context must be inspected.
82     */

86    /**
87     * This value indicates the mode, with values defined as follows:
88     * 
89     * Mode Meaning
90     * ---- -------
91     * 0 reserved
92     * 1 symmetric active
93     * 2 symmetric passive
94     * 3 client
95     * 4 server
96     * 5 broadcast
97     * 6 reserved for NTP control message
98     * 7 reserved for private use
99     * 
100    * In unicast and anycast modes, the client sets this field to 3 (client)
101    * in the request and the server sets it to 4 (server) in the reply. In
102    * multicast mode, the server sets this field to 5 (broadcast).
103    */
104   public byte mode = 0;

108    * This value indicates the stratum level of the local clock, with values
109    * defined as follows:
110    * 
111    * Stratum Meaning
112    * ----------------------------------------------
113    * 0 unspecified or unavailable
114    * 1 primary reference (e.g., radio clock)
115    * 2-15 secondary reference (via NTP or SNTP)
116    * 16-255 reserved

122    * This value indicates the maximum interval between successive messages,
123    * in seconds to the nearest power of two. The values that can appear in
124    * this field presently range from 4 (16 s) to 14 (16284 s); however, most
125    * applications use only the sub-range 6 (64 s) to 10 (1024 s).

134    * found in some workstations.

140    * This value indicates the total roundtrip delay to the primary reference

142    * and negative values, depending on the relative time and frequency
143    * offsets. The values that normally appear in this field range from
144    * negative values of a few milliseconds to positive values of several

158   /**
159    * This is a 4-byte array identifying the particular reference source.
160    * In the case of NTP Version 3 or Version 4 stratum-0 (unspecified) or
161    * stratum-1 (primary) servers, this is a four-character ASCII string, left
162    * justified and zero padded to 32 bits. In NTP Version 3 secondary
163    * servers, this is the 32-bit IPv4 address of the reference source. In NTP
164    * Version 4 secondary servers, this is the low order 32 bits of the latest
165    * transmit timestamp of the reference source. NTP primary (stratum 1)
166    * servers should set this field to a code identifying the external
167    * reference source according to the following list. If the external
168    * reference is one of those listed, the associated code should be used.
169    * Codes for sources not listed can be contrived as appropriate.
170    * 
171    * Code External Reference Source
172    * ---- -------------------------
173    * LOCL uncalibrated local clock used as a primary reference for
174    * a subnet without external means of synchronization
175    * PPS atomic clock or other pulse-per-second source
176    * individually calibrated to national standards
177    * ACTS NIST dialup modem service
178    * USNO USNO modem service
179    * PTB PTB (Germany) modem service
180    * TDF Allouis (France) Radio 164 kHz
181    * DCF Mainflingen (Germany) Radio 77.5 kHz
182    * MSF Rugby (UK) Radio 60 kHz
183    * WWV Ft. Collins (US) Radio 2.5, 5, 10, 15, 20 MHz
184    * WWVB Boulder (US) Radio 60 kHz
185    * WWVH Kaui Hawaii (US) Radio 2.5, 5, 10, 15 MHz
186    * CHU Ottawa (Canada) Radio 3330, 7335, 14670 kHz
187    * LORC LORAN-C radionavigation system
188    * OMEG OMEGA radionavigation system
189    * GPS Global Positioning Service
190    * GOES Geostationary Orbit Environment Satellite
191    */
192   public byte[] referenceIdentifier = { 0, 0, 0, 0 };

224   /**
225    * Constructs a new NtpMessage from an array of bytes.
226    */
227   public NtpMessage(byte[] array) {
228     // See the packet format diagram in RFC 2030 for details
229     leapIndicator = (byte) ((array[0] >> 6) & 0x3);
230     version = (byte) ((array[0] >> 3) & 0x7);
231     mode = (byte) (array[0] & 0x7);
232     stratum = unsignedByteToShort(array[1]);
233     pollInterval = array[2];
234     precision = array[3];
235 
236     rootDelay = (array[4] * 256.0) + unsignedByteToShort(array[5])
237         + (unsignedByteToShort(array[6]) / 256.0) + (unsignedByteToShort(array[7]) / 65536.0);
238 
239     rootDispersion = (unsignedByteToShort(array[8]) * 256.0) + unsignedByteToShort(array[9])
240         + (unsignedByteToShort(array[10]) / 256.0) + (unsignedByteToShort(array[11]) / 65536.0);
241 
242     referenceIdentifier[0] = array[12];
243     referenceIdentifier[1] = array[13];
244     referenceIdentifier[2] = array[14];
245     referenceIdentifier[3] = array[15];
246 
247     referenceTimestamp = decodeTimestamp(array, 16);
248     originateTimestamp = decodeTimestamp(array, 24);
249     receiveTimestamp = decodeTimestamp(array, 32);
250     transmitTimestamp = decodeTimestamp(array, 40);
251   }

256    * Constructs a new NtpMessage in client -> server mode, and sets the

259   public NtpMessage() {
260     // Note that all the other member variables are already set with

262     this.mode = 3;
263     this.transmitTimestamp = (System.currentTimeMillis() / 1000.0) + 2208988800.0;

268   /**
269    * This method constructs the data bytes of a raw NTP packet.
270    */
271   public byte[] toByteArray() {
272     // All bytes are automatically set to 0
273     byte[] p = new byte[48];
274 
275     p[0] = (byte) (leapIndicator << 6 | version << 3 | mode);
276     p[1] = (byte) stratum;
277     p[2] = (byte) pollInterval;
278     p[3] = (byte) precision;
279 
280     // root delay is a signed 16.16-bit FP, in Java an int is 32-bits
281     int l = (int) (rootDelay * 65536.0);
282     p[4] = (byte) ((l >> 24) & 0xFF);
283     p[5] = (byte) ((l >> 16) & 0xFF);
284     p[6] = (byte) ((l >> 8) & 0xFF);
285     p[7] = (byte) (l & 0xFF);
286 
287     // root dispersion is an unsigned 16.16-bit FP, in Java there are no
288     // unsigned primitive types, so we use a long which is 64-bits
289     long ul = (long) (rootDispersion * 65536.0);
290     p[8] = (byte) ((ul >> 24) & 0xFF);
291     p[9] = (byte) ((ul >> 16) & 0xFF);
292     p[10] = (byte) ((ul >> 8) & 0xFF);
293     p[11] = (byte) (ul & 0xFF);
294 
295     p[12] = referenceIdentifier[0];
296     p[13] = referenceIdentifier[1];
297     p[14] = referenceIdentifier[2];
298     p[15] = referenceIdentifier[3];
299 
300     encodeTimestamp(p, 16, referenceTimestamp);
301     encodeTimestamp(p, 24, originateTimestamp);
302     encodeTimestamp(p, 32, receiveTimestamp);
303     encodeTimestamp(p, 40, transmitTimestamp);
304 
305     return p;
306   }

310   /**
311    * Returns a string representation of a NtpMessage
312    */
313   public String toString() {
314     String precisionStr = new DecimalFormat("0.#E0").format(Math.pow(2, precision));
315 
316     return "Leap indicator: " + leapIndicator + "\n" + "Version: " + version + "\n" + "Mode: "
317         + mode + "\n" + "Stratum: " + stratum + "\n" + "Poll: " + pollInterval + "\n"
318         + "Precision: " + precision + " (" + precisionStr + " seconds)\n" + "Root delay: "
319         + new DecimalFormat("0.00").format(rootDelay * 1000) + " ms\n" + "Root dispersion: "
320         + new DecimalFormat("0.00").format(rootDispersion * 1000) + " ms\n"
321         + "Reference identifier: "
322         + referenceIdentifierToString(referenceIdentifier, stratum, version) + "\n"
323         + "Reference timestamp: " + timestampToString(referenceTimestamp) + "\n"
324         + "Originate timestamp: " + timestampToString(originateTimestamp) + "\n"
325         + "Receive timestamp:   " + timestampToString(receiveTimestamp) + "\n"
326         + "Transmit timestamp:  " + timestampToString(transmitTimestamp);
327   }

331   /**
332    * Converts an unsigned byte to a short. By default, Java assumes that
333    * a byte is signed.
334    */
335   public static short unsignedByteToShort(byte b) {
336     if ((b & 0x80) == 0x80)
337       return (short) (128 + (b & 0x7f));
338     else
339       return (short) b;
340   }

344   /**
345    * Will read 8 bytes of a message beginning at <code>pointer</code>
346    * and return it as a double, according to the NTP 64-bit timestamp
347    * format.
348    */
349   public static double decodeTimestamp(byte[] array, int pointer) {
350     double r = 0.0;
351 
352     for (int i = 0; i < 8; i++) {
353       r += unsignedByteToShort(array[pointer + i]) * Math.pow(2, (3 - i) * 8);
354     }
355 
356     return r;
357   }

361   /**
362    * Encodes a timestamp in the specified position in the message
363    */
364   public static void encodeTimestamp(byte[] array, int pointer, double timestamp) {
365     // Converts a double into a 64-bit fixed point
366     for (int i = 0; i < 8; i++) {
367       // 2^24, 2^16, 2^8, .. 2^-32
368       double base = Math.pow(2, (3 - i) * 8);
369 
370       // Capture byte value
371       array[pointer + i] = (byte) (timestamp / base);
372 
373       // Subtract captured value from remaining total
374       timestamp = timestamp - (double) (unsignedByteToShort(array[pointer + i]) * base);
375     }
376 
377     // From RFC 2030: It is advisable to fill the non-significant
378     // low order bits of the timestamp with a random, unbiased
379     // bitstring, both to avoid systematic roundoff errors and as
380     // a means of loop detection and replay detection.
381     array[7] = (byte) (Math.random() * 255.0);
382   }

386   /**
387    * Returns a timestamp (number of seconds since 00:00 1-Jan-1900) as a
388    * formatted date/time string.
389    */
390   public static String timestampToString(double timestamp) {
391     if (timestamp == 0)
392       return "0";
393 
394     // timestamp is relative to 1900, utc is used by Java and is relative
395     // to 1970
396     double utc = timestamp - (2208988800.0);
397 
398     // milliseconds
399     long ms = (long) (utc * 1000.0);
400 
401     // date/time
402     String date = new SimpleDateFormat("dd-MMM-yyyy HH:mm:ss").format(new Date(ms));
403 
404     // fraction
405     double fraction = timestamp - ((long) timestamp);
406     String fractionSting = new DecimalFormat(".000000").format(fraction);
407 
408     return date + fractionSting;
409   }

413   /**
414    * Returns a string representation of a reference identifier according
415    * to the rules set out in RFC 2030.
416    */
417   public static String referenceIdentifierToString(byte[] ref, short stratum, byte version) {
418     // From the RFC 2030:
419     // In the case of NTP Version 3 or Version 4 stratum-0 (unspecified)
420     // or stratum-1 (primary) servers, this is a four-character ASCII
421     // string, left justified and zero padded to 32 bits.
422     if (stratum == 0 || stratum == 1) {
423       return new String(ref);
424     }
425 
426     // In NTP Version 3 secondary servers, this is the 32-bit IPv4
427     // address of the reference source.
428     else if (version == 3) {
429       return unsignedByteToShort(ref[0]) + "." + unsignedByteToShort(ref[1]) + "."
430           + unsignedByteToShort(ref[2]) + "." + unsignedByteToShort(ref[3]);
431     }
432 
433     // In NTP Version 4 secondary servers, this is the low order 32 bits
434     // of the latest transmit timestamp of the reference source.
435     else if (version == 4) {
436       return "" + ((unsignedByteToShort(ref[0]) / 256.0) + (unsignedByteToShort(ref[1]) / 65536.0)
437           + (unsignedByteToShort(ref[2]) / 16777216.0)
438           + (unsignedByteToShort(ref[3]) / 4294967296.0));
439     }
440 
441     return "";
442   }
