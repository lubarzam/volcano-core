8   /**
9    * Data structure used in Hypo71 algorithm.
10   * 
11   * @author Oleg Shepelev
12   */
13  @XmlRootElement
14  public class Station implements Serializable {
15    /**
16     * If IW = *, then this station has zero weight assigned to its P and/or S reading(s).
17     */
18    char IW;
19    /**
20     * Station name.
21     */
22    String NSTA;
23    /**
24     * Degree portion of latitude.
25     */
26    int LAT1;
27    /**
28     * Minute portion of latitude.
29     */
30    double LAT2;
31    /**
32     * Punch N or leave this column blank for stations in northern hemisphere. Punch S for stations in sourthern hemisphere.
33     */
34    char INS;
35    /**
36     * Degree portion of longitude.
37     */
38    int LON1;
39    /**
40     * Minute portion of longitude.
41     */
42    double LON2;
43    /**
44     * Punch E for eastern longitude. W or blank for western.
45     */
46    char IEW;
47    /**
48     * Elevation in meters. This data is not used in the program.
49     */
50    int IELV;
51    double dly;
52    /**
53     * Station delay in seconds.
54     */
55    double FMGC;
56    /**
57     * Station correction for FMAG.
58     */
59    double XMGC;
60    /**
61     * System number is assigned for each station so that the frequency response curve of the seismometer and preamp is specified for the amplitude magnitude calculation (XMAG).
62     */
63    int KLAS;
64    /**
65     * Standard period for XMAG.
66     */
67    double PRR;
68    /**
69     * Standard calibration for XMAG.
70     */
71    double CALR;
72    /**
73     * Calibration indicator.
74     */
75    int ICAL;
76    /**
77     * Year, month and day.
78     */
79    int NDATE;
80    /**
81     * Hour and minute.
82     */
83    int NHRMN;
84  
85    // Missing special fields
86    // NSTA[I] - NSTA, DELI - dly, AZI, EXGAP, RDGAP
87    double AZI;
88    double EXGAP;
89    double RDGAP;
90  
91    // Station output
92    // String STN - NSTA
93    /**
94     * Epicentral distance in km.
95     */
96    double DIST;
97    // AZM - AZI
98    /**
99     * Angle of incidence measured with respect to downward vertical.
100    */
101   int AIN;
102   /**
103    * PRMK from input data.
104    */
105   String PRMK;
106   /**
107    * Hour of arrival time from input data.
108    */
109   int HR;
110   /**
111    * Minute of arraival time from inout data.
112    */
113   int MN;
114   /**
115    * The second's portial of P-arrival time from input data.
116    */
117   double PSEC;
118   /**
119    * Observed P-travet time in sec.
120    */
121   double TPOBS;
122   /**
123    * Calculated travel time in sec/
124    */
125   double TPCAL;
126   /**
127    * If the Station Delay Model is used, then DLY means the station delay in
128    * sec from the input list. If the Variable First-Layer Model is used, then
129    * H1 means the thickness of the first layer in km at this station.
130    */
131   double DLYH1;
132   /**
133    * Residual of P-arrival in seconds.  Or, if S-P interval data is used, then
134    * S-P residual in seconds.
135    */
136   double PRES;
137   /**
138    * Weight used in hypocenter solution for P-arrival (or S-P interval data).
139    */
140   double PWT;
141   /**
142    * Maximum amplitude in mm from input data.
143    */
144   double AMX;
145   /**
146    * Period of maximum amplitude in seconds from imput data.
147    */
148   double PRX;
149   /**
150    * Calibration in mm used in computing XMAG.
151    */
152   double CALX;
153   /**
154    * System number for the station from input data.
155    */
156   int K;
157   /**
158    * Maximum amplitude magnitude computed from AMX, PRX, CALX, and K.
159    */
160   double XMAG;
161   /**
162    * Remark from input data.
163    */
164   double RMK;
165   /**
166    * F-P in seconds from input data.
167    */
168   double FMP;
169   /**
170    * F-P magnitude computed from F-P and DIST.
171    */
172   double FMAG;
173   /**
174    * SRMK from input data.
175    */
176   String SRMK;
177   /**
178    * The second's portion of S-arrival time from input data.
179    */
180   double SSEC;
181   /**
182    * Observed S-travel time (or S-P interval) in sec.
183    */
184   double TSOBS;
185   /**
186    * Residual of S-arrival in sec. Same as PRES if S-P interval
187    * data are used.
188    */
189   double SRES;
190   /**
191    * Weight used in hypocenter solution for S-arrival.
192    * N/A if S-P itnerval data is used.
193    */
194   double SWT;
195   /**
196    * Station time correction in seconds from input data.
197    */
198   String DT;
