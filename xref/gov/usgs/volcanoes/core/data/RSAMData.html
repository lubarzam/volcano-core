79    /**
80     * Get initialized axis to use with histogram graph
81     * @param bin histogram section period
82     * @return initialized axis
83     */
84    protected IAxis getHistogramAxis(BinSize bin) {
85      double startTime = getStartTime();
86      double endTime = getEndTime();
87      int bins = -1;
88      IAxis axis = null;
89  
90      if (bin == BinSize.MINUTE) {
91        startTime -= (startTime - 43200) % 60;
92        endTime -= (endTime - 43200) % 60 - 60;
93        bins = (int) (endTime - startTime) / 60;
94        if (bins > MAX_BINS)
95          bin = BinSize.HOUR;
96        else
97          axis = new FixedAxis(bins, startTime, endTime);
98      }
99      if (bin == BinSize.TENMINUTE) {
100       startTime -= (startTime - 43200) % 600;
101       endTime -= (endTime - 43200) % 600 - 600;
102       bins = (int) (endTime - startTime) / 600;
103       if (bins > MAX_BINS)
104         bin = BinSize.HOUR;
105       else
106         axis = new FixedAxis(bins, startTime, endTime);
107     }
108     if (bin == BinSize.HOUR) {
109       startTime -= (startTime - 43200) % 3600;
110       endTime -= (endTime - 43200) % 3600 - 3600;
111       bins = (int) (endTime - startTime) / 3600;
112       if (bins > MAX_BINS)
113         bin = BinSize.DAY;
114       else
115         axis = new FixedAxis(bins, startTime, endTime);
116     }
117     if (bin == BinSize.DAY) {
118       startTime -= (startTime - 43200) % 86400;
119       endTime -= (endTime - 43200) % 86400 - 86400;
120       bins = (int) (endTime - startTime) / 86400;
121       if (bins > MAX_BINS)
122         bin = BinSize.WEEK;
123       else
124         axis = new FixedAxis(bins, startTime, endTime);
125     }
126     if (bin == BinSize.WEEK) {
127       startTime -= (startTime - 43200) % 604800;
128       endTime -= (endTime - 43200) % 604800 - 604800;
129       bins = (int) (endTime - startTime) / 604800;
130       if (bins > MAX_BINS)
131         bin = BinSize.MONTH;
132       else
133         axis = new FixedAxis(bins, startTime, endTime);
134     }
135     if (bin == BinSize.MONTH) {
136       Date ds = J2kSec.asDate(startTime);
137       Date de = J2kSec.asDate(endTime);
138       bins = Time.getMonthsBetween(ds, de) + 1;
139       if (bins <= MAX_BINS) {
140         Calendar cal = Calendar.getInstance();
141         cal.setTime(ds);
142         cal.set(Calendar.DAY_OF_MONTH, 1);
143         cal.set(Calendar.HOUR_OF_DAY, 0);
144         cal.set(Calendar.MINUTE, 0);
145         cal.set(Calendar.SECOND, 0);
146         cal.set(Calendar.MILLISECOND, 0);
147         double[] edges = new double[bins + 1];
148         for (int i = 0; i < bins + 1; i++) {
149           edges[i] = J2kSec.fromDate(cal.getTime());
150           cal.add(Calendar.MONTH, 1);
151         }
152         axis = new VariableAxis(edges);
153       } else
154         bin = BinSize.YEAR;
155     }
156     if (bin == BinSize.YEAR) {
157       Date ds = J2kSec.asDate(startTime);
158       Date de = J2kSec.asDate(endTime);
159       bins = Time.getYear(de) - Time.getYear(ds) + 1;
160       double edges[] = new double[bins + 1];
161       Calendar cal = Calendar.getInstance();
162       cal.setTime(ds);
163       cal.set(Calendar.MONTH, 1);
164       cal.set(Calendar.DAY_OF_MONTH, 1);
165       cal.set(Calendar.HOUR_OF_DAY, 0);
166       cal.set(Calendar.MINUTE, 0);
167       cal.set(Calendar.SECOND, 0);
168       cal.set(Calendar.MILLISECOND, 0);
169       for (int i = 0; i < bins + 1; i++) {
170         edges[i] = J2kSec.fromDate(cal.getTime());
171         cal.add(Calendar.YEAR, 1);
172       }
173       axis = new VariableAxis(edges);
174     }
175     return axis;
176   }

178   /**
179    * Loops for data matrix, scans columns and fills events matrix
180    * @param threshold event treshold
181    * @param ratio minimum ratio between new and older value to define event
182    * @param maxLength maximum event length (ms)
183    */
184   public void countEvents(double threshold, double ratio, double maxLength) {
185     double oldValue = 0;
186     double olderValue = 0;
187     boolean eventOngoing = false;
188     double eventStart = 0;
189     int eventCount = 0;
190     double[] eventTimes = new double[data.rows()];
191 
192     for (int i = 1; i < data.rows(); i++) {
193       double currentTime = data.get(i, 0);
194       double currentValue = data.get(i, 1);
195 
196       if (currentValue >= threshold && currentValue >= olderValue * ratio) {
197         if (currentTime - eventStart > maxLength) {
198           eventOngoing = false;
199           eventStart = currentTime;
200         } else if (eventOngoing == false) {
201           eventStart = currentTime;
202           eventOngoing = true;
203           eventTimes[eventCount++] = currentTime;
204         }
205       } else {
206         eventOngoing = false;
207         eventStart = currentTime;
208       }
209 
210       olderValue = oldValue;
211       oldValue = currentValue;
212     }
213 
214     if (data.rows() == 1 && Double.isNaN(data.getQuick(0, 0))) {
215       events = DoubleFactory2D.dense.make(0, 2);
216 
217     } else {
218 
219       events = DoubleFactory2D.dense.make(eventCount + 2, 2);
220       events.setQuick(0, 0, data.get(0, 0));
221       events.setQuick(0, 1, 0);
222 
223       int c = 1;
224       for (int i = 1; i <= eventCount; i++) {
225         events.setQuick(i, 0, eventTimes[i - 1]);
226         events.setQuick(i, 1, c++);
227       }
228 
229       events.setQuick(eventCount + 1, 0, data.get(data.rows() - 1, 0));
230       events.setQuick(eventCount + 1, 1, eventCount);
231     }
232   }

234   /**
235    * Get ratio of this data value and given RSAMData data value
236    * on the time interval where they are intersecting.
237    * @return RSAMData
238    */
239   public RSAMData getRatSAM(RSAMData d) {
240     DoubleMatrix2D other = d.getData();
241     ArrayList<double[]> ratList = new ArrayList<double[]>();
242 
243     // if either channels data is null, then make an empty list
244     if ((data.rows() == 1 && Double.isNaN(data.getQuick(0, 0)))
245         || (other.rows() == 1 && Double.isNaN(other.getQuick(0, 0)))) {
246       double[] pt = new double[2];
247       pt[0] = Double.NaN;
248       pt[1] = Double.NaN;
249       ratList.add(pt);
250 
251     } else {
252 
253       int i = 0;
254       int j = 0;
255       while (i < rows() && j < other.rows()) {
256         double t1 = data.getQuick(i, 0);
257         double t2 = other.getQuick(j, 0);
258         if (t1 < t2)
259           i++;
260         else if (t1 > t2)
261           j++;
262         else {
263           try {
264             double[] pt = new double[2];
265             pt[0] = t1;
266             pt[1] = data.getQuick(i++, 1) / other.getQuick(j++, 1);
267             ratList.add(pt);
268           } catch (ArithmeticException e) {
269           }
270         }
271       }
272     }
273 
274     return new RSAMData(ratList);
275   }

285   /**
286    * Dump cumulative data as CSV string
287    * @return string representation in CSV
288    */
289   public String getCountsCSV() {
290     StringBuffer sb = new StringBuffer();
291     for (int i = 0; i < events.rows(); i++) {
292       sb.append(J2kSec.toDateString(events.getQuick(i, 0)) + ",");
293       for (int j = 1; j < events.columns(); j++) {
294         sb.append(events.getQuick(i, j));
295         sb.append(",");
296       }
297       sb.append("\n");
298     }
299     return sb.toString();
300   }

318   public boolean overlaps(RSAMData rsamData) {
319     if (period != rsamData.period)
320       return false;
321 
322     if (getEndTime() < rsamData.getStartTime())
323       return false;
324 
325     if (getStartTime() > rsamData.getEndTime())
326       return false;
327 
328     return true;
329   }

331   public RSAMData combine(RSAMData rsam) {
332     // other wave dominates this wave
333     if (getStartTime() >= rsam.getStartTime() && getEndTime() <= rsam.getEndTime())
334       return rsam;
335 
336     // this wave dominates other wave
337     if (getStartTime() <= rsam.getStartTime() && getEndTime() >= rsam.getEndTime())
338       return this;
339 
340     // this wave is left of other wave
341     if (getStartTime() <= rsam.getStartTime()) {
342       // logger.fine("rows before: " + rows());
343       DoubleMatrix2D[][] ms = new DoubleMatrix2D[2][1];
344       ms[0][0] = data;
345       int i = rsam.findClosestTimeIndexGreaterThan(getEndTime());
346       ms[1][0] = rsam.getData().viewPart(i, 0, rsam.rows() - i, 2);
347       data = DoubleFactory2D.dense.compose(ms);
348       // System.out.println("combine l: " + data.rows() + " " + data.columns());
349       return this;
350     }
351 
352     // this wave is right of other wave
353     if (rsam.getStartTime() <= getStartTime()) {
354       // logger.fine("rows before: " + rows());
355       DoubleMatrix2D[][] ms = new DoubleMatrix2D[2][1];
356       ms[0][0] = rsam.getData();
357       int i = findClosestTimeIndexLessThan(rsam.getEndTime());
358       if (i == -1)
359         i = 0;
360       ms[1][0] = getData().viewPart(i, 0, rows() - i, 2);
361       data = DoubleFactory2D.dense.compose(ms);
362       // logger.fine("combine r: " + data.rows() + " " + data.columns());
363       return this;
364     }
365 
366     // logger.fine("unknown case");
367     return null;
368   }

371   /**
372    * Yield index of datum w/ smallest time >= time
373    * @param time lower bound of times to consider
374    * @return index of time found; -1 if none found
375    */
376   public int findClosestTimeIndexGreaterThan(double time) {
377     for (int i = 0; i < rows(); i++) {
378       if (data.getQuick(i, 0) >= time)
379         return i;
380     }
381     return -1;
382   }
383 
384   /**
385    * Yield index of datum w/ largest time <= time
386    * @param time upper bound of times to consider
387    * @return index of time found; -1 if none found
388    */
389   public int findClosestTimeIndexLessThan(double time) {
390     for (int i = rows() - 1; i >= 0; i--) {
391       if (data.getQuick(i, 0) <= time)
392         return i;
393     }
394     return -1;
395   }
