7   package gov.usgs.volcanoes.core.data;
8   
9   import gov.usgs.volcanoes.core.contrib.HashCodeUtil;
10  import gov.usgs.volcanoes.core.util.UtilException;
11  
12  import java.util.regex.Pattern;
13  
14  /**
15   * An immutable class for holding SCNL data.
16   *
17   * @author Tom Parker
18   */
19  public class Scnl implements Comparable<Scnl> {
20  
21    /** the default location code. */
22    public static final String DEFAULT_LOCATION = "--";
23  
24    /** delimiter used in string representations of a SCNL. */
25    public static final String DELIMITER = "$";
26  
27    /** my channel code. */
28    public final String channel;
29  
30    /** my location code. */
31    public final String location;
32  
33    /** my network code. */
34    public final String network;
35  
36    /** my station code. */
37    public final String station;
38  
39    /**
40     * Construct a SCNL object with a default location field.
41     *
42     * @param station station
43     * @param channel channel
44     * @param network network
45     */
46    public Scnl(String station, String channel, String network) {
47      this(station, channel, network, DEFAULT_LOCATION);
48    }
49  
50    /**
51     * Construct a SCNL object.
52     *
53     * @param station station
54     * @param channel channel
55     * @param network network
56     * @param location location
57     */
58    public Scnl(String station, String channel, String network, String location) {
59      this.station = station;
60      this.channel = channel;
61      this.network = network;
62  
63      if ("".equals(location.trim())) {
64        this.location = DEFAULT_LOCATION;
65      } else {
66        this.location = location;
67      }
68    }
69  
70    /**
71     * Compare this SCNL to another. Sorted by: network, station, channel, location.
72     * 
73     * @param scnl the object to compare
74     * @return an int following compareTo conventions
75     */
76    public int compareTo(Scnl scnl) {
77      final String me = String.format("%s$%s$%s$%s", network, station, channel, location);
78      final String other =
79          String.format("%s$%s$%s$%s", scnl.network, scnl.station, scnl.channel, scnl.location);
80  
81      return (me.compareTo(other));
82    }
83  
84    @Override
85    public boolean equals(Object obj) {
86      if (obj == null) {
87        return false;
88      }
89      if (getClass() != obj.getClass()) {
90        return false;
91      }
92  
93      final Scnl other = (Scnl) obj;
94  
95      if ((station == null) ? (other.station != null) : !station.equals(other.station)) {
96        return false;
97      }
98      if ((channel == null) ? (other.channel != null) : !channel.equals(other.channel)) {
99        return false;
100     }
101     if ((network == null) ? (other.network != null) : !network.equals(other.network)) {
102       return false;
103     }
104     if ((location == null) ? (other.location != null) : !location.equals(other.location)) {
105       return false;
106     }
107     return true;
108   }
109 
110 
111   @Override
112   public int hashCode() {
113     int hash = HashCodeUtil.SEED;
114 
115     hash = HashCodeUtil.hash(hash, station);
116     hash = HashCodeUtil.hash(hash, channel);
117     hash = HashCodeUtil.hash(hash, network);
118     hash = HashCodeUtil.hash(hash, location);
119 
120     return hash;
121   }
122 
123   @Override
124   public String toString() {
125     return toString(DELIMITER);
126   }
127 
128   /**
129    * Return a string representation of the SCNL using the provided delimiter.
130    * 
131    * @param delimiter The string to use to seperate SCNL components
132    * @return A string represenetation of this SCNL
133    */
134   public String toString(String delimiter) {
135     final StringBuffer sb = new StringBuffer();
136     sb.append(station).append(delimiter).append(channel).append(delimiter).append(network)
137         .append(delimiter).append(location);
138     return sb.toString();
139   }
140 
141   /**
142    * Parse a Scnl string.
143    * 
144    * @param scnlString standard delimited string
145    * @return Scnl object
146    * @throws UtilException when string cannot be parsed
147    */
148   public static Scnl parse(String scnlString) throws UtilException {
149     return Scnl.parse(scnlString, DELIMITER);
150   }
151 
152   /**
153    * Parse a Scnl string.
154    * 
155    * @param scnlString delimited string
156    * @param delimiter String passed to String.split()
157    * @return Scnl object
158    * @throws UtilException when string cannot be parsed
159    */
160   public static Scnl parse(String scnlString, String delimiter) throws UtilException {
161     String[] parts = scnlString.split(Pattern.quote(delimiter));
162     if (parts.length > 3) {
163       return new Scnl(parts[0], parts[1], parts[2], parts[3]);
164     } else if (parts.length > 2) {
165       return new Scnl(parts[0], parts[1], parts[2]);
166     } else {
167       throw new UtilException("Cannot parse '" + scnlString + "'");
168     }
169   }
