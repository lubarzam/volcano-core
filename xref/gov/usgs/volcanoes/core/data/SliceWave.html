5   /**

105   /**
106    * Get wave series end time.
107    * 
108    * @return end time

110   private double getTrueEndTime() {
111     return getStartTime() + source.buffer.length * (1 / getSamplingRate());

114   /**
115    * Set time limits to define slice zone, compute statistics.
116    * 
117    * @param t1 start time
118    * @param t2 end time
119    */
120   public void setSlice(double t1, double t2) {
121     if (t1 < source.getStartTime() || t2 > getTrueEndTime() || t1 >= t2) {
122       return;
123     }
124 
125     invalidateStatistics();
126 
127     position = (int) Math.round((t1 - source.getStartTime()) * getSamplingRate());
128     limit = position + (int) Math.round((t2 - t1) * getSamplingRate());
129 
130     if (limit > source.buffer.length) {
131       limit = source.buffer.length;
132     }
133   }
134 
135   /**
136    * Invalidates the cached statistics.
137    */
138   public void invalidateStatistics() {
139     mean = Double.NaN;
140     rsam = Double.NaN;
141     max = -1E300;
142     min = 1E300;
143   }

145   /**
146    * Compute slice statistics.
147    */
148   private void deriveStatistics() {
149     if (source.buffer == null || source.buffer.length == 0) {
150       mean = 0;
151       rsam = 0;
152       max = 0;
153       min = 0;
154       return;
155     }
156     int noDatas = 0;
157     long sum = 0;
158     long rs = 0;
159     for (int i = position; i < limit; i++) {
160       int d = source.buffer[i];
161       if (d != Wave.NO_DATA) {
162         sum += d;
163         rs += Math.abs(d);
164         min = Math.min(min, d);
165         max = Math.max(max, d);
166       } else {
167         noDatas++;
168       }
169     }
170 
171     mean = (double) sum / (double) (samples() - noDatas);
172     rsam = (double) rs / (double) (samples() - noDatas);
173     dataRange = new double[] {min, max};
174 
175     double xm = (limit - position + 1) / 2;
176     double ssxx = 0;
177     double ssxy = 0;
178     for (int i = position; i < limit; i++) {
179       if (source.buffer[i] == Wave.NO_DATA) {
180         continue;
181       }
182       ssxy += (i - xm) * (source.buffer[i] - mean);
183       ssxx += (i - xm) * (i - xm);
184     }
185     m = ssxy / ssxx;
186     b = mean - m * xm;
187   }

241   /**
242    * Determines if all of the samples are NO_DATA samples.
243    * 
244    * @return whether or not this consists entirely of NO_DATA samples
245    */
246   public boolean isData() {
247     for (int i = 0; i < source.buffer.length; i++) {
248       if (source.buffer[i] != Wave.NO_DATA) {
249         return true;
250       }
251     }
252 
253     return false;
254   }

256   /**
257    * Returns an array containing the signal from the slice starting position to the slice ending
258    * position. Detrend data before returning.
259    * 
260    * @return the signal array
261    */
262   public double[] getSignal() {
263 
264     double[] signal;
265 
266     int nsamples = samples();
267 
268     signal = new double[nsamples];
269 
270     reset();
271     for (int i = 0; i < nsamples; i++) {
272       signal[i] = next();
273       if (signal[i] == Wave.NO_DATA) {
274         signal[i] = 0;
275       } else {
276         signal[i] -= (i + 1) * m + b;
277       }
278     }
279 
280     return signal;
281 
282   }

291   /**
292    * Check if read pointer in the slice zone.
293    * 
294    * @return true if more to read
295    */
296   public boolean hasNext() {
297     return readPosition < limit;
298   }
299 
300   /**
301    * Move read pointer in the next position.
302    * 
303    * @return data value in the read pointer position before moving
304    */
305   public double next() {
306     if (hasNext()) {
307       return source.buffer[readPosition++];
308     } else {
309       return Double.NaN;
310     }

318   /**
319    * Dump slice content to CSV string.
320    * 
321    * @return CSV string
322    */
323   public String toCsv() {
324     if (source.buffer == null || source.buffer.length == 0) {
325       return "";
326     }
327 
328     StringBuffer sb = new StringBuffer();
329     double time = getStartTime();
330 
331     for (int i = position; i < limit; i++) {
332       sb.append(J2kSec.toDateString(time) + ',' + source.buffer[i] + '\n');
333       time += (1 / getSamplingRate());
334     }
335 
336     return sb.toString();
337   }
