52    /**
53     * Complete constructor
54     * 
55     * @param s
56     *            Signal
57     * @param sr
58     *            Sampling Rate
59     * @param nf
60     *            FFT length
61     * @param bs
62     *            Bin size (in samples)
63     * @param ol
64     *            Overlap (in samples)
65     * @param b
66     *            Beta value for Kaiser window
67     */
68  
69    public Spectrogram(double[] s, int sr, int nf, int bs, int ol, double b) {
70  
71      signal = s;
72      samplingRate = sr;
73      nfft = nf;
74      binSize = Math.min(bs, s.length);
75      overlap = Math.min(ol, s.length - 1);
76      beta = b;
77  
78      nRows = nfft / 2 + 1;
79      nColumns = (int) Math
80          .floor(((double) getNSamples() - (double) overlap) / ((double) binSize - (double) overlap));
81  
82      // System.out.printf("Spectrogram info:\n");
83      // System.out.printf("\tNumber of samples: %d\n", getNSamples());
84      // System.out.printf("\tSampling Rate: %d\n", samplingRate);
85      // System.out.printf("\tN FFT: %d\n", nfft);
86      // System.out.printf("\tBin Size: %d\n", binSize);
87      // System.out.printf("\tOverlap: %d\n", overlap);
88      // System.out.printf("\tBeta: %f\n", beta);
89      // System.out.printf("\tN rows: %d\n", nRows);
90      // System.out.printf("\tN columns: %d\n", nColumns);
91  
92      frequency = computeFrequency();
93      time = computeTime();
94      window = Util.kaiser(binSize, beta);
95  
96      spectraAmplitude = computeSpectraAmplitude();
97  
98    }

206   /**

209   public double getMinSpectraAmplitude() {
210 
211     double MIN = Double.MAX_VALUE;
212     for (int i = 0; i < nRows; i++)
213       for (int j = 0; j < nColumns; j++)

253   /**
254    * Returns an array of log10 of the scaled spectra amplitudes times the
255    * specified multiplier.
256    * 
257    * @param multiplier
258    *            Multiplier
259    */
260   public double[][] getLogSpectraAmplitude(double multiplier, double reference_amplitude) {
261 
262     double[][] logAmp = new double[nRows][nColumns];
263     for (int i = 0; i < nRows; i++)
264       for (int j = 0; j < nColumns; j++)
265         logAmp[i][j] = multiplier * Math.log10(spectraAmplitude[i][j] / reference_amplitude);
266 
267     return logAmp;
268   }

270   /**
271    * Computes the frequency array
272    */
273   private double[] computeFrequency() {
274 
275     double[] omega = new double[nRows];
276     double delta = (double) samplingRate / (double) (nfft);
277     for (int i = 0; i < omega.length; i++)
278       omega[i] = i * delta;
279     return omega;
280   }
281 
282   /**
283    * Computes the time array
284    */
285   private double[] computeTime() {
286 
287     double[] T = new double[nColumns];
288     double delta = ((double) binSize - (double) overlap) / (double) samplingRate;
289     double alpha = (double) binSize / samplingRate / 2;
290     for (int i = 0; i < T.length; i++)
291       T[i] = i * delta + alpha;
292     return T;
293   }
294 
295   /**
296    * Computes the spectra amplitudes with the FFT.
297    */
298   private double[][] computeSpectraAmplitude() {
299 
300     double[][] specAmp = new double[nRows][nColumns];
301     double[] bin = new double[nfft];
302 
303     int c = 0;
304     for (int i = 0; i < nColumns; i++) {
305 
306       for (int j = 0; j < binSize; j++) {
307         bin[j] = signal[c] * window[j];
308         c++;
309       }
310       c = c - overlap;
311 
312       DoubleFFT_1D transform = new DoubleFFT_1D(nfft);
313       transform.realForward(bin);
314 
315       specAmp[0][i] = Math.abs(bin[0]);
316       specAmp[nfft / 2][i] = Math.abs(bin[1]);
317       for (int j = 2; j < nfft; j = j + 2) {
318         // TODO: figure out why the next line throws an
319         // IndexOutOfBoundsExteption when nfft is set to 0 and binSize
320         // is 117
321         specAmp[j / 2][i] = Math.sqrt(bin[j] * bin[j] + bin[j + 1] * bin[j + 1]);
322         bin[j] = 0;
323         bin[j + 1] = 0;
324       }
325 
326     }
327     return specAmp;
328 
329   }
