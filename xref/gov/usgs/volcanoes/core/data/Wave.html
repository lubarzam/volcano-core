143   /**
144    * Registers the starting time to the nearest even interval based on the wave's sampling rate. The
145    * offset between the original start time and the registered time is stored in
146    * <code>registrationOffset</code>.

148   public void register() {
149     double dif = startTime % getSamplingPeriod();
150     if (dif >= getSamplingPeriod() / 2) {
151       registrationOffset = (getSamplingPeriod() - dif);
152     } else {
153       registrationOffset = -dif;
154     }
155 
156     startTime += registrationOffset;
157   }

243   /**
244    * Computes the FFT of the <code>Wave</code>. This function will perform the necessary
245    * zero-padding in order to make the samples buffer be a power of 2 in size.
246    * 
247    * @return the FFT (n rows x 2 column [real/imaginary]) array
248    * @see FFT
249    */
250   public double[][] fft() {
251     int n = buffer.length;
252     int p2 = (int) Math.ceil(Math.log((double) n) / Math.log(2));
253     int newSize = (int) Math.pow(2, p2);
254     double[][] buf = new double[newSize][2];
255     int m = (int) Math.round(mean());
256 
257     for (int i = 0; i < n; i++) {
258       if (buffer[i] != NO_DATA) {
259         buf[i][0] = buffer[i];
260       }
261     }
262 
263     for (int i = n; i < newSize; i++) {
264       buf[i][0] = m;
265     }
266 
267     FFT.fft(buf);
268 
269     return buf;
270   }

283   /**
284    * Scan data and compute statistic.
285    */
286   private void deriveStatistics() {
287     if (buffer == null || buffer.length == 0) {
288       mean = 0;
289       rsam = 0;
290       max = 0;
291       min = 0;
292       first = 0;
293       return;
294     }
295     int noDatas = 0;
296     long sum = 0;
297     long rs = 0;
298     boolean firstSet = false;
299     for (int i = 0; i < buffer.length; i++) {
300       int d = buffer[i];
301       if (d != NO_DATA) {
302         sum += d;
303         rs += Math.abs(d);
304         min = Math.min(min, d);
305         max = Math.max(max, d);
306         if (!firstSet) {
307           first = d;
308           firstSet = true;
309         }
310       } else {
311         noDatas++;
312       }
313     }
314 
315     mean = (double) sum / (double) (buffer.length - noDatas);
316     rsam = (double) rs / (double) (buffer.length - noDatas);
317     dataRange = new int[] {min, max};
318   }

523    * @param wave the test wave
524    * @return whether or not these waves overlap

526   public boolean overlaps(Wave wave) {
527     // obviously this could be compressed to one line, but this is readable:
528     // either the new wave is completely right of, completely left of or
529     // overlapping the old wave
530     if (getEndTime() <= wave.getStartTime()) {
531       return false;
532     }
533 
534     if (getStartTime() >= wave.getEndTime()) {
535       return false;
536     }
537 
538     return true;
539   }

581     System.arraycopy(buffer, length1, sw2.buffer, 0, length2);
582     sw2.dataType = dataType;
583 
584     return new Wave[] {sw1, sw2};

587   /**
588    * Splits this wave into a list of smaller waves with a maximum number of samples as specified.
589    * Used for breaking a wave into TRACEBUF-sized pieces.
590    * 
591    * @param maxSamples maximum number of samples
592    * @return a list of the split up waves.
593    */
594   public List<Wave> split(int maxSamples) {
595     ArrayList<Wave> list = new ArrayList<Wave>(buffer.length / maxSamples + 2);
596     double ct = startTime;
597     int j = 0;
598     while (j < buffer.length) {
599       Wave sw = new Wave();
600       sw.startTime = ct;
601       sw.samplingRate = samplingRate;
602       sw.dataType = dataType;
603       int numSamples = Math.min(maxSamples, buffer.length - j);
604       sw.buffer = new int[numSamples];
605       System.arraycopy(buffer, j, sw.buffer, 0, numSamples);
606       ct += (double) numSamples * (getSamplingPeriod());
607       j += numSamples;
608       list.add(sw);
609     }
610     return list;
611   }

613   /**
614    * Gets a strict subset of this wave. This function should have more bounds checking. It will fail
615    * if you ask for a portion of wave that is not contained in the original wave.
616    * 
617    * @param t1 the start time of the subset
618    * @param t2 the end time of the subset
619    * @return the subset wave
620    */
621   public Wave subset(double t1, double t2) {
622     if (t1 < getStartTime() || t2 > getEndTime() || t2 < t1) {
623       return this;
624     }
625 
626     if (buffer == null || buffer.length < 1) {
627       return this;
628     }
629 
630     Wave sw = new Wave();
631     sw.startTime = t1;
632     sw.samplingRate = samplingRate;
633     sw.dataType = dataType;
634     sw.registrationOffset = registrationOffset;
635     int samples = (int) Math.floor((t2 - t1) * samplingRate);
636     sw.buffer = new int[samples];
637     int offset = (int) Math.floor((t1 - startTime) * samplingRate);
638     System.arraycopy(buffer, offset, sw.buffer, 0, samples);
639     return sw;
640   }

642   /**
643    * Expiremental. For use with Winston static importers. TODO: handle erase a chunk inside the
644    * existing wave
645    * 
646    * @param t1 start time
647    * @param t2 end time
648    */
649   public void erase(double t1, double t2) {
650     if (t2 < getStartTime() || t1 > getEndTime()) {
651       return; // nothing to erase
652     }
653 
654     if (t1 >= getStartTime() && t2 <= getEndTime()) {
655       return; // erase the middle -- unhandled
656     }
657 
658     if (t1 <= getStartTime() && t2 >= getEndTime()) {
659       // erase the whole wave
660       buffer = null;
661       startTime = Double.NaN;
662       samplingRate = Double.NaN;
663       dataType = null;
664     }
665 
666     if (t2 > getStartTime() && t2 <= getEndTime()) {
667       // erase left side
668       int size = (int) Math.round((t2 - getStartTime()) * samplingRate);
669       int[] buf = new int[buffer.length - size];
670       System.arraycopy(buffer, size, buf, 0, buffer.length - size);
671       buffer = buf;
672       startTime = t2;
673     }
674 
675     if (t1 >= getStartTime() && t1 <= getEndTime()) {
676       // erase right side
677       int size = (int) Math.round((getEndTime() - t1) * samplingRate);
678       int[] buf = new int[buffer.length - size];
679       // System.out.println("new size: " + buf.length);
680       System.arraycopy(buffer, 0, buf, 0, buffer.length - size);
681       buffer = buf;
682     }
683   }

685   /**
686    * Combines this wave with another overlapping one. This function will return a combination of the
687    * two waves. This may modify this wave before returning it. Neither wave is guaranteed to be safe
688    * from modification.
689    * 
690    * <p>This is only intended to work on overlapping waves (as checked by the <code>overlap()</code>
691    * function.
692    * 
693    * @param wave wave object
694    * @return combined wave
695    */
696   public Wave combine(Wave wave) {
697     if (wave == null || wave.numSamples() < 1) {
698       return this;
699     }
700 
701     if (numSamples() < 1) {
702       return wave;
703     }
704 
705     if (samplingRate != wave.getSamplingRate() || !(adjacent(wave) || overlaps(wave))) {
706       return null;
707     }
708 
709     // other wave dominates this wave
710     if (startTime >= wave.getStartTime() && getEndTime() <= wave.getEndTime()) {
711       return wave;
712     }
713 
714     // this wave dominates other wave
715     if (startTime <= wave.getStartTime() && getEndTime() >= wave.getEndTime()) {
716       return this;
717     }
718 
719     double len =
720         Math.max(getEndTime(), wave.getEndTime()) - Math.min(startTime, wave.getStartTime());
721     len *= samplingRate;
722 
723     int newLength = (int) Math.ceil(len);
724     int totalSamples = wave.numSamples() + this.numSamples();
725     newLength = Math.min(newLength, totalSamples);
726     int[] newbuf = new int[newLength];
727 
728     Wave leftWave;
729     Wave rightWave;
730 
731     // this wave is left of other wave
732     if (startTime <= wave.getStartTime()) {
733       leftWave = this;
734       rightWave = wave;
735       // this wave is right of other wave
736     } else {
737       leftWave = wave;
738       rightWave = this;
739     }
740 
741     System.arraycopy(leftWave.buffer, 0, newbuf, 0, leftWave.buffer.length);
742 
743     int i = newbuf.length - rightWave.buffer.length;
744 
745     System.arraycopy(rightWave.buffer, 0, newbuf, i, rightWave.buffer.length);
746 
747     this.buffer = newbuf;
748     this.startTime = leftWave.startTime;
749 
750     return this;
751   }

753   /**
754    * Joins together a list of waves into one large wave. The list must be sorted in time-ascending
755    * order of start time. Technically, as long as the first wave is the earliest and the last wave
756    * ends the latest, this function will work. This function will happily create a wave with large
757    * gaps in it as well.
758    * 
759    * @param waves the list of <code>Wave</code> s
760    * @return the new joined wave
761    */
762   public static Wave join(List<Wave> waves) {
763     if (waves == null || waves.size() == 0) {
764       return null;
765     }
766 
767     if (waves.size() == 1) {
768       return waves.get(0);
769     }
770 
771     double mint = 1E300;
772     double maxt = -1E300;
773     double sr = -1;
774     for (Wave sw : waves) {
775       mint = Math.min(mint, sw.getStartTime());
776       maxt = Math.max(maxt, sw.getEndTime());
777       sr = sw.getSamplingRate();
778     }
779 
780     int samples = (int) ((maxt - mint) * sr);
781 
782     int[] buffer = new int[samples + 1];
783     Arrays.fill(buffer, NO_DATA);
784 
785     for (Wave sw : waves) {
786       int i = (int) Math.round((sw.getStartTime() - mint) * sr);
787       System.arraycopy(sw.buffer, 0, buffer, i, sw.buffer.length);
788     }
789     return new Wave(buffer, mint, sr);
790   }

807     // return waves.get(0);

809     Wave wv0 = waves.get(0);

811     int samples = (int) ((t2 - t1) * wv0.getSamplingRate());

818       System.arraycopy(sw.buffer, 0, buffer, i, sw.buffer.length);
819     }

841    * Integrates the wave. This functions takes the cumulative sum (mean removed) of the wave at each
842    * sample, dividing by the sampling rate.
843    * 
844    * @return an array of the sum at each sample
845    */
846   public double[] integrate() {
847     double[] d = new double[this.buffer.length];
848     double period = getSamplingPeriod();
849     double mean = mean();
850     double sum = 0;
851     for (int i = 0; i < this.buffer.length; i++) {
852       sum += ((double) buffer[i] - mean);
853       d[i] = sum * period;
854     }

875    * Filter data.
876    * 
877    * @param bw Butterworth filter to apply
878    * @param zeroPhaseShift flag for no phase shift
879    */
880   public void filter(Butterworth bw, boolean zeroPhaseShift) {
881     double mean = mean();
882 
883     double[] dbuf = new double[buffer.length + (int) (buffer.length * 0.5)];
884     Arrays.fill(dbuf, mean);
885     int trueStart = (int) (buffer.length * 0.25);
886     for (int i = 0; i < buffer.length; i++) {
887       if (buffer[i] != Wave.NO_DATA) {
888         dbuf[i + trueStart] = buffer[i];
889       }
890     }
891 
892     bw.setSamplingRate(getSamplingRate());
893     bw.create();
894     Filter.filter(dbuf, bw.getSize(), bw.getXCoeffs(), bw.getYCoeffs(), bw.getGain(), 0, 0);
895     if (zeroPhaseShift) {
896       double[] dbuf2 = new double[dbuf.length];
897       for (int i = 0, j = dbuf.length - 1; i < dbuf.length; i++, j--) {
898         dbuf2[j] = dbuf[i];
899       }
900 
901       Filter.filter(dbuf2, bw.getSize(), bw.getXCoeffs(), bw.getYCoeffs(), bw.getGain(), 0, 0);
902 
903       for (int i = 0, j = dbuf2.length - 1 - trueStart; i < buffer.length; i++, j--) {
904         buffer[i] = (int) Math.round(dbuf2[j]);
905       }
906     } else {
907       for (int i = 0; i < buffer.length; i++) {
908         buffer[i] = (int) Math.round(dbuf[i + trueStart]);
909       }
910     }
911     invalidateStatistics();
912   }

1018   /**
1019    * Detrend this Wave. Ignore NO_DATA samples
1020    */
1021   public void detrend() {
1022 
1023     double xm = buffer.length / 2;
1024     double ym = mean();
1025     double ssxx = 0;
1026     double ssxy = 0;
1027     for (int i = 0; i < buffer.length; i++) {
1028       if (buffer[i] != NO_DATA) {
1029         ssxy += (i - xm) * ((double) buffer[i] - ym);
1030         ssxx += (i - xm) * (i - xm);
1031       }
1032     }
1033 
1034     double m = ssxy / ssxx;
1035     double b = ym - m * xm;
1036     for (int i = 0; i < this.buffer.length; i++) {
1037       if (buffer[i] != NO_DATA) {
1038         buffer[i] -= ((double) buffer[i] * m + b);
1039       }
1040     }
1041   }

1052   /**
1053    * Replace data with rolling mean of period p.
1054    * 
1055    * @param p period used for rolling mean
1056    */
1057   public void set2mean(double p) {
1058     int j = 0; // index of oldest value in window
1059     double jtime = 0; // time of oldest value in window
1060     Meaner window = new Meaner();
1061     window.add((double) buffer[0]);
1062     int r = this.buffer.length;
1063     for (int i = 1; i < r; i++) {
1064       double itime = i / samplingRate;
1065       double ival = (double) buffer[i];
1066       window.add(ival);
1067       // While oldest value is outside period, remove it
1068       while (itime - jtime > p) {
1069         window.removeOldest();
1070         j++;
1071         jtime = j / samplingRate;
1072       }
1073       buffer[i] = (int) Math.round(window.avg());
1074     }
1075   }

1112   /**
1113    * Class to maintain a FIFO of doubles & report its mean.
1114    */
1115   private class Meaner {
1116     private LinkedList<Double> data; // the values
1117     private double sum; // their sum
1118 
1119     Meaner() {
1120       data = new LinkedList<Double>();
1121       sum = 0;
1122     }
1123 
1124     /**
1125      * Add val to the queue.
1126      * 
1127      * @param val value to add
1128      */
1129     public void add(double val) {
1130       data.addLast(val);
1131       sum += val;
1132     }
1133 
1134     /**
1135      * Remove the oldest value from the queue.
1136      * 
1137      * @return value removed
1138      */
1139     public double removeOldest() {
1140       Double datum = data.removeFirst();
1141       sum -= datum;
1142       return datum;
1143     }
1144 
1145     /**
1146      * Mean of values in queue.
1147      * 
1148      * @return the mean
1149      */
1150     public double avg() {
1151       return sum / data.size();
1152     }
1153   }

1155   /**
1156    * Class to maintain a FIFO of doubles & report its median.
1157    */
1158   private class Medianer {
1159 
1160     private class MultiMap {
1161       private TreeMap<Double, LinkedList<Integer>> mm;
1162       private TreeMap<Integer, Double> unmm;
1163 
1164       MultiMap() {
1165         mm = new TreeMap<Double, LinkedList<Integer>>();
1166         unmm = new TreeMap<Integer, Double>();
1167       }
1168 
1169       protected void put(Double key, Integer val) {
1170         LinkedList<Integer> entry = mm.get(key);
1171         if (entry == null) {
1172           entry = new LinkedList<Integer>();
1173         }
1174         unmm.put(val, key);
1175         if (entry.size() == 0) {
1176           entry.add(val);
1177           mm.put(key, entry);
1178           return;
1179         }
1180         if (val < entry.getFirst()) {
1181           entry.addFirst(val);
1182         } else {
1183           entry.addLast(val);
1184         }
1185       }
1186 
1187       protected Double lastKey() {
1188         return mm.lastKey();
1189       }
1190 
1191       protected Integer removeLastIndex(Double key) {
1192         LinkedList<Integer> entry = mm.get(key);
1193         // entry should not be null!
1194         Integer index = entry.removeLast();
1195         if (entry.size() == 0) {
1196           mm.remove(key);
1197         }
1198         unmm.remove(index);
1199         return index;
1200       }
1201 
1202       protected Double firstKey() {
1203         return mm.firstKey();
1204       }
1205 
1206       protected Integer removeFirstIndex(Double key) {
1207         LinkedList<Integer> entry = mm.get(key);
1208         // entry should not be null!
1209         Integer index = entry.removeFirst();
1210         if (entry.size() == 0) {
1211           mm.remove(key);
1212         }
1213         unmm.remove(index);
1214         return index;
1215       }
1216 
1217       public int size() {
1218         return unmm.size();
1219       }
1220 
1221       public void addLo(Double key, Integer val, MultiMap other) {
1222         if (other.size() > 0 && key >= other.firstKey()) {
1223           other.put(key, val);
1224           key = other.firstKey();
1225           val = other.removeFirstIndex(key);
1226         }
1227         put(key, val);
1228       }
1229 
1230       public void addHi(Double key, Integer val, MultiMap other) {
1231         if (other.size() > 0 && key <= other.lastKey()) {
1232           other.put(key, val);
1233           key = other.lastKey();
1234           val = other.removeLastIndex(key);
1235         }
1236         put(key, val);
1237       }
1238 
1239       public boolean removeIndex(Integer val) {
1240         Double key = unmm.remove(val);
1241         if (key == null) {
1242           return false;
1243         }
1244         List<Integer> entry = mm.get(key);
1245         if (entry.size() == 1) {
1246           mm.remove(key);
1247         } else {
1248           entry.remove(val);
1249         }
1250         return true;
1251       }
1252     }
1253 
1254     private MultiMap loHalf; // values in lower half
1255     private MultiMap hiHalf; // values in upper half
1256     private int idx1; // values window have indices idx1..idx2
1257     private int idx2;
1258 
1259     Medianer() {
1260       // Invariant: |loHalf| - |hiHalf| = 0 or 1
1261       // All keys in loHalf <= all keys in hiHalf
1262       loHalf = new MultiMap();
1263       hiHalf = new MultiMap();
1264       idx1 = 0;
1265       idx2 = -1;
1266     }
1267 
1268     /**
1269      * Add val to the queue.
1270      * 
1271      * @param val value to add
1272      */
1273     public void add(double val) {
1274       idx2++;
1275       if (loHalf.size() == hiHalf.size()) {
1276         loHalf.addLo(val, idx2, hiHalf);
1277       } else {
1278         hiHalf.addHi(val, idx2, loHalf);
1279       }
1280     }
1281 
1282     /**
1283      * Remove the oldest value from the queue.
1284      * 
1285      * @return value removed
1286      */
1287     public void removeOldest() {
1288       if (!loHalf.removeIndex(idx1)) {
1289         hiHalf.removeIndex(idx1);
1290       }
1291       idx1++;
1292       int loSize = loHalf.size();
1293       int hiSize = hiHalf.size();
1294       if (loSize < hiSize) {
1295         // Shift min of hi to lo
1296         Double d = hiHalf.firstKey();
1297         Integer ix = hiHalf.removeFirstIndex(d);
1298         loHalf.put(d, ix);
1299       } else if (loSize > hiSize + 1) {
1300         // Shift max of lo to hi
1301         Double d = loHalf.lastKey();
1302         Integer ix = loHalf.removeLastIndex(d);
1303         hiHalf.put(d, ix);
1304       }
1305     }
1306 
1307     /**
1308      * Median of values in queue.
1309      * 
1310      * @return the median
1311      */
1312     public double avg() {
1313       if (loHalf.size() == hiHalf.size()) {
1314         return (loHalf.lastKey() + hiHalf.firstKey()) / 2;
1315       }
1316       return loHalf.lastKey();
1317     }
1318   }
