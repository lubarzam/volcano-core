14  import edu.sc.seis.seisFile.sac.SacHeader;
15  import edu.sc.seis.seisFile.sac.SacTimeSeries;

17  /**
18   * A concrete SeismicDataFile class for SAC files.
19   * 
20   * @author Tom Parker
21   */
22  public class SacDataFile extends SeismicDataFile {
23  
24    private SacTimeSeries sac;
25    private SacHeader header;
26  
27    protected SacDataFile(String fileName) {
28      super(fileName, "SAC^");
29    }
30  
31    /**
32     * @see gov.usgs.plot.data.file.SeismicDataFile#read()
33     */
34    public void read() throws IOException {
35      sac = new SacTimeSeries(fileName);
36      header = sac.getHeader();
37      Wave sw = new Wave();
38      sw.setStartTime(J2kSec.fromDate(getStartTime()));
39      sw.setSamplingRate(getSamplingRate());
40      sw.buffer = new int[sac.getY().length];
41      for (int i = 0; i < sac.getY().length; i++) {
42        sw.buffer[i] = Math.round(sac.getY()[i]);
43      }
44      String channel = header.getKstnm().trim() + "$" + header.getKcmpnm().trim() + "$"
45          + header.getKnetwk().trim();
46  
47      String loc = header.getKhole().trim();
48      if (!(loc == null || loc.equals("  ") || loc.equals("--"))) {
49        channel += "$" + loc;
50      }
51      waves.put(channel, sw);
52    }
53  
54    private double getSamplingRate() {
55      return 1 / header.getDelta();
56    }
57  
58    private Date getStartTime() {
59      if (sac == null) {
60        return null;
61      }
62      String ds = header.getNzyear() + "," + header.getNzjday() + "," + header.getNzhour() + ","
63          + header.getNzmin() + "," + header.getNzsec() + "," + header.getNzmsec();
64      SimpleDateFormat format = new SimpleDateFormat("yyyy,DDD,HH,mm,ss,SSS");
65      format.setTimeZone(TimeZone.getTimeZone("GMT"));
66      Date d = null;
67      try {
68        d = format.parse(ds);
69      } catch (ParseException e) {
70        e.printStackTrace();
71      }
72      return d;
73  
74    }
75  
76    /**
77     * @see gov.usgs.plot.data.file.SeismicDataFile#write()
78     */
79    public void write() throws FileNotFoundException, IOException {
80      Wave wave = waves.values().iterator().next();
81      float[] y = new float[wave.buffer.length];
82      for (int i = 0; i < y.length; i++) {
83        y[i] = wave.buffer[i];
84      }
85  
86      SacTimeSeries sac = new SacTimeSeries(getSacHeader(), y);
87      sac.write(fileName);
88    }
89  
90    private SacHeader getSacHeader() {
91      SacHeader header = new SacHeader();
92      String channel = waves.keySet().iterator().next();
93  
94      String[] channelCmp = channel.split("[\\s\\$]");
95      String s = channelCmp.length > 0 ? channelCmp[0] : "";
96      String c = channelCmp.length > 1 ? channelCmp[1] : "";
97      String n = channelCmp.length > 2 ? channelCmp[2] : "";
98      String l = channelCmp.length > 3 ? channelCmp[3] : "";
99  
100     header.setKstnm(s);
101     header.setKcmpnm(c);
102     header.setKnetwk(n);
103     header.setKhole(l);
104 
105     Calendar cal = Calendar.getInstance();
106     cal.setTimeZone(TimeZone.getTimeZone("UTC"));
107     Wave wave = waves.get(channel);
108     cal.setTime(J2kSec.asDate(wave.getStartTime()));
109     header.setNzyear(cal.get(Calendar.YEAR));
110     header.setNzjday(cal.get(Calendar.DAY_OF_YEAR));
111     header.setNzhour(cal.get(Calendar.HOUR_OF_DAY));
112     header.setNzmin(cal.get(Calendar.MINUTE));
113     header.setNzsec(cal.get(Calendar.SECOND));
114     header.setNzmsec(cal.get(Calendar.MILLISECOND));
115 
116     header.setDelta((float) wave.getSamplingPeriod());
117     header.setNpts(wave.numSamples());
118     return header;
119   }
