51    /**
52     * Read.
53     */
54    public void read() throws IOException {
55      Map<String, Map<Long, int[]>> samples = new HashMap<String, Map<Long, int[]>>();
56      Map<String, Float> sampleRates = new HashMap<String, Float>();
57  
58      DataInputStream dis = null;
59      try {
60        dis = new DataInputStream(new BufferedInputStream(new FileInputStream(fileName)));
61      } catch (FileNotFoundException e) {
62        e.printStackTrace();
63        return;
64      }
65  
66      try {
67        boolean isDone = false;
68        while (!isDone) {
69          SeedRecord sr = null;
70          try {
71            /*
72             * Seiscomp's slarchive app preallocates file space. This can result in embedded NULL
73             * space if the app exits unexpectedly, such as in response to a power interruption.
74             */
75            skipNull(dis);
76            sr = SeedRecord.read(dis, 4096);
77          } catch (EOFException e) {
78            isDone = true;
79          } catch (Exception e) {
80            System.err.println(e.getMessage());
81            e.printStackTrace();
82          }
83          if (!(sr instanceof DataRecord)) {
84            continue;
85          }
86          DataRecord dr = (DataRecord) sr;
87          DataHeader dh = dr.getHeader();
88  
89          network = dh.getNetworkCode().trim();
90          station = dh.getStationIdentifier().trim();
91          channel = dh.getChannelIdentifier().trim();
92          location = dh.getLocationIdentifier();
93  
94          String code = station + "$" + channel + "$" + network;
95          if (location != null && !"  ".equals(location)) {
96            code += "$" + location;
97          }
98  
99          if (dr.getDataSize() == 0) {
100           continue;
101         }
102 
103         Map<Long, int[]> parts = samples.get(code);
104         if (parts == null) {
105           parts = new HashMap<Long, int[]>();
106           samples.put(code, parts);
107         }
108 
109         Blockette[] blocketts = dr.getBlockettes(100);
110         float sampleRate;
111         if (blocketts.length != 0) {
112           Blockette100 b100 = (Blockette100) blocketts[0];
113           sampleRate = b100.getActualSampleRate();
114         } else {
115           sampleRate = dh.getSampleRate();
116         }
117         sampleRates.put(code, sampleRate);
118         long start = btimeToDate(dh.getStartBtime()).getTime();
119         parts.put(start, extract(dr));
120       }
121 
122       for (String code : samples.keySet()) {
123         long samplePeriod = (long) (1000 / sampleRates.get(code));
124         Wave wave = join(code, samplePeriod, samples.get(code));
125         waves.put(code, wave);
126       }
127     } catch (Throwable t) {
128       t.printStackTrace();
129     } finally {
130       try {
131         dis.close();
132       } catch (IOException e) {
133         e.printStackTrace();
134       }
135     }
136   }

138   private void skipNull(DataInputStream dis) throws IOException {
139     if (!dis.markSupported()) {
140       return;
141     }
142     dis.mark(Integer.MAX_VALUE);
143     while (dis.readByte() == 0) {
144       dis.mark(Integer.MAX_VALUE);
145     }
146     dis.reset();
147   }

149   private Wave join(String code, long samplePeriodMs, Map<Long, int[]> samples) {
150     Set<Long> times = samples.keySet();
151     long firstTime = Long.MAX_VALUE;
152     long lastTime = Long.MIN_VALUE;
153 
154     for (long time : times) {
155       firstTime = Math.min(firstTime, time);
156 
157       long end = time + samples.get(time).length * samplePeriodMs;
158       lastTime = Math.max(lastTime, end);
159     }
160 
161     int sampleCount = (int) ((lastTime - firstTime + 1) / samplePeriodMs);
162 
163     int[] allSamples = new int[sampleCount];
164     Arrays.fill(allSamples, Wave.NO_DATA);
165 
166     for (long time : times) {
167       int[] buf = samples.get(time);
168       int idx = (int) ((time - firstTime) / samplePeriodMs);
169       System.arraycopy(buf, 0, allSamples, idx, buf.length);
170     }
171 
172     Wave wave = new Wave(allSamples, J2kSec.fromDate(new Date(firstTime)), 1000 / samplePeriodMs);
173     return wave;
174   }

176   private int[] extract(DataRecord dr)
177       throws UnsupportedCompressionType, CodecException, SeedFormatException {
178     int numPts = dr.getHeader().getNumSamples();
179     int[] data = new int[numPts];
180     int numSoFar = 0;
181     DecompressedData decompData = dr.decompress();
182     int[] temp = decompData.getAsInt();
183     System.arraycopy(temp, 0, data, numSoFar, temp.length);
184     numSoFar += temp.length;
185     return data;
186   }

188   private Date btimeToDate(Btime btime) {
189     Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
190     cal.set(Calendar.YEAR, btime.getYear());
191     cal.set(Calendar.DAY_OF_YEAR, btime.getDayOfYear());
192     cal.set(Calendar.HOUR_OF_DAY, btime.getHour());
193     cal.set(Calendar.MINUTE, btime.getMin());
194     cal.set(Calendar.SECOND, btime.getSec());
195     cal.set(Calendar.MILLISECOND, btime.getTenthMilli() / 10);
196     return cal.getTime();
197   }

199   /**
200    * write.
201    */
202   public synchronized void write() throws IOException {
203     DataOutputStream dos = null;
204     try {
205       dos = new DataOutputStream(new BufferedOutputStream(new FileOutputStream(fileName)));
206     } catch (FileNotFoundException e1) {
207       e1.printStackTrace();
208     }
209 
210     int seq = 1;
211     for (String channel : waves.keySet()) {
212       List<Wave> wavesSegments = waves.get(channel).split(512);
213       for (Wave wave : wavesSegments) {
214         DataHeader header = new DataHeader(seq++, 'D', false);
215 
216         String[] channelCmp = channel.split("$");
217         if (channelCmp.length < 3) {
218           channelCmp = channel.split("_");
219         }
220         if (channelCmp.length < 3) {
221           channelCmp = channel.split(" ");
222         }
223 
224         String s = channelCmp.length > 0 ? channelCmp[0] : "";
225         String c = channelCmp.length > 1 ? channelCmp[1] : "";
226         String n = channelCmp.length > 2 ? channelCmp[2] : "";
227         String l = channelCmp.length > 3 ? channelCmp[3] : "";
228 
229         header.setStationIdentifier(s);
230         header.setChannelIdentifier(c);
231         header.setNetworkCode(n);
232         header.setLocationIdentifier(l);
233 
234         header.setNumSamples((short) wave.numSamples());
235         header.setSampleRate(wave.getSamplingRate());
236         Btime btime = new Btime(J2kSec.asDate(wave.getStartTime()));
237         header.setStartBtime(btime);
238 
239         DataRecord record = new DataRecord(header);
240 
241         try {
242           Blockette1000 blockette1000 = new Blockette1000();
243           blockette1000.setEncodingFormat((byte) B1000Types.STEIM2);
244           blockette1000.setWordOrder((byte) 1);
245 
246           // record length as a power of 2
247           blockette1000.setDataRecordLength((byte) (12));
248 
249           record.addBlockette(blockette1000);
250 
251           SteimFrameBlock data = null;
252 
253           try {
254             data = Steim2.encode(wave.buffer, 63);
255             record.setData(data.getEncodedData());
256             record.write(dos);
257           } catch (SteimException e) {
258             e.printStackTrace();
259           }
260         } catch (SeedFormatException e) {
261           e.printStackTrace();
262         }
263 
264       }
265     }
266     if (dos != null) {
267       dos.close();
268     }
269   }
