42    /**
43     * @see gov.usgs.plot.data.file.SeismicDataFile#read()
44     */
45    public void read() throws IOException {
46      FileInputStream fis = new FileInputStream(fileName);
47      BufferedInputStream buf = new BufferedInputStream(fis);
48  
49      detectArchitecture(buf);
50      readEventFileHeader(buf);
51      for (int i = 0; i < channelCount; i++) {
52        readChannel(buf);
53      }
54    }

56    /*
57     * length of first header line length is always 80. Infer byte order and int
58     * length based on how that length is represented
59     */
60    private void detectArchitecture(BufferedInputStream bis) throws IOException {
61      byte[] bytes = new byte[8];
62      bis.mark(Integer.MAX_VALUE);
63      bis.read(bytes);
64      bis.reset();
65  
66      byteOrder = (bytes[0] == 0x50) ? ByteOrder.LITTLE_ENDIAN : ByteOrder.BIG_ENDIAN;
67      boolean is64Bit = (bytes[7] == 0x50 || bytes[7] == 0x00) ? true : false;
68      machineIntLength = is64Bit ? 8 : 4;
69    }

71    private void readEventFileHeader(BufferedInputStream bis) throws IOException {
72      String headerLine = new String(readRecord(bis));
73      channelCount = Integer.parseInt(headerLine.substring(30, 33).trim());
74      // System.out.println("CHANNEL COUNT " + channelCount + " :" + headerLine.substring(30, 33) +
75      // ":");
76  
77      // skip the rest of the header lines. We don't need them.
78      int numHeaderLines = 2 + ((channelCount + 2) / 3);
79      for (int i = 1; i < Math.max(numHeaderLines, 12); i++) {
80        readRecord(bis);
81      }
82    }

84    private void readChannel(BufferedInputStream bis) throws IOException {
85      byte[] channelHeader = readRecord(bis);
86  
87      String code = extractCode(channelHeader);
88  
89      double start = Time.ewToj2k(extractStartTime(channelHeader) / 1000d);
90      double samplingRate = Double.parseDouble(new String(channelHeader, 36, 7).trim());
91      int sampleCount = Integer.parseInt(new String(channelHeader, 43, 7).trim());
92  
93      char c = (char) channelHeader[76];
94      if (c == ' ') {
95        c = 2;
96      }
97      int intLength = Integer.parseInt("" + c);
98  
99      int[] samples = new int[sampleCount];
100     int sampleIndex = 0;
101     while (sampleIndex < sampleCount) {
102       byte[] data = readRecord(bis);
103 
104       for (int i = 0; i < data.length; i += intLength) {
105         byte[] b = Arrays.copyOfRange(data, i, i + intLength);
106         samples[sampleIndex++] = decodeInt(b);
107       }
108     }
109     waves.put(code, new Wave(samples, start, samplingRate));
110   }

112   private long extractStartTime(byte[] header) {
113     Calendar c = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
114 
115     int year = Integer.parseInt(new String(header, 9, 3).trim()) + 1900;
116     c.set(Calendar.YEAR, year);
117 
118     int month = Integer.parseInt(new String(header, 17, 2).trim());
119     c.set(Calendar.MONTH, month - 1);
120 
121     int day = Integer.parseInt(new String(header, 20, 2).trim());
122     c.set(Calendar.DAY_OF_MONTH, day);
123 
124     int hour = Integer.parseInt(new String(header, 23, 2).trim());
125     c.set(Calendar.HOUR_OF_DAY, hour);
126 
127     int min = Integer.parseInt(new String(header, 26, 2).trim());
128     c.set(Calendar.MINUTE, min);
129 
130     double sec = Float.parseFloat(new String(header, 29, 6).trim());
131     c.set(Calendar.SECOND, (int) sec);
132     double ms = (sec - Math.floor(sec)) * 1000;
133     c.set(Calendar.MILLISECOND, (int) ms);
134 
135     return c.getTimeInMillis();
136   }

139   /*
140    * Extract channel code from event file channel header. Attempt to
141    * accommodate a bug in earthworm versions prior to 7.7.
142    */
143   private String extractCode(byte[] header) {
144     // try to accommodate EW bug which misplaced the third character of the
145     // component. Fixed in EW 7.7
146     boolean ewBug = false;
147     char c1 = (char) header[7];
148     char c2 = (char) header[8];
149     if ((c1 == 'E' || c1 == 'Z' || c1 == 'N') && !(c2 == 'E' || c2 == 'Z' || c2 == 'N')) {
150       ewBug = true;
151     }
152 
153     String cmp;
154     String loc;
155     if (ewBug) {
156       cmp = new String(header, 5, 2) + new String(header, 7, 1);
157       loc = new String(header, 8, 1) + new String(header, 12, 1);
158     } else {
159       cmp = new String(header, 5, 2) + new String(header, 8, 1);
160       loc = new String(header, 7, 1) + new String(header, 12, 1);
161     }
162 
163     channel = cmp.trim();
164 
165     if (location == null) {
166       location = loc;
167     }
168 
169     if (network == null) {
170       network = new String(header, 16, 1) + new String(header, 19, 1);
171     }
172 
173     String station = new String(header, 0, 5).trim();
174     String code = station + readSeparator + channel + readSeparator + network;
175     if (!"--".equals(location) && !"  ".equals(location)) {
176       code += readSeparator + location.trim();
177     }
178 
179     return code;
180   }

183   /**
184    * @see gov.usgs.plot.data.file.SeismicDataFile#write()
185    */
186   public void write() throws IOException {
187     FileOutputStream fos = new FileOutputStream(new File(fileName));
188     writeEventFileHeader(fos);
189     for (String code : waves.keySet()) {
190       Wave wave = waves.get(code);
191       writeChannelHeader(fos, code);
192       byte[] bytes = new byte[wave.numSamples() * 4];
193       for (int i = 0; i < wave.numSamples(); i++) {
194         System.arraycopy(intAsBytes(wave.buffer[i]), 0, bytes, i * 4, 4);
195       }
196       writeRecord(fos, bytes);
197     }
198 
199     fos.close();
200   }

203   public void writeEventFileHeader(FileOutputStream fos) throws IOException {
204     byte[] header = new byte[80];
205     Arrays.fill(header, (byte) 0x20);
206 
207     byte[] b;
208     b = ("" + waves.size()).getBytes();
209     System.arraycopy(b, 0, header, 31, Math.min(b.length, 3));
210 
211     double fileStart = Double.MAX_VALUE;
212     double fileEnd = -Double.MAX_VALUE;
213     for (String code : waves.keySet()) {
214       fileStart = Math.min(fileStart, waves.get(code).getStartTime());
215       fileEnd = Math.max(fileEnd, waves.get(code).getEndTime());
216     }
217     Calendar c = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
218     c.setTime(J2kSec.asDate(fileStart));
219 
220     b = ("" + (c.get(Calendar.YEAR) - 1900)).getBytes();
221     System.arraycopy(b, 0, header, 33, Math.min(b.length, 3));
222 
223     b = ("" + c.get(Calendar.DAY_OF_YEAR)).getBytes();
224     System.arraycopy(b, 0, header, 37, Math.min(b.length, 3));
225 
226     b = ("" + (c.get(Calendar.MONTH) + 1)).getBytes();
227     System.arraycopy(b, 0, header, 41, Math.min(b.length, 2));
228 
229     b = ("" + c.get(Calendar.DAY_OF_MONTH)).getBytes();
230     System.arraycopy(b, 0, header, 44, Math.min(b.length, 2));
231 
232     b = ("" + c.get(Calendar.HOUR_OF_DAY)).getBytes();
233     System.arraycopy(b, 0, header, 47, Math.min(b.length, 2));
234 
235     b = ("" + c.get(Calendar.MINUTE)).getBytes();
236     System.arraycopy(b, 0, header, 50, Math.min(b.length, 2));
237 
238     float f = c.get(Calendar.SECOND) + c.get(Calendar.MILLISECOND) / 1000;
239     b = String.format("%6.3f", f).getBytes();
240     System.arraycopy(b, 0, header, 53, Math.min(b.length, 6));
241 
242     b = String.format("%9.3f", (fileEnd - fileStart)).getBytes();
243     System.arraycopy(b, 0, header, 60, 9);
244     writeRecord(fos, header);
245 
246     Arrays.fill(header, (byte) 0x20);
247     writeRecord(fos, header);
248 
249     int chanIdx = 0;
250     for (String code : waves.keySet()) {
251       if (chanIdx % 3 == 0) {
252         if (chanIdx > 0) {
253           writeRecord(fos, header);
254         }
255         Arrays.fill(header, (byte) 0x20);
256       }
257 
258       String[] comps = code.split(writeSeparator);
259       if (comps.length > 0) {
260         b = comps[0].getBytes();
261         System.arraycopy(b, 0, header, chanIdx % 3 * 26 + 1, Math.min(b.length, 4));
262         if (b.length > 4) {
263           System.arraycopy(b, 4, header, chanIdx % 3 * 26 + 9, 1);
264         }
265       }
266 
267       if (comps.length > 1) {
268         b = comps[1].getBytes();
269         System.arraycopy(b, 0, header, chanIdx % 3 * 26 + 5, Math.min(b.length, 2));
270         if (b.length > 2) {
271           System.arraycopy(b, 2, header, chanIdx % 3 * 26 + 8, 1);
272         }
273       }
274 
275       Wave wave = waves.get(code);
276       double channelStart = wave.getStartTime() - fileStart;
277       b = String.format("%7.2f", channelStart).getBytes();
278       System.arraycopy(b, 0, header, chanIdx % 3 * 26 + 10, 7);
279 
280       double dataLen = (wave.getEndTime() - wave.getSamplingPeriod()) - wave.getStartTime();
281       b = String.format("%8.2f", dataLen).getBytes();
282       System.arraycopy(b, 0, header, chanIdx % 3 * 26 + 18, 8);
283     }
284     writeRecord(fos, header);
285     Arrays.fill(header, (byte) 0x20);
286 
287     for (int i = 3 + ((channelCount + 2) / 3); i < 12; i++) {
288       writeRecord(fos, header);
289     }
290 
291   }

294   /**
295    * Write channel header.
296    * @param fos file output stream
297    * @param code channel code
298    * @throws IOException exception
299    */
300   public void writeChannelHeader(FileOutputStream fos, String code) throws IOException {
301 
302     byte[] header = new byte[1040];
303     Arrays.fill(header, (byte) 0x20);
304 
305     byte[] b;
306     String[] comps = code.split(writeSeparator);
307     if (comps.length > 0) {
308       b = comps[0].getBytes();
309       System.arraycopy(b, 0, header, 0, Math.min(b.length, 5));
310     }
311 
312     if (comps.length > 1) {
313       b = comps[1].getBytes();
314       System.arraycopy(b, 0, header, 5, Math.min(b.length, 2));
315       if (b.length > 2) {
316         System.arraycopy(b, 2, header, 8, 1);
317       }
318     }
319 
320     if (comps.length > 2) {
321       b = comps[2].getBytes();
322       System.arraycopy(b, 0, header, 16, 1);
323       if (b.length > 1) {
324         System.arraycopy(b, 1, header, 19, 1);
325       }
326     }
327 
328     if (comps.length > 3) {
329       b = comps[3].getBytes();
330       System.arraycopy(b, 0, header, 7, 1);
331       if (b.length > 1) {
332         System.arraycopy(b, 1, header, 12, 1);
333       }
334     }
335 
336     Calendar c = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
337     Wave wave = waves.get(code);
338     Date da = J2kSec.asDate(wave.getStartTime());
339     c.setTime(da);
340 
341     b = String.format("%3d", c.get(Calendar.YEAR) - 1900).getBytes();
342     System.arraycopy(b, 0, header, 9, Math.min(b.length, 3));
343 
344     b = String.format("%3d", c.get(Calendar.DAY_OF_YEAR)).getBytes();
345     System.arraycopy(b, 0, header, 13, Math.min(b.length, 3));
346 
347     b = String.format("%2d", c.get(Calendar.MONTH) + 1).getBytes();
348     System.arraycopy(b, 0, header, 17, Math.min(b.length, 2));
349 
350     b = String.format("%2d", c.get(Calendar.DAY_OF_MONTH)).getBytes();
351     System.arraycopy(b, 0, header, 20, Math.min(b.length, 2));
352 
353     b = String.format("%2d", c.get(Calendar.HOUR_OF_DAY)).getBytes();
354     System.arraycopy(b, 0, header, 23, Math.min(b.length, 2));
355 
356     b = String.format("%2d", c.get(Calendar.MINUTE)).getBytes();
357     System.arraycopy(b, 0, header, 26, Math.min(b.length, 2));
358     float f = c.get(Calendar.SECOND) + c.get(Calendar.MILLISECOND) / 1000;
359 
360     b = String.format("%6.3f", f).getBytes();
361     System.arraycopy(b, 0, header, 29, b.length);
362 
363     b = String.format("%7.3f", wave.getSamplingRate()).getBytes();
364     System.arraycopy(b, 0, header, 36, b.length);
365 
366     b = String.format("%7d", wave.numSamples()).getBytes();
367     System.arraycopy(b, 0, header, 43, b.length);
368 
369     header[76] = '4';
370 
371     writeRecord(fos, header);
372   }

376   // read a single record in fortran "unformatted" form
377   private byte[] readRecord(BufferedInputStream bis) throws IOException {
378     int recordLen = readInt(bis, machineIntLength);
379     byte[] record = new byte[recordLen];
380     int read = bis.read(record);
381 
382     if (read != readInt(bis, machineIntLength)) {
383       throw new IOException("Corrupt seisan record. Read " + read + ", expected " + recordLen);
384     }
385 
386     return record;
387   }

390   // write a single record in fortran "unformatted" form
391   private void writeRecord(FileOutputStream fos, byte[] bytes) throws IOException {
392     int recordLen = bytes.length;
393     fos.write(intAsBytes(recordLen));
394     fos.write(bytes);
395     fos.write(intAsBytes(recordLen));
396   }

399   private byte[] intAsBytes(int i) {
400     return ByteBuffer.allocate(4).putInt(i).array();
401   }

404   private int readInt(BufferedInputStream bis, int len) throws IOException {
405     byte[] buf = new byte[len];
406     int read = bis.read(buf);
407     if (read != len) {
408       throw new IOException("Bad int read.");
409     }
410 
411     return decodeInt(buf);
412   }
413

415   private int decodeInt(byte[] b) {
416     ByteBuffer buffer = ByteBuffer.wrap(b);
417     buffer.order(byteOrder);
418     return (b.length == 8) ? (int) buffer.getLong() : buffer.getInt();
419   }
