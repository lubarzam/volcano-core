15  /**
16   * A concrete SeismicDataFile class for seismic data in a two-column text file as exported from
17   * Matlab.
18   * 
19   * @author Tom Parker
20   * 
21   */
22  public class TextDataFile extends SeismicDataFile {
23  
24    protected TextDataFile(String fileName) {
25      super(fileName, "TXT^");
26    }
27  
28    /**
29     * @see gov.usgs.plot.data.file.SeismicDataFile#read()
30     */
31    public void read() throws IOException {
32      double startTime;
33      IntVector iv = new IntVector(5000, 10000);
34  
35      BufferedReader in = new BufferedReader(new FileReader(fileName));
36      Wave sw = new Wave();
37  
38      String line = in.readLine();
39      startTime = Double.parseDouble(line.substring(0, line.indexOf(" ")));
40      sw.setStartTime(Time.ewToj2k(startTime / 1000));
41      iv.add(Integer.parseInt(line.substring(line.indexOf(" ") + 1)));
42  
43      line = in.readLine();
44      double nt = Double.parseDouble(line.substring(0, line.indexOf(" ")));
45      sw.setSamplingRate(1 / ((nt - startTime) / 1000));
46      iv.add(Integer.parseInt(line.substring(line.indexOf(" ") + 1)));
47  
48      while (line != null) {
49        iv.add(Integer.parseInt(line.substring(line.indexOf(" ") + 1)));
50        line = in.readLine();
51      }
52  
53      sw.buffer = iv.getResizedInts();
54      in.close();
55  
56      String channel = getChannelFromFilename(fileName);
57      waves.put(channel, sw);
58    }
59  
60    /**
61     * @see gov.usgs.plot.data.file.SeismicDataFile#write()
62     */
63    public void write() throws IOException {
64      PrintWriter out = new PrintWriter(new BufferedWriter(new FileWriter(fileName)));
65  
66      for (Wave wave : waves.values()) {
67        double ct = Time.j2kToEw(wave.getStartTime());
68        for (int i = 0; i < wave.buffer.length; i++) {
69          out.println(Math.round(ct * 1000) + " " + wave.buffer[i]);
70          ct += wave.getSamplingPeriod();
71        }
72      }
73      out.close();
74  
75    }
76  
77    private String getChannelFromFilename(String fileName) {
78      File file = new File(fileName);
79      String channel = file.getName();
80      channel = channel.replaceAll("\\.(txt|TXT)", "");
81      channel = channel.replaceAll("_", "\\$");
82      channel = channel.replaceAll(" ", "\\$");
83      return channel;
84    }
85  }
