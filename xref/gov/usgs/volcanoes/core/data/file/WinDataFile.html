44    static int intFromSingleByte(byte b) {
45      return b;

48    private int intFromFourBytes(byte[] bites) {
49      converter4.clear();
50      converter4.mark();
51      converter4.put(bites);
52      converter4.rewind();
53      return converter4.getInt();
54    }
55  
56    private int intFromThreeBytes(byte[] bites) {
57      byte pad = (byte) ((bites[0] < 0) ? -1 : 0);
58      byte[] padded = new byte[] {pad, bites[0], bites[1], bites[2]};
59      return intFromFourBytes(padded);
60    }
61  
62    private short shortFromTwoBytes(byte[] bites) {
63      converter2.clear();
64      converter2.mark();
65      converter2.put(bites);
66      converter2.rewind();
67      return converter2.getShort();
68    }

70    private int decodeBcd(byte[] b) {
71      StringBuffer buf = new StringBuffer(b.length * 2);
72      for (int i = 0; i < b.length; ++i) {
73        buf.append((char) (((b[i] & 0xf0) >> 4) + '0'));
74        if ((i != b.length) && ((b[i] & 0xf) != 0x0A)) {
75          buf.append((char) ((b[i] & 0x0f) + '0'));
76        }
77      }
78      return Integer.parseInt(buf.toString());
79    }

81    /**
82     * Reads the header from the given stream.
83     * 
84     * @param dis DataInputStream to read WIN from
85     * @throws FileNotFoundException if the file cannot be found
86     * @throws IOException if it isn't a WIN file
87     */
88    private void readHeader(ChannelData c, DataInputStream dis)
89        throws FileNotFoundException, IOException {
90      // read first 4 byte: file size
91      byte[] fourBytes = new byte[4];
92      byte[] oneByte = new byte[1];
93  
94      dis.readFully(fourBytes);
95      c.packetSize = intFromFourBytes(fourBytes);
96  
97      // read next 6 bytes: yy mm dd hh mi ss
98      dis.readFully(oneByte);
99      c.year = 2000 + decodeBcd(oneByte);
100 
101     dis.readFully(oneByte);
102     c.month = decodeBcd(oneByte);
103 
104     dis.readFully(oneByte);
105     c.day = decodeBcd(oneByte);
106 
107     dis.readFully(oneByte);
108     c.hour = decodeBcd(oneByte);
109 
110     dis.readFully(oneByte);
111     c.minute = decodeBcd(oneByte);
112 
113     dis.readFully(oneByte);
114     c.second = decodeBcd(oneByte);
115   }

117   /**
118    * Read the data portion of WIN format from the given stream.
119    * 
120    * @param dis DataInputStream to read WIN from
121    * @throws IOException if it isn't a WIN file
122    */
123   private void readData(final ChannelData header, final DataInputStream dis) throws IOException {
124     int bytesRead = 10;
125 
126     do {
127       ChannelData c = new ChannelData(header);
128       c.inBuf = new ArrayList<Integer>();
129       byte[] oneByte = new byte[1];
130       dis.readFully(oneByte);
131 
132       dis.readFully(oneByte);
133       c.channelNumber = intFromSingleByte(oneByte[0]);
134 
135       dis.readFully(oneByte);
136       byte sampleRateUpperBits = (byte) (oneByte[0] & 0xF);
137       c.dataSize = intFromSingleByte(oneByte[0]) >> 4;
138 
139       dis.readFully(oneByte);
140       c.samplingRate = intFromSingleByte(oneByte[0]) + (sampleRateUpperBits << 4);
141 
142       byte[] fourBytes = new byte[4];
143       dis.readFully(fourBytes);
144       int accum = intFromFourBytes(fourBytes);
145 
146       c.inBuf.add(accum);
147 
148       float[] d = new float[(int) c.samplingRate - 1];
149 
150       bytesRead += 8;
151       if (c.dataSize == 0) {
152         for (int ix = 0; ix < ((int) c.samplingRate - 1); ix++) {
153           accum += dis.readByte();
154           c.inBuf.add(accum);
155 
156         }
157       } else if (c.dataSize == 1) {
158         for (int ix = 0; ix < ((int) c.samplingRate - 1); ix++) {
159           accum += dis.readByte();
160           c.inBuf.add(accum);
161           bytesRead++;
162         }
163       } else if (c.dataSize == 2) {
164         byte[] twoBytes = new byte[2];
165         for (int ix = 0; ix < ((int) c.samplingRate - 1); ix++) {
166           dis.readFully(twoBytes);
167           accum += shortFromTwoBytes(twoBytes);
168           d[ix] = accum;
169           c.inBuf.add(accum);
170           bytesRead += 2;
171         }
172       } else if (c.dataSize == 3) {
173         byte[] threeBytes = new byte[3];
174         for (int ix = 0; ix < ((int) c.samplingRate - 1); ix++) {
175           dis.readFully(threeBytes);
176           accum += intFromThreeBytes(threeBytes);
177           d[ix] = accum;
178           c.inBuf.add(accum);
179           bytesRead += 3;
180         }
181       } else if (c.dataSize == 4) {
182         for (int ix = 0; ix < ((int) c.samplingRate - 1); ix++) {
183           dis.readFully(fourBytes);
184           accum += intFromFourBytes(fourBytes);
185 
186           d[ix] = accum;
187           c.inBuf.add(accum);
188           bytesRead += 4;
189         }
190       }
191       List<ChannelData> list = channelMap.get(c.channelNumber);
192       if (list == null) {
193         list = new ArrayList<ChannelData>();
194         channelMap.put(c.channelNumber, list);
195       }
196       list.add(c);
197     } while (bytesRead < header.packetSize);
198   }

200   /**
201    * @see gov.usgs.plot.data.file.SeismicDataFile#read()
202    */
203   public void read() throws IOException {
204     // Read WIN configuration file
205     if (configFile != null) {
206       FileReader fileReader = new FileReader(configFile);
207       BufferedReader reader = new BufferedReader(fileReader);
208       timeZone = reader.readLine();
209       channelInfo.clear();
210       int num = 0;
211       while (true) {
212         String line = reader.readLine();
213         if (line == null) {
214           break;
215         }
216         channelInfo.put(num, line);
217         num++;
218       }
219       reader.close();
220       fileReader.close();
221     }
222 
223     // Read WIN file
224     FileInputStream fis = new FileInputStream(fileName);
225     BufferedInputStream buf = new BufferedInputStream(fis);
226     DataInputStream dis = new DataInputStream(buf);
227     while (dis.available() != 0) {
228       ChannelData cur = new ChannelData();
229       readHeader(cur, dis);
230       readData(cur, dis);
231     }
232     dis.close();
233 
234     // create wave objects
235     for (List<ChannelData> channels : channelMap.values()) {
236       List<Wave> subParts = new ArrayList<Wave>(channels.size());
237       int index = channels.get(0).channelNumber;
238       String channel = channelInfo.get(index);
239       if (channel == null) {
240         channel = Integer.toString(index);
241       } else {
242         channel = channel.replaceAll(" ", "\\$");
243       }
244       for (ChannelData c : channels) {
245         subParts.add(toWave(c));
246         Wave wave = Wave.join(subParts);
247         waves.put(channel, wave);
248       }
249     }
250   }

252   private Wave toWave(ChannelData c) {
253     Wave sw = new Wave();
254     sw.setStartTime(J2kSec.fromDate(getStartTime(c)));
255     sw.setSamplingRate(c.samplingRate);
256     sw.buffer = new int[c.inBuf.size()];
257     for (int j = 0; j < c.inBuf.size(); j++) {
258       sw.buffer[j] = Math.round(c.inBuf.get(j));
259     }
260     return sw;
261   }

263   /**
264    * Get start time of data.
265    * 
266    * @return start time of data
267    */
268   private Date getStartTime(ChannelData c) {
269     Calendar cal = Calendar.getInstance(TimeZone.getTimeZone(timeZone));
270     cal.setTimeInMillis(0);
271     cal.set(Calendar.YEAR, c.year);
272     cal.set(Calendar.MONTH, c.month - 1);
273     cal.set(Calendar.DAY_OF_MONTH, c.day);
274     cal.set(Calendar.HOUR_OF_DAY, c.hour);
275     cal.set(Calendar.MINUTE, c.minute);
276     cal.set(Calendar.SECOND, c.second);
277     cal.setTimeZone(TimeZone.getTimeZone("UTC"));
278     return cal.getTime();
279   }
