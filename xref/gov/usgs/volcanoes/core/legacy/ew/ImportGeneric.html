35    class ExpectedHeartbeatThread extends Thread implements MessageListener {
36      private boolean kill = false;
37  
38      public ExpectedHeartbeatThread() {
39        super("ExpectedHeartbeat");
40        lastHeartbeat = System.currentTimeMillis();
41        start();
42      }
43  
44      public void kill() {
45        kill = true;
46        interrupt();
47      }
48  
49      @Override
50      public void messageReceived(Message msg) {
51        logger.finer("Received heartbeat.");
52        final String recv = msg.bytesToString().trim();
53  
54        if (!recv.equals(recvIDString)) {
55          logger.fine("Warning: heartbeat message, '" + recv + "', does not match expected message, '"
56              + recvIDString + "'.");
57        }
58        lastHeartbeat = System.currentTimeMillis();
59        if (msg.sendAck) {
60          sendAck(msg.seq);
61        }
62      }
63  
64      @Override
65      public void run() {
66        while (!kill) {
67          try {
68            Thread.sleep(expectedHeartbeatInterval);
69            if (System.currentTimeMillis() - lastHeartbeat >= expectedHeartbeatInterval * 2) {
70              logger.fine("Have not received heartbeat recently.  Reconnecting.");
71              needReconnect = true;
72              kill = true;
73            }
74          } catch (final InterruptedException e) {
75          } catch (final OutOfMemoryError e) {
76            outOfMemoryErrorOccurred(e);
77          }
78        }
79        logger.fine("Expected heartbeat thread died.");
80      }
81    }

99      @Override

107         } catch (final InterruptedException e) {
108         } catch (final IOException e) {
109           logger.warning("Exception in heartbeat sending thread, reconnecting.");
110           needReconnect = true;
111           kill = true;
112           // reconnect?
113         } catch (final OutOfMemoryError e) {
114           outOfMemoryErrorOccurred(e);
115         }
116       }

118     }

202   public void close() {
203     try {
204       if (socket.isClosed()) {
205         return;
206       }
207       connected = false;
208       heartbeat.kill();
209       expectedHeartbeat.kill();
210       removeListener(MessageType.TYPE_HEARTBEAT, expectedHeartbeat);
211       socketOut.close();
212       socketWriter.close();
213       socketIn.close();
214       socket.close();
215     } catch (final IOException e) {

220   public boolean connect() {
221     Boolean b = null;
222     final Retriable<Boolean> result = new Retriable<Boolean>("Import.connect()", maxRetries) {
223       @Override
224       public boolean attempt() throws UtilException {
225         output = false;
226         try {
227           socket = new Socket();
228           socket.connect(new InetSocketAddress(host, port), timeout);
229           socket.setSoLinger(false, 0);
230           socket.setSoTimeout(timeout);
231           socketOut = new DataOutputStream(socket.getOutputStream());
232           socketWriter = new PrintWriter(socket.getOutputStream());
233           socketIn = new DataInputStream(new BufferedInputStream(socket.getInputStream()));
234           logger.fine("ImportGeneric connected.");
235           result = new Boolean(true);
236           heartbeat = new HeartbeatThread();
237           heartbeat.start();
238           expectedHeartbeat = new ExpectedHeartbeatThread();
239           addListener(MessageType.TYPE_HEARTBEAT, expectedHeartbeat);
240           connected = true;
241           needReconnect = false;
242           if (!ImportGeneric.this.isAlive()) {
243             start();
244           }
245           return true;
246         } catch (final SocketTimeoutException e) {
247           // if (!quiet)
248           // logger.severe("Import.connect() socket timeout: could not connect.");
249         } catch (final IOException e) {
250           // if (!quiet)
251           // logger.severe("ImportGeneric.connect() IOException: " +
252           // e.getMessage());
253         }
254         // quiet = true;
255         return false;
256       }
257 
258       @Override
259       public void giveUp() {
260         if (System.currentTimeMillis() - lastConnectUpdate > 10 * 60 * 1000) {
261           logger.severe(
262               "ImportGeneric: can not connect, will update periodically while attempting to reconnect.");
263           try {
264             Thread.sleep(1000);
265           } catch (final Exception e) {
266           }
267           lastConnectUpdate = System.currentTimeMillis();
268         }
269       }
270     };
271     try {
272       b = result.go();
273     } catch (final UtilException e) {
274       // Do nothing
275     }
276     return b != null && b.booleanValue();
277   }

329   @Override
330   public void run() {
331     // logger.fine("Starting listener thread.");
332     byte lastByte = ESC;
333     while (!shutdown) {
334       try {
335         if (needReconnect) {
336           close();
337           connect();
338           continue;
339         }
340         if (!connected) {
341           Thread.sleep(250);
342           continue;
343         }
344 
345         boolean start = false;
346 
347         try {
348           final byte b = socketIn.readByte();
349           if (b == STX) {
350             if (lastByte != ESC) {
351               start = true;
352             } else {
353               lastByte = b;
354             }
355           }
356         } catch (final EOFException e) {
357           needReconnect = true;
358         }
359 
360         if (start) {
361           int i = 0;
362           int seq = Integer.MAX_VALUE;
363           boolean done = false;
364           boolean escape = false;
365           while (!done) {
366             final byte b3 = socketIn.readByte();
367             if (!escape && b3 == ETX) {
368               done = true;
369             } else if (!escape && b3 == ESC) {
370               escape = true;
371             } else {
372               msgBuf[i++] = b3;
373               escape = false;
374             }
375 
376             // check for sequence number and reset index if found
377             if (i == 6 && seq == Integer.MAX_VALUE) {
378               seq = findSeq(new String(msgBuf, 0, 6));
379               if (seq != Integer.MIN_VALUE) {
380                 i = 0;
381               }
382             }
383           }
384           // message done
385           msgBuf[i++] = 0;
386           final Message msg = MessageFactory.createMessage(msgBuf, i, seq);
387           dispatchMessage(msg);
388         }
389         if (shutdown) {
390           close();
391         }
392       } catch (final Exception e) {
393         // TODO: suppress logging
394         logger.severe("Main loop exception: " + e.getMessage() + ", will attempt reconnect.");
395         e.printStackTrace();
396         needReconnect = true;
397       } catch (final OutOfMemoryError e) {
398         outOfMemoryErrorOccurred(e);
399       }
400     }
401   }

419   public void sendMsg(MessageLogo l, String s) throws IOException {
420     final byte[] b = new byte[50];
421     b[0] = STX;
422     final byte[] lb = l.toDataStreamBytes();
423     System.arraycopy(lb, 0, b, 1, 9);
424     int j = 10;
425     for (int i = 0; i < s.length(); i++, j++) {
426       b[j] = ByteUtil.intToByte(s.charAt(i));
427     }
428     b[j++] = 0;
429     b[j++] = ETX;
430     socketOut.write(b, 0, j);
431     socketOut.flush();
432   }
