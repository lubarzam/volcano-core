52    public Menu getMenu(final String reqID) {
53      Menu ret = null;
54      Retriable<Menu> mr = new Retriable<Menu>("WaveServer.getMenu()", maxRetries) {
55        public void attemptFix() {
56          close();
57        }
58  
59        public boolean attempt() throws UtilException {
60          try {
61            if (!connected())
62              connect();
63            writeString("MENU: " + reqID + "\n");
64            String info = readString();
65            result = new Menu(info);
66            return true;
67          } catch (Exception e) {
68            LOGGER.warn("getMenu() IOException: {}", e.getMessage());
69          }
70          return false;
71        }
72      };
73      try {
74        ret = mr.go();
75      } catch (UtilException e) {
76        // Do nothing
77      }
78      return ret;
79    }

222   public List<TraceBuf> getTraceBufs(final String station, final String comp, final String network,
223       final String location, final double start, final double end) {
224     List<TraceBuf> ret = null;
225     Retriable<List<TraceBuf>> rt =
226         new Retriable<List<TraceBuf>>("WaveServer.getTraceBufs()", maxRetries) {
227           public void attemptFix() {
228             close();
229           }
230 
231           public boolean attempt() throws UtilException {
232             try {
233               if (!connected())
234                 connect();
235               int spanIndex = 5;
236               int lengthIndex = 9;
237               String cmd = "GETSCNRAW: GS ";
238               String loc = null;
239               boolean isTb2 = false;
240               if (location != null) {
241                 cmd = "GETSCNLRAW: GS ";
242                 loc = location;
243                 isTb2 = true;
244                 spanIndex++;
245                 lengthIndex++;
246               }
247 
248               String req = cmd + station + " " + comp + " " + network + " "
249                   + (loc == null ? "" : loc + " ") + start + " " + end;
250               socketOut.print(req + "\n");
251               socketOut.flush();
252               String info = readString();
253               String[] ss = info.split(" ");
254 
255               // debugging code attempting to isolate reported problem with WSV
256               if (ss.length < spanIndex + 1) {
257                 LOGGER.warn("Can't parse server response.");
258                 LOGGER.warn("Sent: {}", req);
259                 LOGGER.warn("Received: {}", info);
260 
261                 return false;
262               }
263 
264               if (!ss[spanIndex].equals("F"))
265                 return true;
266               int bytes = Integer.parseInt(ss[lengthIndex]);
267               byte[] buf = readBinary(bytes);
268 
269               DataInputStream dis = new DataInputStream(new ByteArrayInputStream(buf));
270               List<TraceBuf> tbs = new ArrayList<TraceBuf>(100);
271               while (dis.available() > 0) {
272                 TraceBuf tb = new TraceBuf();
273                 tb.processBytes(dis, isTb2);
274                 tbs.add(tb);
275               }
276               result = tbs;
277               return true;
278             } catch (SocketTimeoutException e) {
279               LOGGER.warn("getTraceBufs() timeout.");
280             } catch (Exception e) {
281               LOGGER.warn("getTraceBufs() exception: {}", e.getMessage());
282               e.printStackTrace();
283             }
284             return false;
285           }
286         };
287     try {
288       ret = rt.go();
289     } catch (UtilException e) {
290       // Do nothing
291     }
292     return ret;
293   }

307   public static void main(String[] args) {
308     if (args.length == 2) {
309       String server = args[0];
310       int port = Integer.MAX_VALUE;
311       try {
312         port = Integer.parseInt(args[1]);
313       } catch (Exception e) {
314       }
315       if (port == Integer.MAX_VALUE) {
316         System.err.println("Bad port.");
317         System.exit(1);
318       }
319       WaveServer ws = new WaveServer(server, port);
320       if (!ws.connect()) {
321         System.err.println("Could not connect to " + server + ":" + port);
322         System.exit(1);
323       }
324       System.out.println("Getting menu...");
325       Menu menu = ws.getMenu();
326       System.out.println(menu.toString());
327       ws.close();
328     } else {
329       System.out.println("java gov.usgs.earthworm.WaveServer [host] [port]");
330       System.exit(1);
331     }
332     System.exit(0);
333   }
