6   /**
7    * A factory for creating the correct subclass of <code>Message</code>.
8    *
9    * @author Dan Cervelli
10   */
11  public class MessageFactory {
12    /**
13     * Creates the correct type of message from a buffer of bytes.
14     * 
15     * @param b byte buffer
16     * @param i number of bytes that comprise message (!= b.length)
17     * @param findSeq Search for sequence number in message?
18     * @return the message
19     * @throws IOException
20     */
21    public static Message createMessage(byte[] b, int i, int seq) throws IOException {
22      MessageLogo logo = new MessageLogo(b);
23      Message msg = null;
24      switch (logo.getType()) {
25        case TYPE_TRACEBUF:
26          msg = TraceBuf.createFromBytes(b, i, seq);
27          break;
28        case TYPE_TRACEBUF2:
29          msg = TraceBuf.createFromBytesAsTraceBuf2(b, i, seq);
30          break;
31        case TYPE_HEARTBEAT:
32          msg = new Message(b, i, seq);
33          break;
34        default:
35          System.out.println("Unknown logo type: " + logo.getType());
36          msg = new Message(b, i, seq);
37          break;
38      }
39  
40      return msg;
41    }
42  }
