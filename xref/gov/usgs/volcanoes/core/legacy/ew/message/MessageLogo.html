5   import java.text.DecimalFormat;
6   import java.text.NumberFormat;
7   
8   /**
9    * A class for dealing with Earthworm message logos.
10   * 
11   * @author Dan Cervelli
12   */
13  public class MessageLogo {
14    private MessageType type;
15    private byte module;
16    private byte installationID;
17  
18  
19    public MessageLogo() {}
20  
21    public MessageType getType() {
22      return type;
23    }
24  
25    public void setType(MessageType t) {
26      type = t;
27  
28    }
29  
30    public void setModule(byte b) {
31      module = b;
32    }
33  
34    public byte getModule() {
35      return module;
36    }
37  
38    public void setInstallationId(byte b) {
39      installationID = b;
40    }
41  
42    public byte getInstallationId() {
43      return installationID;
44    }
45  
46    public MessageLogo(byte[] b) {
47      this(b, 0);
48    }
49  
50    public MessageLogo(byte[] b, int offset) {
51      String s = new String(b, offset, 9);
52      installationID = ByteUtil.intToByte(Integer.parseInt(s.substring(0, 3).trim()));
53      module = ByteUtil.intToByte(Integer.parseInt(s.substring(3, 6).trim()));
54      type = MessageType.fromInt(Integer.parseInt(s.substring(6, 9).trim()));
55    }
56  
57    private NumberFormat numberFormat;
58  
59    public byte[] toDataStreamBytes() {
60      if (numberFormat == null)
61        numberFormat = new DecimalFormat("000");
62  
63      String message = numberFormat.format(ByteUtil.byteToInt(installationID))
64          + numberFormat.format(ByteUtil.byteToInt(module)) + numberFormat.format(type.getType());
65  
66      byte[] b = new byte[9];
67      for (int i = 0; i < b.length; i++)
68        b[i] = ByteUtil.intToByte((int) message.charAt(i));
69  
70      return b;
71    }
72  
73    public String toString() {
74      return "Message: type=" + type + " module=" + module + " inst=" + installationID;
75    }
76  
77  }
