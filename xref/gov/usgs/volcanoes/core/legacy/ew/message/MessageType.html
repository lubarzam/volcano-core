3   /**
4    * An enum listing Earthworm message types.
5    * 
6    * @author Tom Parker
7    */
8   
9   public enum MessageType {
10  
11    TYPE_WILDCARD(0), // wildcard value - DO NOT CHANGE!!!
12    TYPE_ADBUF(1), // multiplexed waveforms from DOS adsend
13    TYPE_ERROR(2), // error
14    TYPE_HEARTBEAT(3), // heartbeat
15    TYPE_NANOBUF(4), // single-channel waveforms from nanometrics
16    TYPE_ACK(6), // used by import_ack
17    TYPE_PICK2K(10), // P-wave arrival time (with 4 digit year)
18    TYPE_CODA2K(11), // coda info (plus station code) from pick_ew
19    // TYPE_PICK2(12), // P-wave arrival time from picker & pick_ew
20    // TYPE_CODA2(13), // coda info from picker & pick_ew
21    TYPE_HYP2000ARC(14), // hyp2000 (Y2K hypoinverse) event archive
22    TYPE_H71SUM2K(15), // hypo71-format hypocenter summary msg (with 4-digit year) from
23                       // eqproc/eqprelim
24    // TYPE_HINVARC(17), // hypoinverse event archive msg from eqproc/eqprelim
25    // TYPE_H71SUM(18), // hypo71-format summary msg from eqproc/eqprelim
26    TYPE_TRACEBUF2(19), // single-channel waveforms from NT adsend, getdst2, nano2trace, rcv_ew,
27                        // import_ida...
28    TYPE_TRACEBUF(20), // single-channel waveforms from NT adsend, getdst2, nano2trace, rcv_ew,
29                       // import_ida...
30    TYPE_LPTRIG(21), // single-channel long-period trigger from lptrig & evanstrig
31    TYPE_CUBIC(22), // cubic-format summary msg from cubic_msg
32    TYPE_CARLSTATRIG(23), // single-channel trigger from carlstatrig
33    // TYPE_TRIGLIST(24), // trigger-list msg (used by tracesave modules) from arc2trig, trg_assoc,
34    // carlsubtrig
35    TYPE_TRIGLIST2K(25), // trigger-list msg (with 4-digit year) used by tracesave modules from
36                         // arc2trig, trg_assoc, carlsubtrig
37    TYPE_TRACE_COMP_UA(26), // compressed waveforms from compress_UA
38    TYPE_STRONGMOTION(27), // single-instrument peak accel, peak velocity, peak displacement, spectral
39                           // acceleration
40    TYPE_MAGNITUDE(28), // event magnitude: summary plus station info
41    TYPE_STRONGMOTIONII(29);
42  
43    private int type;
44  
45    private MessageType(int t) {
46      type = t;
47    }
48  
49    public static MessageType fromInt(int i) {
50      for (MessageType m : MessageType.values())
51        if (m.getType() == i)
52          return m;
53  
54      return null;
55    }
56  
57    public int getType() {
58      return type;
59    }
60  }
