1   package gov.usgs.volcanoes.core.legacy.plot.color;
2   
3   import java.awt.Color;
4   
5   /**
6    * A simple utility class to cycle through colors. This is used by the Valve
7    * grapher to cycle through colors of graph lines.
8    *
9    * @author  Dan Cervelli
10   */
11  public class ColorCycler {
12    /** Custom color declarations
13     */
14    private static Color VALVE_BROWN = new Color(165, 42, 42);
15    private static Color VALVE_GOLD = new Color(218, 165, 32);
16    private static Color VALVE_CORAL = new Color(255, 114, 86);
17  
18    /** The colors in the cycle.
19     */
20    private static final Color[] colorCycle =
21        new Color[] {Color.BLUE, Color.GREEN, VALVE_GOLD, VALVE_BROWN, VALVE_CORAL, Color.RED};
22  
23    private int currentColor;
24  
25    /** Generic constructor.
26     */
27    public ColorCycler() {
28      currentColor = 0;
29    }
30  
31    /**
32     * Constructor
33     * @param start Starting color number (see colorCycle array)
34     */
35    public ColorCycler(int start) {
36      currentColor = start;
37    }
38  
39    /** Gets the next color in the cycle.
40     * @return the next color
41     */
42    public Color getNextColor() {
43      return colorCycle[currentColor++ % colorCycle.length];
44    }
45  
46    /** Resets the color cycler
47     */
48    public void reset() {
49      currentColor = 0;
50    }
51  }
