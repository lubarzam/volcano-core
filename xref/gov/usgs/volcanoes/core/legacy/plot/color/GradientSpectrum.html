3   import java.awt.Color;

5   /**
6    * <p>Spectra with gradient color filling.</p>
7    * 
8    * $Log: not supported by cvs2svn $
9    * @author Dan Cervelli
10   */
11  public class GradientSpectrum extends Spectrum {
12    /**
13     * Constructor. Initial parameters is RGB color components for starting and ending color
14     * @param size count of colors in the spectra
15     */
16    public GradientSpectrum(int size, double r1, double g1, double b1, double r2, double g2,
17        double b2) {
18      colors = new Color[size];
19      double dr = (r2 - r1) / (double) size;
20      double dg = (g2 - g1) / (double) size;
21      double db = (b2 - b1) / (double) size;
22      for (int i = 0; i < size; i++)
23        colors[i] = new Color((float) (r1 + dr * i), (float) (g1 + dg * i), (float) (b1 + db * i));
24  
25      paletteBytes = null;
26      palette = null;
27    }
28  }
