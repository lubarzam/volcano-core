79    /**
80     * Create string array with label text according to numberFormat setting.
81     * 
82     * @param t array of double values
83     * @param log if logarithm scale enabled
84     * @param m never used
85     * @param b never used
86     * @return String array the same size with labels text
87     */
88    public static String[] createLabels(double[] t, boolean log, double m, double b) {
89      String[] labels = new String[t.length];
90      for (int i = 0; i < t.length; i++) {
91        if (log) {
92          labels[i] = "10^" + numberFormat.format(t[i]);
93        } else {
94          double v = t[i];
95          double exp = Util.getExp(v);
96          labels[i] =
97              (exp >= 5 ? numberFormat.format(v / Math.pow(10, exp)) + "e" + numberFormat.format(exp)
98                  : numberFormat.format(v));
99        }
100 
101     }
102     return labels;
103   }

105   /**
106    * Create and add to Frame Renderer specialized renderer to process plot title according
107    * configured properties.
108    * 
109    * @param fr Frame renderer to process
110    */
111   private void createTitle(FrameRenderer fr) {
112     if (title == null || title.length() == 0) {
113       return;
114     }
115 
116     switch (titleLocation) {
117       case TOP:
118       case INSET:
119         AxisRenderer ar = fr.getAxis();
120         if (titleFont == null) {
121           titleFont = Font.decode("dialog-plain-12");
122         }
123 
124         TextRenderer label =
125             new TextRenderer(fr.getGraphWidth() - (title.length() * PIXELS_PER_CHARACTER) + 32,
126                 fr.getGraphY() + 16, title, Color.BLACK);
127         label.font = titleFont;
128         if (titleBackground != null) {
129           RectangleRenderer rr = ar.getFrame();
130           rr.color = Color.GRAY;
131           rr = new RectangleRenderer();
132           rr.rect = new Rectangle2D.Double();
133           FontRenderContext frc = new FontRenderContext(new AffineTransform(), false, false);
134           rr.rect.setFrame(titleFont.getStringBounds(title, frc));
135           // rr.rect.x = fr.getGraphX() + 3;
136           rr.rect.y = fr.getGraphY() + 3;
137           rr.rect.x = fr.getGraphWidth() - (title.length() * PIXELS_PER_CHARACTER) + 29;
138           rr.rect.width += 6;
139           rr.rect.height += 2;
140           rr.color = Color.GRAY;
141           rr.backgroundColor = titleBackground;
142           ar.addPostRenderer(rr);
143         }
144         ar.addPostRenderer(label);
145         break;
146       default:
147         break;
148     }
149   }

151   /**
152    * Create and add to Frame Renderer specialized renderer to process X axis according configured
153    * properties.
154    * 
155    * @param fr Frame renderer to process
156    */
157   private void createXAxis(FrameRenderer fr) {
158     AxisRenderer axis = fr.getAxis();
159     boolean doAxis = true;
160     Object[] stt = null;
161     boolean log = false;
162     switch (xAxis) {
163       case NONE:
164         doAxis = false;
165         break;
166 
167       case LOG:
168         log = true;
169         // fall through
170 
171       case LINEAR:
172 
173         String[] st;
174         double[] t;
175 
176         if (log) {
177           int x1 = (int) Math.floor(fr.getMinXAxis());
178           int x2 = (int) Math.ceil(fr.getMaxXAxis());
179 
180           int n = (x2 - x1) * 9;
181 
182           st = new String[n];
183           t = new double[n];
184 
185           int c = 0;
186           double inc;
187           double value;
188 
189           for (int i = x1; i < x2; i++) {
190             inc = Math.pow(10, i);
191 
192             if (i < fr.getMinXAxis() || i > fr.getMaxXAxis()) {
193               t[c] = Double.NaN;
194               st[c] = "";
195             } else {
196               t[c] = i;
197               st[c] = numberFormat.format(Math.pow(10, i)) + " Hz";
198             }
199 
200             c++;
201             for (int j = 1; j < 9; j++) {
202               value = Math.log10(Math.pow(10, i) + inc * j);
203               if (value < fr.getMinXAxis() || value > fr.getMaxXAxis()) {
204                 t[c] = Double.NaN;
205               } else {
206                 t[c] = value;
207               }
208 
209               st[c] = "";
210               c++;
211             }
212           }
213         } else {
214           if (hTicks == -1) {
215             hTicks = fr.getGraphWidth() / 108;
216           }
217           t = SmartTick.autoTick(fr.getMinXAxis(), fr.getMaxXAxis(), hTicks, false);
218           st = new String[t.length];
219           for (int i = 0; i < t.length; i++) {
220             double val = t[i];
221             st[i] = numberFormat.format(val) + " Hz";
222           }
223         }
224 
225         int count = 0;
226         for (double d : t) {
227           if (!Double.isNaN(d)) {
228             count++;
229           }
230         }
231         double[] smallT = new double[count];
232         String[] smallSt = new String[count];
233 
234         count = 0;
235         for (int i = 0; i < t.length; i++) {
236           if (!Double.isNaN(t[i])) {
237             smallT[count] = t[i];
238             smallSt[count++] = st[i];
239           }
240         }
241 
242         stt = new Object[] {smallT, smallSt};
243         break;
244 
245       case TIME:
246         if (hTicks == -1) {
247           hTicks = fr.getGraphWidth() / 108;
248         }
249         stt = SmartTick.autoTimeTick(fr.getMinXAxis(), fr.getMaxXAxis(), hTicks);
250         break;
251       default:
252         break;
253     }
254     if (doAxis) {
255       if (stt != null) {
256         if (xAxisBottomTickLength > 0) {
257           axis.createBottomTicks((double[]) stt[0], xAxisBottomTickLength, Color.BLACK);
258         }
259         if (xAxisTopTickLength > 0) {
260           axis.createTopTicks((double[]) stt[0], xAxisTopTickLength, Color.BLACK);
261         }
262         if (xAxisGrid != Grid.NONE) {
263           axis.createVerticalGridLines((double[]) stt[0]);
264         }
265 
266         if (xAxisLabels) {
267           axis.createBottomTickLabels((double[]) stt[0], (String[]) stt[1]);
268         }
269         if (xAxisLabel != null) {
270           addLabel(fr, yAxisLabel, Location.BOTTOM);
271         }
272         if (xUnit != null) {
273           axis.setBottomLabelAsText(xUnit);
274         }
275       }
276     }
277   }

279   /**
280    * Create and add to FrameRenderer specialized renderer to process Y axis according configured
281    * properties.
282    * 
283    * @param fr Frame renderer to process
284    */
285   private void createYAxis(FrameRenderer fr) {
286     AxisRenderer axis = fr.getAxis();
287     boolean doAxis = true;
288     double[] yt = null;
289     double[] ytm = null;
290     boolean log = false;
291     switch (yAxis) {
292       case NONE:
293         break;
294       case LOG:
295         log = true;
296       case LINEAR:
297         if (log) {
298           vTicks = (int) (fr.getMaxYAxis() - fr.getMinYAxis());
299         } else if (vTicks == -1) {
300           vTicks = fr.getGraphHeight() / 24;
301         }
302 
303         yt = SmartTick.autoTick(fr.getMinYAxis(), fr.getMaxYAxis(), vTicks, false);
304 
305         ytm = new double[yt.length];
306         for (int i = 0; i < yt.length; i++) {
307           ytm[i] = (yt[i] - fr.getYAxisOffset()) / fr.getYAxisMult();
308         }
309         break;
310       default:
311         break;
312     }
313 
314     if (doAxis) {
315       if (yt != null) {
316         if (yAxisLeftTickLength > 0) {
317           axis.createLeftTicks(ytm, yAxisLeftTickLength, Color.BLACK);
318         }
319         if (yAxisRightTickLength > 0) {
320           axis.createRightTicks(ytm, true, yAxisRightTickLength, Color.BLACK);
321         }
322         if (yAxisGrid != Grid.NONE) {
323           axis.createHorizontalGridLines(ytm);
324         }
325         if (yAxisLabels) {
326           axis.createLeftTickLabels(ytm,
327               createLabels(yt, log, fr.getYAxisMult(), fr.getYAxisOffset()));
328         }
329         if (yAxisLabel != null) {
330           addLabel(fr, yAxisLabel, Location.LEFT);
331         }
332         if (yUnit != null) {
333           axis.setLeftLabelAsText(yUnit);
334         }
335       }
336     }
337   }

351   /**
352    * Add to FrameRenderer specialized renderer to draw label.
353    * 
354    * @param fr Frame renderer to process
355    * @param text label text
356    * @param loc label location
357    * @param font label font
358    * @param bgColor label color
359    * @return Created TextRenderer to process label
360    */
361   public static TextRenderer addLabel(FrameRenderer fr, String text, Location loc, Font font,
362       Color bgColor) {
363     TextRenderer tr = new TextRenderer();
364     tr.text = text;
365     if (font != null) {
366       tr.font = font;
367     }
368     if (bgColor != null) {
369       tr.backgroundColor = bgColor;
370     }
371     switch (loc) {
372       case LEFT:
373         tr.x = 6;
374         tr.y = fr.getGraphY() + fr.getGraphHeight() / 2;
375         tr.horizJustification = TextRenderer.CENTER;
376         tr.orientation = -90.0f;
377         fr.getAxis().addRenderer(tr);
378         break;
379       case RIGHT:
380         tr.x = fr.getGraphX() + fr.getGraphWidth() + 15;
381         tr.y = fr.getGraphY() + fr.getGraphHeight() / 2;
382         tr.horizJustification = TextRenderer.CENTER;
383         tr.orientation = 90.0f;
384         fr.getAxis().addRenderer(tr);
385         break;
386       case BOTTOM:
387         tr.x = fr.getGraphX() + fr.getGraphWidth() / 2;
388         tr.y = fr.getGraphY() + fr.getGraphHeight() - 6;
389         tr.horizJustification = TextRenderer.CENTER;
390         fr.getAxis().addRenderer(tr);
391         break;
392       default:
393         break;
394     }
395     return tr;
396   }

398   /**
399    * Process FrameRenderer and add configured renderers for axises and title.
400    * 
401    * @param fr Frame renderer to process
402    */
403   public void decorate(FrameRenderer fr) {
404     if (hasAxis) {
405       AxisRenderer ar = new AxisRenderer(fr);
406       ar.createDefault();
407       if (!hasFrame) {
408         ar.setFrame(null);
409       }
410       fr.setAxis(ar);
411 
412       createXAxis(fr);
413       createYAxis(fr);
414       createTitle(fr);
415     }
416   }
