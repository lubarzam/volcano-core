15  /**
16   * <p>

23  public class GeoImage implements Comparable<GeoImage> {
24    private String filename;
25    private GeoRange range;
26    private int pixelWidth;

28    private double minScale;

30    private int priority = 0;

34    /**
35     * Default constructor
36     */
37    public GeoImage() {}

39    /**
40     * Constructor
41     * 
42     * @param is
43     *            comma-separated parameter string: pixel width and height, and
44     *            coordinates for west, east, south and north edge.
45     */
46    public GeoImage(String is) {
47      String[] ss = is.split(",");
48      filename = ss[0].trim();
49      pixelWidth = Integer.parseInt(ss[1].trim());
50      pixelHeight = Integer.parseInt(ss[2].trim());
51  
52      double w = Double.parseDouble(ss[3].trim());
53      double e = Double.parseDouble(ss[4].trim());
54      double s = Double.parseDouble(ss[5].trim());
55      double n = Double.parseDouble(ss[6].trim());
56      range = new GeoRange(w, e, s, n);
57      image = null;
58      if (ss.length > 7) {
59        minScale = Double.parseDouble(ss[7].trim());
60        maxScale = Double.parseDouble(ss[8].trim());
61      } else {
62        minScale = Double.NaN;
63        maxScale = Double.NaN;
64      }
65      if (ss.length == 10) {
66        priority = Integer.parseInt(ss[9].trim());
67      }
68    }

71     * Constructor
72     * 
73     * @param root
74     *            base directory to store images
75     * @param is
76     *            comma-separated parameter string: pixel width and height, and

78     */
79    public GeoImage(String root, String is) {
80      this(is);
81      filename = root + File.separatorChar + filename;
82    }

90     * 4. Toolkit.createImage()

92     * TJP adds: nonetheless, this happens once per mouse click and does not 
93     * need to be highly optimized to maintain acceptable responsiveness. 
94     * Compatibility and maintainability trumps.
95     */
96    private void loadImage() {
97      if (filename.equals("memory"))
98        return;
99  
100     CodeTimer ct = new CodeTimer("imageLoad: " + filename);
101 
102     try {
103 
104       /*
105        * I'm pulling com.sun.image.codec.jpeg
106        * 
107        * Using it is in poor form. Sun continues to threaten to remove the
108        * classes, and they are unlikely to be supported in non-sun JVM's.
109        * Yes, it's faster. However, I'm not sure it matters. Image loads
110        * are infrequent, and adding even 100ms to the time taken by a map
111        * load is tolerable.
112        * 
113        * Code left behind in case I'm mistaken.
114        */
115       // if (filename.toLowerCase().endsWith("jpg"))
116       // {

121       // FileInputStream(filename));

124       // else
125       {

127       }
128     } catch (Exception ex) {
129       ex.printStackTrace();
130     }
131     ct.stopAndReport();
132   }

139       image = null;

142   /**
143    * Create GeoImage stored in RAM
144    * 
145    * @param pi
146    *            image to draw
147    * @param r
148    *            edge coordinates
149    */
150   public static GeoImage createMemoryImage(BufferedImage pi, GeoRange r) {
151     GeoImage result = new GeoImage();
152     result.filename = "memory";
153     result.range = new GeoRange(r);
154     result.image = pi;
155     result.pixelWidth = pi.getWidth();
156     result.pixelHeight = pi.getHeight();
157     return result;
158   }

245   public boolean inScale(double sc, double area) {
246     if (Double.isNaN(minScale) && Double.isNaN(maxScale)) {
247       return area > 0.8;
248     } else
249       return sc >= minScale && sc < maxScale;
250   }
