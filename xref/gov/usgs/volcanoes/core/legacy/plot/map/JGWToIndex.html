5   import java.awt.image.BufferedImage;
6   import java.io.File;
7   import java.io.IOException;
8   
9   import javax.imageio.ImageIO;
10  
11  /**
12   * <p>Scans dir for .jgw files and send to stdout strings as follow:</p>
13   * <p>name, width, height, west_coord, east_coord, south_coord, north_coord</p>
14   * <p>Syntax is: JGWToIndex [dir_name]</p>
15   * <p>Default is current directory.</p>
16   * 
17   * @author Dan Cervelli
18   */
19  public class JGWToIndex {
20    public static void main(String[] args) throws IOException {
21      String dir = ".";
22      if (args.length > 0)
23        dir = args[0];
24      String[] files = new File(dir).list();
25      if (files == null) {
26        throw new RuntimeException("No file list.");
27      }
28      for (String fn : files) {
29        if (!fn.endsWith(".jgw"))
30          continue;
31        ResourceReader rr = ResourceReader.getResourceReader(fn);
32        double[] jgw = new double[6];
33        for (int i = 0; i < 6; i++)
34          jgw[i] = Double.parseDouble(rr.nextLine());
35        String jpg = fn.substring(0, fn.length() - 2) + "pg";
36        BufferedImage image = ImageIO.read(new File(jpg));
37        double w = jgw[4];
38        double e = w + image.getWidth() * jgw[0];
39        double n = jgw[5];
40        double s = n + image.getHeight() * jgw[3];
41        System.out.printf("%s, %d, %d, %f, %f, %f, %f\n", jpg, image.getWidth(), image.getHeight(), w,
42            e, s, n);
43      }
44    }
45  }
