23  /**
24   * <p>
25   * Request whole earth image from Web Map Service Server 1000x800 size in jpeg
26   * format
27   * </p>
28   * 
29   * @author Dan Cervelli
30   */
31  public class WMSImageGrabber {
32    protected String server = "http://wms.jpl.nasa.gov/wms.cgi";
33    protected final static Logger LOGGER = LoggerFactory.getLogger(WMSImageGrabber.class);
34    protected BufferedReader in;
35    protected InputStream inputStream;
36  
37    private void getReader(String resource) throws MalformedURLException, IOException {
38      if (resource.indexOf("://") != -1) {
39        URL url = new URL(resource);
40        inputStream = url.openStream();
41        in = new BufferedReader(new InputStreamReader(inputStream));
42      } else {
43        in = new BufferedReader(new FileReader(resource));
44      }
45    }
46  
47    public void getCapabilities() {
48      String request = server + "?REQUEST=GetCapabilities";
49      try {
50        getReader(request);
51        String s;
52        while ((s = in.readLine()) != null)
53          LOGGER.info(s);
54      } catch (Exception e) {
55        e.printStackTrace();
56      }
57    }
58  
59    public BufferedImage getImage() {
60      BufferedImage image = null;
61      String request = server + "?REQUEST=GetMap&LAYERS=BMNG&STYLES=Sep&WIDTH=1000&HEIGHT=800"
62          + "&EXCEPTIONS=application/vnd.ogc.se_inimage&SRS=EPSG%3A4326&BBOX=-180,-90,180,90&FORMAT=image/jpeg";
63      try {
64        getReader(request);
65        // JPEGImageDecoder codec =
66        // JPEGCodec.createJPEGDecoder(inputStream);
67        // image = codec.decodeAsBufferedImage();
68  
69        // TODO: make sure this works
70        image = ImageIO.read(inputStream);
71      } catch (Exception e) {
72        e.printStackTrace();
73      }
74      return image;
75    }

77    /**
78     * Main method. Open screen frame with generated image.
79     * 
80     * @param args
81     *            none
82     */
83    public static void main(String[] args) {
84      WMSImageGrabber w = new WMSImageGrabber();
85      // w.getCapabilities();
86      final BufferedImage bi = w.getImage();
87  
88      JFrame f = new JFrame("Test");
89      f.setSize(1200, 1000);
90      f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
91      f.setLocationRelativeTo(null);
92      f.setContentPane(new JButton(new ImageIcon(bi)));
93      f.setVisible(true);
94    }
