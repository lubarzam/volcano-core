17  /**
18   * <p>This is a Renderer that draws vectors with error ellipses.  Access the public member variables directly 
19   * to set renderer specifications.</p>
20   * 
21   * @author Dan Cervelli
22   */
23  public class EllipseVectorRenderer implements Renderer {
24    protected final static Logger LOGGER = LoggerFactory.getLogger(EllipseVectorRenderer.class);
25  
26    /** The x coordinate for the vector tail.
27     */
28    public double x;
29  
30    /** The y coordinate for the vector tail.
31     */
32    public double y;
33  
34    /** The x coordinate of the vector head.
35     */
36    public double u;
37    /** The y coordinate of the vector head.
38     */
39    public double v;
40  
41    /** The height of the vertical component.
42    */
43    public double z;
44  
45    public boolean drawEllipse = true;
46    public double ellipseWidth;
47    public double ellipseHeight;
48    public double ellipseOrientation;
49    public double sigZ;
50  
51    private double scale = 1;
52  
53    /** Flag for displaying horizontal components (x and y).
54     */
55    public boolean displayHoriz = true;
56  
57    /** Flag for displaying vertical component (z).
58    */
59    public boolean displayVert = true;
60  
61    public Color colorHoriz = Color.RED;
62    public Color colorVert = Color.BLUE;
63  
64    /** Coordinate transformer for this vector.
65     */
66    public FrameRenderer frameRenderer;
67  
68    private static final double[] SCALES = new double[] {100000, 50000, 20000, 10000, 5000, 2000,
69        1000, 500, 200, 100, 50, 20, 10, 5, 2, 1, 0.5, 0.2, 0.1, 0.05, 0.02, 0.01};
70  
71    private static GeneralPath arrowhead;
72  
73    static {
74      arrowhead = new GeneralPath();
75  
76      arrowhead.moveTo(0, 0);
77      arrowhead.lineTo(0.5f, -1.5f);
78      arrowhead.lineTo(0, -1.3f);
79      arrowhead.lineTo(-0.5f, -1.5f);
80  
81      arrowhead.closePath();
82    }
83  
84    /** Generic empty constructor.
85     */
86    public EllipseVectorRenderer() {}
87  
88    /** Gets the best scale for the vector based on a specified magnitude.
89     * @param maxMag the magnitude to base the scale on
90     * @return the best scale
91     */
92    public static double getBestScale(double maxMag) {
93      double dm = 1E300;
94      int mi = -1;
95      for (int i = 0; i < SCALES.length; i++)
96        if (Math.abs(maxMag - SCALES[i]) < dm) {
97          dm = Math.abs(maxMag - SCALES[i]);
98          mi = i;
99        }
100 
101     return SCALES[mi];
102   }
103 
104   /** Sets the head position based on azimuth and magnitude.
105    * @param az the azimuth
106    * @param mag the magnitude
107    */
108   public void setUVByAngleMag(double az, double mag) {
109     u = x + Math.cos(Math.toRadians(az)) * mag;
110     v = y + Math.sin(Math.toRadians(az)) * mag;
111   }
112 
113   /** Gets the azimuth of this vector.
114    * @return the azimuth
115    */
116   public double getAngle() {
117     return Math.toDegrees(Math.atan2(-u, -v));
118   }
119 
120   /** Gets the magnitude of this vector.
121    * @return the magnitude
122    */
123   public double getMag() {
124     return Math.sqrt(u * u + v * v);
125   }
126 
127   /** Sets the scale for this vector.
128    * @param s the scale
129    */
130   public void setScale(double s) {
131     scale = s;
132   }
133 
134   /** Gets the scale for this vector.
135    * @return the scale
136    */
137   public double getScale() {
138     return scale;
139   }
140 
141   /** Renders the vector and ellipse.
142    * @param g the graphics object upon which to render
143    */
144   public void render(Graphics2D g) {
145     AffineTransform origAT = g.getTransform();
146     Color origColor = g.getColor();
147     Paint origPaint = g.getPaint();
148     Stroke origStroke = g.getStroke();
149 
150     Object aa = g.getRenderingHint(RenderingHints.KEY_ANTIALIASING);
151     g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
152 
153     Line2D.Double line = new Line2D.Double();
154 
155     // draw blue vertical vector
156     if (displayVert) {
157       line.setLine(frameRenderer.getXPixel(x), frameRenderer.getYPixel(y),
158           frameRenderer.getXPixel(x), frameRenderer.getYPixel(y + z * scale));
159       g.setColor(colorVert);
160       g.draw(line);
161       g.translate(frameRenderer.getXPixel(x), frameRenderer.getYPixel(y + z * scale));
162       if (z > 0)
163         g.rotate(Math.toRadians(180));
164       g.setStroke(new BasicStroke(0.1f));
165       double headScaleVert = Math.abs(z * scale * frameRenderer.getYScale() * 0.2);
166       headScaleVert = Math.min(12, headScaleVert);
167       headScaleVert = Math.max(5, headScaleVert);
168       g.scale(headScaleVert, headScaleVert);
169       g.fill(arrowhead);
170 
171       // draw blue vertical error bars
172       if (drawEllipse) {
173         g.setTransform(origAT);
174         g.setStroke(new BasicStroke(2.0f));
175         g.setColor(colorVert);
176         // logger.fine(z + " " + sigZ);
177         line.setLine(frameRenderer.getXPixel(x) - 3,
178             frameRenderer.getYPixel(y + (z - sigZ) * scale), frameRenderer.getXPixel(x) + 3,
179             frameRenderer.getYPixel(y + (z - sigZ) * scale));
180         g.draw(line);
181         line.setLine(frameRenderer.getXPixel(x) - 3,
182             frameRenderer.getYPixel(y + (z + sigZ) * scale), frameRenderer.getXPixel(x) + 3,
183             frameRenderer.getYPixel(y + (z + sigZ) * scale));
184         g.draw(line);
185       }
186     }
187 
188     // draw red vector
189     if (displayHoriz) {
190       g.setTransform(origAT);
191       g.setStroke(origStroke);
192       line.setLine(frameRenderer.getXPixel(x), frameRenderer.getYPixel(y),
193           frameRenderer.getXPixel(x + u * scale), frameRenderer.getYPixel(y + v * scale));
194       // logger.fine("render - x:" + frameRenderer.getXPixel(x) + "/y:" + frameRenderer.getXPixel(y)
195       // + "/u:" + frameRenderer.getXPixel(x + u * scale) + "/v:" + frameRenderer.getXPixel(y + v *
196       // scale));
197       g.setColor(colorHoriz);
198       g.draw(line);
199       g.translate(frameRenderer.getXPixel(x + u * scale), frameRenderer.getYPixel(y + v * scale));
200       g.rotate(Math.toRadians(getAngle()));
201       // logger.fine("angle:" + getAngle());
202       // logger.fine("radians:" + Math.toRadians(getAngle()));
203       g.setStroke(new BasicStroke(0.1f));
204       double headScale = getMag() * scale * frameRenderer.getXScale() * 0.2;
205       headScale = Math.min(12, headScale);
206       headScale = Math.max(5, headScale);
207       g.scale(headScale, headScale);
208       g.fill(arrowhead);
209 
210       // draw error ellipse
211       if (drawEllipse) {
212         g.setTransform(origAT);
213         g.setStroke(origStroke);
214         g.setColor(colorHoriz);
215         Ellipse2D.Double e =
216             new Ellipse2D.Double(0, 0, ellipseWidth * scale, ellipseHeight * scale);
217         e.x = -e.width / 2;
218         e.y = -e.height / 2;
219 
220         g.translate(frameRenderer.getXPixel(x + u * scale), frameRenderer.getYPixel(y + v * scale));
221         g.rotate(ellipseOrientation);
222         g.scale(frameRenderer.getXScale(), frameRenderer.getYScale());
223         g.setStroke(new BasicStroke((float) (1.0 / (float) frameRenderer.getXScale())));
224 
225         g.draw(e);
226         g.setTransform(origAT);
227       }
228     }
229 
230     g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, aa);
231     g.setStroke(origStroke);
232     g.setPaint(origPaint);
233     g.setColor(origColor);
234     g.setTransform(origAT);
235   }
