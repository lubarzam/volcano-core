172   /**
173    * Get pixel X coordinate, taking into account raw structure of helicorder.
174    * 
175    * @param x time value
176    * @return pixel x coordinate
177    */
178   public double helicorderGetXPixel(double x) {
179     while (x < hcMinX) {
180       x += timeChunk;
181     }
182     double tx = (x - hcMinX) % timeChunk;
183     return (tx * helicorderGetXScale()) + graphX;
184   }

186   /**
187    * Get pixel Y coordinate, taking into account raw structure of helicorder.
188    * 
189    * @param x time value
190    * @param y data value
191    * @return pixel y coordinate
192    */
193   public double helicorderGetYPixel(double x, double y) {
194     int row = numRows - (int) ((x - hcMinX) / timeChunk) - 1;
195     return graphY + graphHeight - ((y - hcMinY) * helicorderGetYScale())
196         - ((double) row * rowHeight);
197   }

428   /**
429    * Render graph.
430    * 
431    * @param g where to draw to
432    */
433   public void render(Graphics2D g) {
434 
435     if (data == null) {
436       return;
437     }
438 
439     if (decorator != null) {
440       decorator.decorate(this);
441     }
442 
443     AffineTransform origAT = g.getTransform();
444     Color origColor = g.getColor();
445     Shape origClip = g.getClip();
446 
447     if (axis != null) {
448       axis.render(g);
449     }
450 
451     g.setClip(new Rectangle(graphX + 1, graphY + 1, graphWidth - 1, graphHeight - 1));
452 
453     if (!noData) {
454 
455       DoubleMatrix2D j2k = data.getTimes();
456       DoubleMatrix2D min = data.getMin();
457       DoubleMatrix2D max = data.getMax();
458 
459       double t1, x, y, w, h, ymax, ymin;
460       double t2 = 0;
461 
462       double bias = Double.NaN;
463       int lastRow = -1;
464       int numRows = j2k.rows();
465       Color lastColor = null;
466       if (color != null) {
467         g.setColor(color);
468       }
469       for (int j = 0; j < numRows; j++) {
470         t1 = j2k.getQuick(j, 0);
471         int k = ((int) ((t1 - hcMinX) / timeChunk)) % defaultColors.length;
472         if (k < 0) {
473           k = 0;
474         }
475 
476         if (color == null) {
477           if (lastColor != defaultColors[k]) {
478             g.setColor(defaultColors[k]);
479             lastColor = defaultColors[k];
480           }
481         }
482 
483         t2 = t1 + 1;
484 
485         int r = getRow(t2);
486         if (r != lastRow) {
487           double st = hcMinX + r * timeChunk;
488           bias = data.getBiasBetween(st, st + timeChunk);
489           lastRow = r;
490         }
491 
492         x = helicorderGetXPixel(t1);
493         w = helicorderGetXPixel(t2) - x;
494         ymax = max.getQuick(j, 0);
495         ymin = min.getQuick(j, 0);
496 
497         if (ymax == Integer.MIN_VALUE || ymin == Integer.MIN_VALUE) {
498           continue;
499         }
500 
501         ymax -= bias;
502         ymin -= bias;
503 
504         if (showClip && (ymax >= clipValue || ymin <= -clipValue)) {
505           lastClipTime = t1;
506           if (color == null) {
507             if (lastColor != Color.red) {
508               g.setColor(Color.red);
509               lastColor = Color.red;
510             }
511           }
512         }
513 
514         if (ymax > clipValue) {
515           ymax = clipValue;
516         }
517 
518         if (ymin < -clipValue) {
519           ymin = -clipValue;
520         }
521 
522         y = helicorderGetYPixel(t1, ymax);
523         h = helicorderGetYPixel(t1, ymin) - y;
524         int hgt = (int) (h + 1);
525         if (hgt < 1) {
526           hgt = 1;
527         }
528         if (forceCenter) {
529           y = helicorderGetYPixel(t1, 0) - hgt / 2;
530         }
531         g.fillRect((int) (x + 1), (int) (y + 1), (int) (w + 1), hgt);
532       }
533 
534       g.setClip(origClip);
535       g.setColor(origColor);
536       g.setTransform(origAT);
537 
538       if (largeChannelDisplay && channel != null) {
539         Font oldFont = g.getFont();
540         g.setFont(LARGE_FONT);
541         String c = channel.replace('_', ' ');
542 
543         FontMetrics fm = g.getFontMetrics();
544         Font f = g.getFont();
545         float s = f.getSize();
546         int width = fm.stringWidth(c);
547         while ((width / (double) graphWidth > .5) && (--s > 1)) {
548           g.setFont(f.deriveFont(s));
549           fm = g.getFontMetrics();
550           width = fm.stringWidth(c);
551         }
552         int height = fm.getAscent() + fm.getDescent();
553         int lw = width + 20;
554 
555         if (alertClip && lastClipTime > t2 - alertClipTimeout && clipWav != null) {
556           g.setColor(Color.red);
557           if ((lastClipTime > lastClipAlert + alertClipTimeout)) {
558             lastClipAlert = t2;
559             playClipAlert();
560           }
561         } else if (alertClip) {
562           g.setColor(new Color(128, 255, 128, 192));
563         } else {
564           g.setColor(new Color(255, 255, 255, 192));
565         }
566 
567         g.fillRect(graphX + graphWidth / 2 - lw / 2, 3, lw, height);
568         g.setColor(Color.black);
569         g.drawRect(graphX + graphWidth / 2 - lw / 2, 3, lw, height);
570 
571         g.drawString(c, graphX + graphWidth / 2 - width / 2, height - fm.getDescent());
572         g.setFont(oldFont);
573 
574       } else if (alertClip && lastClipTime > t2 - alertClipTimeout && clipWav != null) {
575         if ((lastClipTime > lastClipAlert + alertClipTimeout)) {
576           lastClipAlert = t2;
577           playClipAlert();
578         }
579       }
580     }
581 
582     g.setClip(origClip);
583     g.setColor(origColor);
584     g.setTransform(origAT);
585 
586     if (getLegendRenderer() != null) {
587       getLegendRenderer().render(g);
588     }
589 
590     if (axis != null) {
591       axis.postRender(g);
592     }
593   }

602   /**
603    * Minimum decorator.
604    */
605   class MinimumDecorator extends FrameDecorator {
606     public void decorate(FrameRenderer fr) {
607       axis = new AxisRenderer(fr);
608       axis.createDefault();
609 
610       int minutes = (int) Math.round(timeChunk / 60.0);
611       int majorTicks = minutes;
612       if (minutes > 30 && minutes < 180) {
613         majorTicks = minutes / 5;
614       } else if (minutes >= 180 && minutes < 360) {
615         majorTicks = minutes / 10;
616       } else if (minutes >= 360) {
617         majorTicks = minutes / 20;
618       }
619       double[] mjt = SmartTick.intervalTick(minX, maxX, majorTicks);
620       if (xTickMarks) {
621         axis.createBottomTicks(null, mjt);
622         axis.createTopTicks(null, mjt);
623         axis.createVerticalGridLines(mjt);
624       }
625 
626       String[] btl = new String[mjt.length];
627       for (int i = 0; i < mjt.length; i++) {
628         btl[i] = Long.toString(Math.round(mjt[i] / 60.0));
629       }
630 
631       double[] labelPosLr = new double[numRows];
632       String[] leftLabelText = new String[numRows];
633       DateFormat timeFormat = new SimpleDateFormat("HH:mm");
634       DateFormat dayFormat = new SimpleDateFormat("yyyy-MM-dd");
635       timeFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
636       dayFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
637       double pixelsPast = 0;
638       double pixelsPerRow = graphHeight / numRows;
639       for (int i = numRows - 1; i >= 0; i--) {
640         pixelsPast += pixelsPerRow;
641         labelPosLr[i] = i + 0.5;
642         // TODO: fix
643         // java.util.Date dtz = Util.j2KToDate(hcMaxX - (i + 1) * timeChunk + timeZoneOffset *
644         // 3600);
645         java.util.Date dtz = J2kSec.asDate(hcMaxX - (i + 1) * timeChunk);
646         String ftl = timeFormat.format(dtz);
647 
648         leftLabelText[i] = null;
649         if (pixelsPast > 20) {
650           leftLabelText[i] = ftl;
651           pixelsPast = 0;
652         }
653       }
654       if (yTickValues) {
655         axis.createLeftTickLabels(labelPosLr, leftLabelText);
656       }
657       if (xUnits) {
658         axis.addRenderer(new TextRenderer(graphX, graphY - 3,
659             channel + ", " + dayFormat.format(J2kSec.asDate(hcMaxX))));
660       }
661 
662       double[] hg = new double[numRows - 1];
663       for (int i = 0; i < numRows - 1; i++) {
664         hg[i] = i + 1.0;
665       }
666 
667       if (yTickMarks) {
668         axis.createHorizontalGridLines(hg);
669       }
670 
671       axis.setBackgroundColor(Color.white);
672     }
673   }

678   class StandardDecorator extends FrameDecorator {
679     public void decorate(FrameRenderer fr) {
680       axis = new AxisRenderer(fr);
681       axis.createDefault();
682       // if (numRows <= 0)
683       // return;
684 
685       int minutes = (int) Math.round(timeChunk / 60.0);
686       int majorTicks = minutes;
687       if (minutes > 30 && minutes < 180) {
688         majorTicks = minutes / 5;
689       } else if (minutes >= 180 && minutes < 360) {
690         majorTicks = minutes / 10;
691       } else if (minutes >= 360) {
692         majorTicks = minutes / 20;
693       }
694       double[] mjt = SmartTick.intervalTick(minX, maxX, majorTicks);
695 
696       int minorTicks = 0;
697       if (minutes <= 30) {
698         minorTicks = (int) Math.round(timeChunk / 10.0);
699       } else if (minutes > 30 && minutes <= 180) {
700         minorTicks = minutes;
701       } else if (minutes > 180) {
702         minorTicks = minutes / 5;
703       }
704       double[] mnt = SmartTick.intervalTick(minX, maxX, minorTicks);
705 
706       // x axis decorations
707       if (showDecorator) {
708         if (xUnits) {
709           axis.setBottomLabelAsText("+ Minutes");
710         }
711         if (xTickValues) {
712           String[] btl = new String[mjt.length];
713           for (int i = 0; i < mjt.length; i++) {
714             btl[i] = Long.toString(Math.round(mjt[i] / 60.0));
715           }
716           axis.createBottomTickLabels(mjt, btl);
717         }
718       }
719       if (xTickMarks) {
720         axis.createBottomTicks(mjt, mnt);
721         axis.createTopTicks(mjt, mnt);
722         axis.createVerticalGridLines(mjt);
723       }
724 
725       double[] labelPosLr = new double[numRows];
726       String[] leftLabelText = new String[numRows];
727       String[] rightLabelText = new String[numRows];
728 
729       DateFormat utcTimeFormat = new SimpleDateFormat("HH:mm");
730       utcTimeFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
731 
732       DateFormat utcDayFormat = new SimpleDateFormat("MM-dd");
733       utcDayFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
734 
735       DateFormat localTimeFormat = new SimpleDateFormat("HH:mm");
736       localTimeFormat.setTimeZone(timeZone);
737 
738       DateFormat localDayFormat = new SimpleDateFormat("MM-dd");
739       localDayFormat.setTimeZone(timeZone);
740 
741       double timeOffset = timeZone.getOffset((long) Time.j2kToEw(getViewEndTime())) / 1000;
742 
743       double pixelsPast = 0;
744       double pixelsPerRow = graphHeight / numRows;
745       String lastDayL = "";
746       String lastDayR = "";
747       for (int i = numRows - 1; i >= 0; i--) {
748         pixelsPast += pixelsPerRow;
749         labelPosLr[i] = i + 0.5;
750         double j2kStart = hcMaxX - (i + 1) * timeChunk;
751         double j2kEnd = j2kStart + timeChunk;
752 
753         Date start = J2kSec.asDate(j2kStart);
754         String localTime = localTimeFormat.format(start);
755         String localDay = localDayFormat.format(start);
756 
757         Date end = J2kSec.asDate(j2kEnd);
758         String utcTime = utcTimeFormat.format(end);
759         String utcDay = utcDayFormat.format(end);
760 
761         leftLabelText[i] = null;
762         if (!localDay.equals(lastDayL)) {
763           leftLabelText[i] = localDay + "           ";
764         }
765 
766         if (timeOffset != 0 && !utcDay.equals(lastDayR)) {
767           rightLabelText[i] = "           " + utcDay;
768         }
769 
770         lastDayL = localDay;
771         lastDayR = utcDay;
772 
773         if (pixelsPast > 20) {
774           if (leftLabelText[i] != null) {
775             leftLabelText[i] = localDay + " " + localTime;
776           } else {
777             leftLabelText[i] = localTime;
778           }
779 
780           if (timeOffset != 0) {
781             if (rightLabelText[i] != null) {
782               rightLabelText[i] = utcTime + " " + utcDay;
783             } else {
784               rightLabelText[i] = utcTime;
785             }
786           }
787           pixelsPast = 0;
788         }
789       }
790       if (yTickValues) {
791         axis.createLeftTickLabels(labelPosLr, leftLabelText);
792         axis.createRightTickLabels(labelPosLr, rightLabelText);
793       }
794       if (showDecorator && xUnits) {
795         boolean inDst = timeZone.inDaylightTime(J2kSec.asDate(getViewEndTime()));
796         axis.setBottomLeftLabelAsText(
797             "Time (" + timeZone.getDisplayName(inDst, TimeZone.SHORT) + ")");
798         if (timeOffset != 0) {
799           axis.setBottomRightLabelAsText("Time (UTC)");
800         }
801       }
802 
803       double[] hg = new double[numRows - 1];
804       for (int i = 0; i < numRows - 1; i++) {
805         hg[i] = i + 1.0;
806       }
807 
808       if (yTickMarks) {
809         axis.createHorizontalGridLines(hg);
810       }
811       axis.setBackgroundColor(Color.white);
812     }
813   }

822   /**
823    * Play clip alert.
824    */
825   private void playClipAlert() {
826     Runnable r = new Runnable() {
827       public void run() {
828         File soundFile = new File(clipWav);
829         AudioInputStream audioInputStream = null;
830         try {
831           audioInputStream = AudioSystem.getAudioInputStream(soundFile);
832           AudioFormat audioFormat = audioInputStream.getFormat();
833           DataLine.Info info = new DataLine.Info(SourceDataLine.class, audioFormat);
834 
835           SourceDataLine line = (SourceDataLine) AudioSystem.getLine(info);
836           line.open(audioFormat);
837           line.start();
838 
839           int nBytesRead = 0;
840           byte[] abData = new byte[1024];
841           while (nBytesRead != -1) {
842             nBytesRead = audioInputStream.read(abData, 0, abData.length);
843             if (nBytesRead >= 0) {
844               line.write(abData, 0, nBytesRead);
845             }
846           }
847 
848           line.drain();
849           line.close();
850         } catch (Exception e) {
851           //
852         }
853       }
854     };
855     Thread t = new Thread(r);
856     t.start();
857   }
