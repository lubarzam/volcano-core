22  /**
23   * <p>Histogram2DRenderer is a FrameRenderer that renders a 2-D histogram.
24   * The histogram code is from the Colt library.  See the application overview

30  public class Histogram2DRenderer extends FrameRenderer {
31    private Histogram2D histogram;

37    /** Constructs a new Histogram2DRenderer for renderering the specified 
38     * histrogram.
39     * @param hist the histogram
40     */
41    public Histogram2DRenderer(Histogram2D hist) {
42      histogram = hist;
43      renderers = new ArrayList<Renderer>();
44      log = false;
45      spectrum = Jet.getInstance();
46      inverted = false;
47    }

73    /** Gets a renderer that draws the proper scale/key/legend for this
74     * Histogram2DRenderer based on its current settings.  This could made a 
75     * bit more intelligent -- it makes some dumb ticks.
76     *
77     * @param oX the x-pixel location for the scale
78     * @param oY the y-pixel location for the scale
79     * @return the scale renderer
80     */
81    public Renderer getScaleRenderer(final int oX, final int oY) {
82      return new Renderer() {
83  
84        /** The render function for the scale renderer.
85         * @param g the graphics object upon which to render
86         */
87        public void render(Graphics2D g) {
88          Rectangle2D.Double rect = new Rectangle2D.Double();
89  
90          spectrum.renderScale(g, oX, oY, 400, 10, false, true);
91          g.setPaint(Color.BLACK);
92          rect.setRect(oX - 1, oY - 1, 402, 11);
93          g.draw(rect);
94          int[] mmb = histogram.minMaxBins();
95          double min = 0;
96          double max = 0;
97          double dm = 0;
98          if (log) {
99            min = (histogram.binHeight(mmb[0], mmb[1]));
100           max = Math.log(histogram.binHeight(mmb[2], mmb[3])) / Math.log(10);
101         } else {
102           min = histogram.binHeight(mmb[0], mmb[1]);
103           max = histogram.binHeight(mmb[2], mmb[3]);
104         }
105         dm = max - min;
106         // String pre = (log ? "10^" : "");
107 
108         NumberFormat nf = NumberFormat.getInstance();
109         nf.setMaximumFractionDigits(log ? 2 : 0);
110         Line2D.Double line = new Line2D.Double();
111         FontMetrics fm = g.getFontMetrics(g.getFont());
112         for (int i = 0; i <= 6; i++) {
113           int x = (int) Math.round((double) oX + (double) i * 66.66666667);
114           line.setLine(x, oY - 1, x, oY - 5);
115           g.draw(line);
116 
117           double c = min + ((double) i / 6d) * dm;
118           double d = (log ? Math.pow(10, c) : c);
119           // String s = pre + nf.format(min + ((double)i / 6d) * dm);
120           String s = nf.format(Math.round(d));
121           // if (s.equals("10^0"))
122           // s = "0";
123           g.drawString(s, x - fm.stringWidth(s) / 2, oY - 7);
124         }
125         g.drawString("Earthquake Density", oX + 148, oY + 27);
126       }
127     };
128   }

130   /** Renderer the 2-D histogram.  This uses the Jet color spectrum.
131    * @param g the graphics object upon which to render
132    */
133   public void render(Graphics2D g) {
134     int xBins = histogram.xAxis().bins();
135     int yBins = histogram.yAxis().bins();
136 
137     Color origColor = g.getColor();
138     Paint origPaint = g.getPaint();
139     AffineTransform origAT = g.getTransform();
140     Shape origClip = g.getClip();
141     Object aa = g.getRenderingHint(RenderingHints.KEY_ANTIALIASING);
142 
143     if (axis != null)
144       axis.render(g);
145 
146     g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
147     g.setClip(new Rectangle(graphX + 1, graphY + 1, graphWidth - 1, graphHeight - 1));
148     Rectangle2D.Double rect = new Rectangle2D.Double();
149     double xOffset = 0;
150     double yOffset = 0;
151     int[] mmb = histogram.minMaxBins();
152     double min = 0;
153     double max = 0;
154     double dm = 0;
155     if (log) {
156       min = (histogram.binHeight(mmb[0], mmb[1]));
157       max = Math.log(histogram.binHeight(mmb[2], mmb[3])) / Math.log(10);
158     } else {
159       min = histogram.binHeight(mmb[0], mmb[1]);
160       max = histogram.binHeight(mmb[2], mmb[3]);
161     }
162     dm = max - min;
163 
164     for (int j = 0; j < yBins; j++) {
165       double binHeight = histogram.yAxis().binWidth(j);
166       xOffset = 0;
167       if (!inverted)
168         yOffset += binHeight;
169       for (int i = 0; i < xBins; i++) {
170         double binWidth = histogram.xAxis().binWidth(i);
171         double xVal = getXPixel(histogram.xAxis().lowerEdge() + xOffset);
172         double yVal;
173 
174         if (inverted)
175           yVal = getYPixel(histogram.yAxis().upperEdge() - yOffset);
176         else
177           yVal = getYPixel(histogram.yAxis().lowerEdge() + yOffset);
178 
179         rect.setRect(xVal, yVal, binWidth * getXScale(), binHeight * getYScale());
180 
181         xOffset += binWidth;
182         // System.out.println(rect);
183         double count = 0;
184         if (log)
185           count = Math.log(histogram.binHeight(i, j)) / Math.log(10);
186         else
187           count = histogram.binHeight(i, j);
188         count -= min;
189         int color = (int) Math.round(count / dm * (spectrum.colors.length - 1));
190         Color clr = spectrum.colors[color];
191 
192         // If the color is 0, make it completely transparent so that the entire image isn't
193         // coated in blue.
194         if (color == 0)
195           clr = new Color(clr.getRed(), clr.getGreen(), clr.getBlue(), 0);
196         else
197           clr = new Color(clr.getRed(), clr.getGreen(), clr.getBlue(), 160);
198         g.setPaint(clr);
199         g.fill(rect);
200 
201         // offset += binWidth;
202         // getXPixel(minX + binWidth), graphHeight - getYPixel(histogram.binHeight(i)));
203         /*
204          * if (fillColor != null)
205          * {
206          * g.setPaint(fillColor);
207          * g.fill(rect);
208          * }
209          * if (strokeColor != null)
210          * {
211          * g.setPaint(Color.black);
212          * g.draw(rect);
213          * }
214          */
215       }
216       if (inverted)
217         yOffset += binHeight;
218     }
219 
220     // g.translate(-graphX, -graphY);
221     g.setClip(origClip);
222 
223     for (Renderer renderer : renderers)
224       renderer.render(g);
225     // for (int i = 0; i < renderers.size(); i++)
226     // ((Renderer)renderers.elementAt(i)).render(g);
227 
228     g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, aa);
229     g.setTransform(origAT);
230     g.setPaint(origPaint);
231     g.setColor(origColor);
232   }
