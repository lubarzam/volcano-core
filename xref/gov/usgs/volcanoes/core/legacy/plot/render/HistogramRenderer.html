16  /**
17   * <p>HistogramRenderer is a FrameRenderer that renders a traditional 1-D histogram.
18   * The histogram code is from the Colt library.</p>
19   *
20   * $Log: not supported by cvs2svn $
21   * Revision 1.2  2005/08/28 19:14:10  dcervelli
22   * Allows null histograms.
23   *
24   * Revision 1.1  2005/08/26 18:05:04  uid879
25   * Initial avosouth commit.
26   *
27   * @author Dan Cervelli
28   */
29  public class HistogramRenderer extends FrameRenderer {
30  
31    private Histogram1D histogram;
32  
33    /** The fill color of the histogram boxes.
34     */
35    public Color fillColor;
36  
37    /** The stroke <i>color</i> of the histrogram boxes.
38     */
39    public Color strokeColor;
40  
41    private List<Renderer> renderers;
42  
43    /** Constructs a new HistogramRenderer for renderering the specified 
44     * histrogram.
45     * @param h the histogram
46     */
47    public HistogramRenderer(Histogram1D h) {
48      histogram = h;
49      fillColor = new Color(0.0f, 0.0f, 0.56f);
50      strokeColor = Color.BLACK;
51      renderers = new ArrayList<Renderer>();
52      if (histogram.entries() == 0)
53        NO_DATA = true;
54    }
55  
56    /** Add a generic renderer.
57     * @param r the Renderer.
58     */
59    public void addRenderer(Renderer r) {
60      renderers.add(r);
61    }
62  
63    /** Sets the default extents of the parent FrameRenderer.  The vertical
64     * extent is such that there is a small bit of space above the tallest
65     * bin.
66     */
67    public void setDefaultExtents() {
68      if (histogram == null) {
69        minX = 0;
70        maxX = 1;
71        minY = 0;
72        maxY = 1;
73      } else {
74        minX = histogram.xAxis().lowerEdge();
75        maxX = histogram.xAxis().upperEdge();
76        minY = 0;
77        maxY = histogram.binHeight(histogram.minMaxBins()[1]) * 1.1;
78        if (maxY == 0)
79          maxY = 1;
80      }
81    }
82  
83    /** Creates a standard legend, a small line and point sample followed by
84     * the specified names.
85     * @param s the legend names
86     */
87    public void createDefaultLegendRenderer(String[] s) {
88  
89      setLegendRenderer(new LegendRenderer(NO_DATA));
90      getLegendRenderer().x = graphX + 6;
91      getLegendRenderer().y = graphY + 6;
92      getLegendRenderer().sided_line = true;
93      DataPointRenderer dpr = new DataPointRenderer('s', 8);
94      dpr.color = strokeColor;
95      dpr.filled = true;
96      dpr.fillColor = fillColor;
97      ShapeRenderer sr = null;
98      if (renderers.size() > 0) {
99        Renderer r = renderers.get(0);
100       if (r instanceof MatrixRenderer) {
101         MatrixRenderer mr = (MatrixRenderer) r;
102         if ((mr.pointRenderers != null) && (mr.pointRenderers.length > 0)) {
103           sr = new ShapeRenderer(new Line2D.Double(graphX, graphY, graphX + 6, graphY));
104           sr.color = mr.pointRenderers[0].color;
105         } else if ((mr.lineRenderers != null) && (mr.lineRenderers.length > 0)) {
106           sr = new ShapeRenderer(new Line2D.Double(graphX, graphY, graphX + 6, graphY));
107           sr.color = mr.lineRenderers[0].color;
108         }
109       }
110     }
111     getLegendRenderer().addLine(sr, dpr, s[0]);
112   }
113 
114   /** Renderers the histogram.
115    * @param g the graphics object upon which to render
116    */
117   public void render(Graphics2D g) {
118     Color origColor = g.getColor();
119     Paint origPaint = g.getPaint();
120     AffineTransform origAT = g.getTransform();
121     Shape origClip = g.getClip();
122 
123     if (axis != null)
124       axis.render(g);
125 
126     if (histogram != null) {
127       int bins = histogram.xAxis().bins();
128 
129       g.setClip(new Rectangle(graphX + 1, graphY + 1, graphWidth - 1, graphHeight - 1));
130       Rectangle2D.Double rect = new Rectangle2D.Double();
131       double offset = 0;
132       for (int i = 0; i < bins; i++) {
133         double binWidth = histogram.xAxis().binWidth(i);
134         rect.setRect(getXPixel(histogram.xAxis().lowerEdge() + offset),
135             getYPixel(histogram.binHeight(i)), binWidth * getXScale(),
136             histogram.binHeight(i) * getYScale());
137         offset += binWidth;
138 
139         if (fillColor != null) {
140           g.setPaint(fillColor);
141           g.fill(rect);
142         }
143         if (strokeColor != null) {
144           g.setPaint(Color.black);
145           g.draw(rect);
146         }
147       }
148       g.setClip(origClip);
149     }
150 
151     for (Renderer renderer : renderers)
152       renderer.render(g);
153 
154     if (getLegendRenderer() != null)
155       getLegendRenderer().render(g);
156 
157     g.setTransform(origAT);
158     g.setPaint(origPaint);
159     g.setColor(origColor);
160   }
