6   /**
7    * <p>Frame renderer with inverted Y axis</p>
8    * 
9    * @author Tom Parker
10   */
11  public class InvertedFrameRenderer extends BasicFrameRenderer {
12  
13    /** Constructor that accepts a LineData object.
14     */
15    public InvertedFrameRenderer() {
16      super();
17    }
18  
19    /** Gets a transformed y-pixel (Transformer implemenatation).
20     * @param y the y-coordinate in user space
21     * @return the y-pixel
22     */
23    public double getYPixel(double y) {
24      // return graphHeight - ((y - minY) * getYScale()) + (double)graphY;
25      return ((y - minY) * getYScale()) + (double) graphY;
26  
27    }
28  
29    /** Sets the axis to an empty Renderer
30    */
31    public void createEmptyAxis() {
32      axis = new InvertedAxisRenderer(this);
33    }
34  
35    /** Creates the default axis using SmartTick.
36     * @param hTicks suggested number of horizontal ticks
37     * @param vTicks suggested number of vertical ticks
38     * @param isHTicks "
39     * @param isVTicks
40     * @param hExpand should horizontal be expanded for better ticks
41     * @param vExpand should vertical be expanded for better ticks
42     * @param isHValues flag if we draw values near horizontal ticks
43     * @param isVValues flag if we draw values near vertical ticks
44     */
45    public void createDefaultAxis(int hTicks, int vTicks, boolean isHTicks, boolean isVTicks,
46        boolean hExpand, boolean vExpand, boolean isHValues, boolean isVValues) {
47      if (axis == null)
48        axis = new InvertedAxisRenderer(this);
49      axis.createDefault();
50      createDefaultYAxis(vTicks, isVTicks, isVValues, vExpand);
51      if (hTicks > 0) {
52        double[] t = SmartTick.autoTick(getMinXAxis(), getMaxXAxis(), hTicks, hExpand);
53        if (hExpand) {
54          minX = t[0];
55          maxX = t[t.length - 1];
56        }
57        if (isHTicks) {
58          axis.createTopTicks(t);
59          axis.createBottomTicks(t);
60          axis.createVerticalGridLines(t);
61        }
62        if (isHValues) {
63          axis.createBottomTickLabels(t, null);
64        }
65      }
66    }
67  
68    /** Gets the default translation for plot height
69     * @param plotHeight height of plot
70     * @return default translation for plot height
71     */
72    public double[] getDefaultTranslation(int plotHeight) {
73      double dx = (maxX - minX) / (graphWidth);
74      double dy = (maxY - minY) / (graphHeight);
75  
76      /*
77       * double d[] = new double[] {
78       * dx, minX - dx * gMinX,
79       * dy, minY - dy * (plot.getHeight() - gMaxY),
80       * minX, maxX, minY, maxY};
81       */
82  
83      double[] translation = new double[] {dx, minX - dx * graphX, dy * -1,
84          maxY + dy * (plotHeight - (graphHeight) - graphY), // <- how far the graph is from the
85                                                             // bottom of the plot
86          minX, maxX, minY, maxY};
87  
88      return translation;
89    }
90  }
