1   package gov.usgs.volcanoes.core.legacy.plot.render;
2   
3   import java.awt.BasicStroke;
4   import java.awt.Color;
5   import java.awt.Graphics2D;
6   import java.awt.Stroke;
7   import java.awt.geom.Line2D;
8   
9   /**
10   * A renderer that renders a line
11   * TODO: extend ShapeRenderer
12   * 
13   * @author Dan Cervelli
14   */
15  public class LineRenderer implements Renderer {
16    public static final Stroke DASHED_STROKE = new BasicStroke(1.0f, BasicStroke.CAP_BUTT,
17        BasicStroke.JOIN_MITER, 10.0f, new float[] {1.0f, 2.0f}, 0.0f);
18  
19    // Some sort of Java bug causes non-dashed strokes to occasionally to
20    // be offset by one pixel from a dashed stroke. Use this stroke
21    // to line up perfectly with other dashed strokes.
22    public static final Stroke TICK_STROKE = new BasicStroke(1.0f, BasicStroke.CAP_BUTT,
23        BasicStroke.JOIN_MITER, 10.0f, new float[] {1.0f, 0.0f}, 0.0f);
24  
25    /** The Line.
26     */
27    public Line2D.Double line;
28  
29    /** The Stroke.
30     */
31    public Stroke stroke;// = NORMAL_STROKE;
32  
33    /** The Color.
34     */
35    public Color color;
36  
37    /** Generic empty constructor.
38     */
39    public LineRenderer() {}
40  
41    /** Renders the line object using the internally stored stroke and color.  
42     * @param g the graphics object upon which to render
43     */
44    public void render(Graphics2D g) {
45  
46      Stroke origStroke = g.getStroke();
47      Color origColor = g.getColor();
48  
49      if (stroke != null)
50        g.setStroke(stroke);
51  
52      if (color != null)
53        g.setColor(color);
54      if (line != null && !(Double.isNaN(line.x1) || Double.isNaN(line.x2) || Double.isNaN(line.y1)
55          || Double.isNaN(line.y2)))
56        g.draw(line);
57  
58      g.setStroke(origStroke);
59      g.setColor(origColor);
60    }
61  }
