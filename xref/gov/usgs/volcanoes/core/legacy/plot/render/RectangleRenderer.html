1   package gov.usgs.volcanoes.core.legacy.plot.render;

3   import java.awt.Color;
4   import java.awt.Graphics2D;
5   import java.awt.geom.Rectangle2D;
6   
7   /**
8    * <p>A simple Renderer that renders a Rectangle2D object.  This class exists
9    * in case functionality to snazz up the rectangle is wanted (rounded corners,
10   * double frames, etc).</p>
11   *
12   * TODO: extend ShapeRenderer
13   * 
14   * $Log: not supported by cvs2svn $
15   * Revision 1.1  2005/08/26 18:05:04  uid879
16   * Initial avosouth commit.
17   *
18   * @author  Dan Cervelli
19   */
20  public class RectangleRenderer implements Renderer {
21    /** The Rectangle.
22     */
23    public Rectangle2D.Double rect;
24  
25    public Color backgroundColor = null;
26    public Color color = null;
27  
28    /** Generic constructor.
29     */
30    public RectangleRenderer() {
31      rect = new Rectangle2D.Double();
32    }
33  
34    /** Renders the rectangle.
35     * @param g the Graphics2D object to render to
36     */
37    public void render(Graphics2D g) {
38      Color c = g.getColor();
39      if (backgroundColor != null) {
40        g.setColor(backgroundColor);
41        g.fill(rect);
42        if (color == null)
43          g.setColor(c);
44      }
45      if (color != null)
46        g.setColor(color);
47      g.draw(rect);
48      g.setColor(c);
49    }
50  
51  }
