20  public class RenderedImageDataRenderer extends FrameRenderer {
21    private double dataMinX, dataMaxX, dataMinY, dataMaxY;
22    private RenderedImage image;
23  
24    /**
25     * Empty constructor.
26     */
27    public RenderedImageDataRenderer() {}
28  
29    /** Constructor that specifies the image. 
30     * @param img the image
31     */
32    public RenderedImageDataRenderer(RenderedImage img) {
33      image = img;
34    }
35  
36    /**
37     * Sets the image.
38     * @param img the image
39     */
40    public void setImage(RenderedImage img) {
41      image = img;
42    }
43  
44    /** Sets the data extents of this image.
45     * @param minX the minimum x
46     * @param maxX the maximum x
47     * @param minY the minimum y
48     * @param maxY the maximum y
49     */
50    public void setDataExtents(double minX, double maxX, double minY, double maxY) {
51      dataMinX = minX;
52      dataMaxX = maxX;
53      dataMinY = minY;
54      dataMaxY = maxY;
55    }
56  
57    /** Renderers the portion of the image that is visible (based on the data
58     * extents).
59     * @param g the graphics object upon which to render
60     */
61    public void render(Graphics2D g) {
62      AffineTransform origAT = g.getTransform();
63      Shape origClip = g.getClip();
64  
65      if (axis != null)
66        axis.render(g);
67  
68      if (image != null) {
69        g.clip(new Rectangle(graphX + 1, graphY + 1, graphWidth, graphHeight));
70        AffineTransform at = new AffineTransform();
71        int imageWidth = image.getWidth();
72        int imageHeight = image.getHeight();
73        double dataWidth = dataMaxX - dataMinX;
74        double ratioWidth = dataWidth / getWidth();
75        double dataHeight = dataMaxY - dataMinY;
76        double ratioHeight = dataHeight / getHeight();
77        at.scale(ratioWidth * (double) graphWidth / (double) imageWidth,
78            ratioHeight * (double) graphHeight / (double) imageHeight);
79        g.translate(getXPixel(dataMinX), getYPixel(dataMaxY));
80        g.drawRenderedImage(image, at);
81        g.translate(-getXPixel(dataMinX), -getYPixel(dataMaxY));
82      }
83  
84      g.setClip(origClip);
85      g.setTransform(origAT);
86  
87      if (getLegendRenderer() != null)
88        getLegendRenderer().render(g);
89    }
90  }
