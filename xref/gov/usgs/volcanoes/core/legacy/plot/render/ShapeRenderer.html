1   package gov.usgs.volcanoes.core.legacy.plot.render;
2   
3   import java.awt.BasicStroke;
4   import java.awt.Color;
5   import java.awt.Graphics2D;
6   import java.awt.RenderingHints;
7   import java.awt.Shape;
8   import java.awt.Stroke;
9   
10  /**
11   * A simple Renderer that renders a Shape object with a particular color and stroke.
12   *
13   * @author  Dan Cervelli
14   */
15  public class ShapeRenderer implements Renderer {
16    /** The Color.
17     */
18    public Color color;
19  
20    /** The Shape.
21     */
22    public Shape shape;
23  
24    /** The Stroke.
25     */
26    public Stroke stroke = new BasicStroke(1.0f);
27  
28    public boolean antiAlias;
29  
30    /** Generic constructor.
31     */
32    public ShapeRenderer(Shape s) {
33      shape = s;
34    }
35  
36    /** Renders the Shape.
37     * @param g the Graphics2D object to render to
38     */
39    public void render(Graphics2D g) {
40      Stroke origStroke = g.getStroke();
41      Color origColor = g.getColor();
42      Object origRenderingHint = g.getRenderingHint(RenderingHints.KEY_ANTIALIASING);
43  
44      if (antiAlias)
45        g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
46  
47      if (stroke != null)
48        g.setStroke(stroke);
49  
50      if (color != null)
51        g.setColor(color);
52  
53      if (shape != null)
54        g.draw(shape);
55  
56      g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, origRenderingHint);
57      g.setColor(origColor);
58      g.setStroke(origStroke);
59    }
60  
61  }
