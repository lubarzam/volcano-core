8   public class MinimumWaveRenderer extends SliceWaveRenderer {
9     /**
10     * Set axis ticks and labels values 
11     */
12    public void update() {
13      this.setExtents(viewStartTime, viewEndTime, minY, maxY);
14      int hTicks = graphWidth / 108;
15      int vTicks = graphHeight / 24;
16      this.createDefaultAxis(hTicks, vTicks);
17      this.setXAxisToTime(hTicks, false, false);
18      this.getAxis().setInnerLeftLabelAsText(yLabelText, -46);
19      if (title != null)
20        this.getAxis().setLeftLabelAsText(title, -56);
21    }
