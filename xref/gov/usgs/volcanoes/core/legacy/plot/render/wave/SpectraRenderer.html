86    protected class DefaultSpectraFrameDecorator extends DefaultFrameDecorator {
87      public DefaultSpectraFrameDecorator() {
88        if (yUnitText != null) {
89          this.yUnit = yUnitText;
90        }
91        if (yLabelText != null) {
92          this.yAxisLabel = yLabelText;
93        }
94        if (xUnits) {
95          this.xUnit = "Frequency (Hz)";
96        }
97        this.xAxisLabels = xTickValues;
98        this.yAxisLabels = yTickValues;
99        if (!xTickMarks) {
100         hTicks = 0;
101         xAxisGrid = Grid.NONE;
102       }
103       if (!yTickMarks) {
104         vTicks = 0;
105         yAxisGrid = Grid.NONE;
106       }
107       this.title = channelTitle;
108       this.titleBackground = Color.WHITE;
109     }

113       yAxis = (logPower) ? DefaultFrameDecorator.YAxis.LOG : DefaultFrameDecorator.YAxis.LINEAR;

118   /**
119    * Compute spectra for slice.
120    * Reinitialize frame decorator with this renderer data.
121    * @param oldMaxPower 
122    * @return maximum spectra power value
123    */
124 
125   public void update() {
126 
127     if (decorator == null)
128       decorator = new DefaultSpectraFrameDecorator();
129 
130     decorator.update();
131 
132     int nfft = (int) Util.getPreviousPowerOf2(wave.samples());
133     Spectra spectra = new Spectra(wave.getSignal(), wave.getSamplingRate(), nfft);
134 
135     DoubleMatrix2D dm = DoubleFactory2D.dense.make(spectra.getMatrix(logPower, logFreq));
136 
137     setData(dm);
138     setVisible(0, false);
139 
140     double minf = Math.max(minFreq, wave.getSamplingRate() / nfft);
141     double maxf = Math.min(maxFreq, wave.getNyquist());
142 
143     double X1 = logFreq ? Math.log10(minf) : minf;
144     double X2 = logFreq ? Math.log10(maxf) : maxf;
145     double Y1 =
146         logPower ? Math.log10(spectra.getMinPower(minf, maxf)) : spectra.getMinPower(minf, maxf);
147     double Y2 =
148         logPower ? Math.log10(spectra.getMaxPower(minf, maxf)) : spectra.getMaxPower(minf, maxf);
149 
150     setExtents(X1, X2, Y1, Y2);
151 
152     createDefaultLineRenderers(color);
153     decorator.decorate(this);
154 
155   }
