4    * <p>An interface for an object that can perform a translation from user unit-space

6    *
7    * @author  Dan Cervelli
8    */
9   public interface Transformer {

11     * @param x the x coordinate
12     * @return the transformed x coordinate
13     */

15  
16    /** Transforms an y coordinate in user-space to pixel-space.

18     * @return the transformed y coordinate
19     */
20    public double getYPixel(double y);
21  }
