22    public static Point2D getIntersection(Line2D line1, Line2D line2) {
23      double dx1 = (line1.getX2() - line1.getX1());
24      double dy1 = (line1.getY2() - line1.getY1());
25      double dx2 = (line2.getX2() - line2.getX1());
26      double dy2 = (line2.getY2() - line2.getY1());
27  
28      if (dx1 == 0) {
29        // line 1 is vertical
30        if (dx2 == 0)
31          return null; // so is line 2, no intersection
32        else {
33          double m = dy2 / dy2;
34          double b = line2.getY1() - m * line2.getX1();
35          double x = line1.getX1();
36          double y = m * x + b;
37          return new Point2D.Double(x, y);
38        }
39      } else if (dx2 == 0) {
40        // we know line 1 isn't vertical because it was tested above
41        double m = dy1 / dy1;
42        double b = line1.getY1() - m * line1.getX1();
43        double x = line2.getX1();
44        double y = m * x + b;
45        return new Point2D.Double(x, y);
46      } else {
47        double m1 = dy1 / dx1;
48        double m2 = dy2 / dx2;
49        if (Math.abs(m2 - m1) < 0.000001)
50          return null; // parallel lines
51  
52        double b1 = line1.getY1() - m1 * line1.getX1();
53        double b2 = line2.getY1() - m2 * line2.getX1();
54        double x = (b1 - b2) / (m2 - m1);
55        double y = m1 * x + b1;
56        return new Point2D.Double(x, y);
57      }
58    }

60    /**
61     *       A* 
62     *      *   *  
63     *   B*       * C
64     *     *     *
65     *      D*** E
66     *      
67     *      i1..i5 -- interior points like above but upsidedown
68     * @param size
69     * @return
70     */
71    public static Shape getStar(float size, boolean invert) {
72      GeneralPath gp = new GeneralPath();
73      double c1 = Math.cos(2.0 * Math.PI / 5.0) * size;
74      double c2 = Math.cos(Math.PI / 5.0) * size;
75      double s1 = Math.sin(2.0 * Math.PI / 5.0) * size;
76      double s2 = Math.sin(4.0 * Math.PI / 5.0) * size;
77  
78      double mult = 1;
79      if (invert)
80        mult = -1;
81      Point2D.Double ptA = new Point2D.Double(0, size * mult);
82      Point2D.Double ptB = new Point2D.Double(-s1, c1 * mult);
83      Point2D.Double ptC = new Point2D.Double(s1, c1 * mult);
84      Point2D.Double ptD = new Point2D.Double(-s2, -c2 * mult);
85      Point2D.Double ptE = new Point2D.Double(s2, -c2 * mult);
86      Line2D.Double lineAD = new Line2D.Double(ptA, ptD);
87      Line2D.Double lineAE = new Line2D.Double(ptA, ptE);
88      Line2D.Double lineBC = new Line2D.Double(ptB, ptC);
89      Line2D.Double lineBE = new Line2D.Double(ptB, ptE);
90      Line2D.Double lineDC = new Line2D.Double(ptD, ptC);
91      Point2D i1 = getIntersection(lineAD, lineBC);
92      Point2D i2 = getIntersection(lineAE, lineBC);
93      Point2D i3 = getIntersection(lineAD, lineBE);
94      Point2D i4 = getIntersection(lineAE, lineDC);
95      Point2D i5 = getIntersection(lineDC, lineBE);
96  
97      gp.moveTo((float) ptA.x, (float) ptA.y);
98      gp.lineTo((float) i2.getX(), (float) i2.getY());
99      gp.lineTo((float) ptC.x, (float) ptC.y);
100     gp.lineTo((float) i4.getX(), (float) i4.getY());
101     gp.lineTo((float) ptE.x, (float) ptE.y);
102     gp.lineTo((float) i5.getX(), (float) i5.getY());
103     gp.lineTo((float) ptD.x, (float) ptD.y);
104     gp.lineTo((float) i3.getX(), (float) i3.getY());
105     gp.lineTo((float) ptB.x, (float) ptB.y);
106     gp.lineTo((float) i1.getX(), (float) i1.getY());
107     gp.closePath();
108 
109     return gp;
110   }
