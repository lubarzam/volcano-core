4    * A class for the Goertzel algorithm.  
5    *
6    * @author Tom Parker
7    */
8   public class Goertzel {
9   
10    public static double goertzel(double freq, double sampleR, int[] signal) {
11      return goertzel(freq, sampleR, signal, true);
12    }
13  
14    public static double goertzel(double freq, double sampleR, int[] signal, boolean hamming) {
15      int len = signal.length;
16  
17      // hamming window -
18      if (hamming)
19        for (int i = 0; i < signal.length; i++)
20          signal[i] *= (0.54 - 0.46 * Math.cos(2 * Math.PI * i / sampleR));
21  
22      double s;
23      double s_prev = 0;
24      double s_prev2 = 0;
25      double coeff = 2 * Math.cos(2 * Math.PI * freq / sampleR);
26      for (int i = 0; i < len; i++) {
27        s = signal[i] + coeff * s_prev - s_prev2;
28        s_prev2 = s_prev;
29        s_prev = s;
30      }
31  
32      double power = s_prev2 * s_prev2 + s_prev * s_prev - coeff * s_prev2 * s_prev;
33      return power;
34    }
35  }
