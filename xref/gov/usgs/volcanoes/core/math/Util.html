43    /** Computes the value of the modified Bessel function of the first 
44     * kind of order 0.
45     * @param x the function argument
46     * @return the value of the function
47     * 
48     * From: A Numerical Library for Scientists and Engineers, Hang T. Lau
49     * (2004), Chapman & Hall ISBN 1-58488-430-4
50     */
51  
52    public static double bessi0(double x) {
53      if (x == 0.0)
54        return 1.0;
55      if (Math.abs(x) <= 15.0) {
56        double z, denominator, numerator;
57        z = x * x;
58        numerator = (z
59            * (z * (z * (z
60                * (z * (z * (z
61                    * (z * (z * (z
62                        * (z * (z * (z * (z * 0.210580722890567e-22 + 0.380715242345326e-19)
63                            + 0.479440257548300e-16) + 0.435125971262668e-13) + 0.300931127112960e-10)
64                        + 0.160224679395361e-7) + 0.654858370096785e-5) + 0.202591084143397e-2)
65                    + 0.463076284721000e0) + 0.754337328948189e2) + 0.830792541809429e4)
66                + 0.571661130563785e6) + 0.216415572361227e8) + 0.356644482244025e9)
67            + 0.144048298227235e10);
68        denominator =
69            (z * (z * (z - 0.307646912682801e4) + 0.347626332405882e7) - 0.144048298227235e10);
70        return -numerator / denominator;
71      } else {
72        return Math.exp(Math.abs(x)) * nonexpbessi0(x);
73      }
74    }

76    /** Computes the value of the modified Bessel function of the first 
77     * kind of order 0 multipled by e^(-x).
78     * @param x the function argument
79     * @return the value of the function
80     * 
81     * From: A Numerical Library for Scientists and Engineers, Hang T. Lau
82     * (2004), Chapman & Hall ISBN 1-58488-430-4
83     */
84  
85    public static double nonexpbessi0(double x) {
86      if (x == 0.0)
87        return 1.0;
88      if (Math.abs(x) <= 15.0) {
89        return Math.exp(-Math.abs(x)) * bessi0(x);
90      } else {
91        int i;
92        double sqrtx, br, br1, br2, z, z2, numerator, denominator;
93        double ar1[] =
94            {0.2439260769778, -0.115591978104435e3, 0.784034249005088e4, -0.143464631313583e6};
95        double ar2[] = {1.0, -0.325197333369824e3, 0.203128436100794e5, -0.361847779219653e6};
96        x = Math.abs(x);
97        sqrtx = Math.sqrt(x);
98        br1 = br2 = 0.0;
99        z = 30.0 / x - 1.0;
100       z2 = z + z;
101       for (i = 0; i <= 3; i++) {
102         br = z2 * br1 - br2 + ar1[i];
103         br2 = br1;
104         br1 = br;
105       }
106       numerator = z * br1 - br2 + 0.346519833357379e6;
107       br1 = br2 = 0.0;
108       for (i = 0; i <= 3; i++) {
109         br = z2 * br1 - br2 + ar2[i];
110         br2 = br1;
111         br1 = br;
112       }
113       denominator = z * br1 - br2 + 0.865665274832055e6;
114       return (numerator / denominator) / sqrtx;
115     }
116   }

118   /** Computes an N-point Kaiser window
119    * @param windowLength length of the Kaiser window
120    * @param beta the parameter for the Bessel function
121    * @return the window 
122    * 
123    * pcervelli, 2011/07/10
124    */
125 
126   public static double[] kaiser(int windowLength, double beta) {
127 
128     double[] window;
129     double B0;
130     double xind;
131 
132     int i;
133     int odd;
134     int n;
135 
136     window = new double[windowLength];
137     B0 = Math.abs(Util.bessi0(beta));
138     xind = (windowLength - 1) * (windowLength - 1);
139     odd = windowLength % 2;
140     n = (windowLength + odd) / 2;
141 
142     window[n - 1] = 1;
143     for (i = 0; i < n - odd; i++) {
144       window[i + n] =
145           Math.abs(Util.bessi0(beta * Math.sqrt(1 - Math.pow(2 * i + 1 + odd, 2) / xind)) / B0);
146       window[n - i - 1 - odd] = window[i + n];
147     }
148 
149     return window;
150   }
