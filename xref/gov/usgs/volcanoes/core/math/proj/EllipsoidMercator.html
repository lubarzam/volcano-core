12  public class EllipsoidMercator extends Projection {
13    private Point2D.Double origin;
14  
15    public EllipsoidMercator() {
16      name = "EllipsoidMercator";
17      origin = new Point2D.Double(0, 0);
18      ellipsoid = Ellipsoid.ELLIPSOIDS[11];
19    }
20  
21    @Override
22    public Double forward(Double lonLat) {
23      double r = ellipsoid.equatorialRadius;
24      double lambdaO = origin.getX() * DEG2RAD;
25  
26      double phi = lonLat.getY() * DEG2RAD;
27      double lambda = lonLat.getX() * DEG2RAD;
28  
29      double l = (lambda - lambdaO);
30      while (l > Math.PI)
31        l -= Math.PI * 2;
32      while (l < -Math.PI)
33        l += Math.PI * 2;
34      double x = r * l;
35      // double y = r * Math.log(Math.tan(Math.PI / 4 + phi / 2));
36      double sphi = Math.sin(phi);
37      double e = Math.sqrt(ellipsoid.eccentricitySquared);
38      double t2 = Math.pow((1 - e * sphi) / (1 + e * sphi), e);
39      double y = r * 0.5 * Math.log((1 + sphi) / (1 - sphi) * t2);
40  
41      return new Point2D.Double(x, y);
42    }
43  
44    @Override
45    public double getScale(Double lonLat) {
46      double e2 = ellipsoid.eccentricitySquared;
47      double sp = Math.sin(Math.toRadians(lonLat.y));
48      double top = Math.sqrt(1 - e2 * sp * sp);
49      double bottom = Math.cos(Math.toRadians(lonLat.y));
50      return top / bottom;
51    }
52  
53    @Override
54    public Double inverse(Double xy) {
55      double r = ellipsoid.equatorialRadius;
56      double e = Math.sqrt(ellipsoid.eccentricitySquared);
57      double lambdaO = origin.getX() * DEG2RAD;
58      double lambda = lambdaO + xy.x / r;
59  
60      double t = Math.exp(-xy.y / r);
61      double phi = Math.PI / 2 - 2 * Math.atan(t);
62      for (int i = 0; i < 5; i++) {
63        double sphi = Math.sin(phi);
64        double t2 = (1 - e * sphi) / (1 + e * sphi);
65        double t3 = t * Math.pow(t2, e / 2);
66        phi = Math.PI / 2 - 2 * Math.atan(t3);
67      }
68  
69      return new Point2D.Double(lambda * RAD2DEG, phi * RAD2DEG);
70    }
71  
72    public static void main(String[] args) throws Exception {
73      EllipsoidMercator merc = new EllipsoidMercator();
74      System.out.println(merc.forward(new Point2D.Double(-121.5, 35.5)));
75      System.out.println(merc.inverse(new Point2D.Double(-13525318, 4207225)));
76    }
77  }
