77    public double[] getProjectedExtents(GeoRange gr) {
78      Point2D.Double[] pt = new Point2D.Double[8];
79      pt[0] = new Point2D.Double(gr.getWest(), gr.getNorth());
80      pt[1] = new Point2D.Double(gr.getWest(), gr.getSouth());
81      pt[2] = new Point2D.Double(gr.getEast(), gr.getNorth());
82      pt[3] = new Point2D.Double(gr.getEast(), gr.getSouth());
83      pt[4] =
84          new Point2D.Double(GeoRange.normalize(gr.getWest() + gr.getLonRange() / 2), gr.getSouth());
85      pt[5] =
86          new Point2D.Double(GeoRange.normalize(gr.getWest() + gr.getLonRange() / 2), gr.getNorth());
87      pt[6] = new Point2D.Double(gr.getWest(), gr.getSouth() + (gr.getNorth() - gr.getSouth()) / 2);
88      pt[7] = new Point2D.Double(gr.getEast(), gr.getSouth() + (gr.getNorth() - gr.getSouth()) / 2);
89  
90      double minX = Double.MAX_VALUE;
91      double minY = Double.MAX_VALUE;
92      double maxX = -Double.MAX_VALUE;
93      double maxY = -Double.MAX_VALUE;
94  
95      for (int i = 0; i < 8; i++) {
96        // System.out.print(pt[i]);
97        pt[i] = forward(pt[i]);
98        // System.out.println(" " + pt[i]);
99        minX = Math.min(minX, pt[i].x);
100       minY = Math.min(minY, pt[i].y);
101       maxX = Math.max(maxX, pt[i].x);
102       maxY = Math.max(maxY, pt[i].y);
103     }
104 
105     return new double[] {minX, maxX, minY, maxY};
106   }

108   public GeoRange getGeoRange(Point2D.Double center, double xm, double ym) {
109     Point2D.Double pt = forward(center);
110     double left = pt.x - xm / 2;
111     double right = pt.x + xm / 2;
112     double top = pt.y + ym / 2;
113     double bottom = pt.y - ym / 2;
114     return getGeoRange(left, right, bottom, top);

117   public GeoRange getGeoRange(double left, double right, double bottom, double top) {
118     Point2D.Double[] pt = new Point2D.Double[8];
119     pt[0] = new Point2D.Double(left, top);
120     pt[1] = new Point2D.Double(left, bottom);
121     pt[2] = new Point2D.Double(right, top);
122     pt[3] = new Point2D.Double(right, bottom);
123     double dx = right - left;
124     double dy = top - bottom;
125     // System.out.println("dx: " + dx + " dy: " + dy);
126     pt[4] = new Point2D.Double(left + dx / 2, top);
127     pt[5] = new Point2D.Double(left + dx / 2, bottom);
128     pt[6] = new Point2D.Double(left, bottom + dy / 2);
129     pt[7] = new Point2D.Double(right, bottom + dy / 2);
130 
131     double west = Double.MAX_VALUE;
132     double south = Double.MAX_VALUE;
133     double east = -Double.MAX_VALUE;
134     double north = -Double.MAX_VALUE;
135 
136     for (int i = 0; i < 8; i++) {
137       // System.out.print(pt[i]);
138       pt[i] = inverse(pt[i]);
139       // System.out.println(" P-> " + pt[i]);
140       west = Math.min(west, pt[i].x);
141       south = Math.min(south, pt[i].y);
142       east = Math.max(east, pt[i].x);
143       north = Math.max(north, pt[i].y);
144     }
145     GeoRange gr = new GeoRange(west, east, south, north);
146     return gr;
147   }

150   public static double distanceBetween(Point2D.Double pt1, Point2D.Double pt2) {
151     if (pt1 == null || pt2 == null || Double.isNaN(pt1.x) || Double.isNaN(pt1.y)
152         || Double.isNaN(pt2.x) || Double.isNaN(pt2.y))
153       return Double.NaN;
154 
155     double phi1 = Math.toRadians(pt1.y);
156     double phi2 = Math.toRadians(pt2.y);
157     double lam1 = Math.toRadians(pt1.x);
158     double lam2 = Math.toRadians(pt2.x);
159     double dlam = lam2 - lam1;
160 
161     double a = Math.cos(phi2) * Math.sin(dlam);
162     double b = Math.cos(phi1) * Math.sin(phi2);
163     double c = Math.sin(phi1) * Math.cos(phi2) * Math.cos(dlam);
164     double d = Math.sin(phi1) * Math.sin(phi2);
165     double e = Math.cos(phi1) * Math.cos(phi2) * Math.cos(dlam);
166     double f = Math.atan2(Math.sqrt(a * a + (b - c) * (b - c)), (d + e));
167 
168     double r = Ellipsoid.ELLIPSOIDS[0].equatorialRadius;
169 
170     return f * r;
171   }

173   public static double distanceBetweenRad(Point2D.Double pt1, Point2D.Double pt2) {
174     if (pt1 == null || pt2 == null || Double.isNaN(pt1.x) || Double.isNaN(pt1.y)
175         || Double.isNaN(pt2.x) || Double.isNaN(pt2.y))
176       return Double.NaN;
177 
178     double phi1 = Math.toRadians(pt1.y);
179     double phi2 = Math.toRadians(pt2.y);
180     double lam1 = Math.toRadians(pt1.x);
181     double lam2 = Math.toRadians(pt2.x);
182     double dlam = lam2 - lam1;
183 
184     double a = Math.cos(phi2) * Math.sin(dlam);
185     double b = Math.cos(phi1) * Math.sin(phi2);
186     double c = Math.sin(phi1) * Math.cos(phi2) * Math.cos(dlam);
187     double d = Math.sin(phi1) * Math.sin(phi2);
188     double e = Math.cos(phi1) * Math.cos(phi2) * Math.cos(dlam);
189     double f = Math.atan2(Math.sqrt(a * a + (b - c) * (b - c)), (d + e));
190

203   public static double azimuthTo(Point2D.Double pt1, Point2D.Double pt2) {
204     if (pt1 == null || pt2 == null || Double.isNaN(pt1.x) || Double.isNaN(pt1.y)
205         || Double.isNaN(pt2.x) || Double.isNaN(pt2.y))
206       return Double.NaN;
207 
208     double phi1 = Math.toRadians(pt1.y);
209     double phi2 = Math.toRadians(pt2.y);
210     double lam1 = Math.toRadians(pt1.x);
211     double lam2 = Math.toRadians(pt2.x);
212     double dlam = lam2 - lam1;
213 
214     double a = Math.sin(dlam);
215     double b = Math.sin(phi1) * Math.cos(dlam);
216     double c = Math.cos(phi1) * Math.tan(phi2);
217     double az = Math.atan2(b - c, a);
218     az = Math.toDegrees(az) + 90;
219     if (az < 0)
220       az += 360;
221     if (az >= 360)
222       az -= 360;
223     return az;
224   }

226   public static Point2D.Double getPointFrom(Point2D.Double origin, double c, double az) {
227     double phi1 = origin.y * DEG2RAD;
228     double lam1 = origin.x * DEG2RAD;
229     // c = c / 6372795.477598;
230     c = c / Ellipsoid.ELLIPSOIDS[0].equatorialRadius;
231     az = az * DEG2RAD;
232     Point2D.Double pt = new Point2D.Double();
233     pt.y = Math.asin(Math.sin(phi1) * Math.cos(c) + Math.cos(phi1) * Math.sin(c) * Math.cos(az))
234         * RAD2DEG;
235     pt.x = (lam1 + Math.atan2(Math.sin(c) * Math.sin(az),
236         (Math.cos(phi1) * Math.cos(c) - Math.sin(phi1) * Math.sin(c) * Math.cos(az)))) * RAD2DEG;
237     return pt;
238   }

240   public static Point2D.Double[] getPointsFrom(Point2D.Double origin, double c, int n) {
241     double phi1 = origin.y * DEG2RAD;
242     double lam1 = origin.x * DEG2RAD;
243     // c = c / 6372795.477598;
244     c = c / Ellipsoid.ELLIPSOIDS[0].equatorialRadius;
245     Point2D.Double[] pts = new Point2D.Double[n];
246     for (int i = 0; i < n; i++) {
247       double az = ((double) i / (double) n * 360.0) * DEG2RAD;
248       Point2D.Double pt = new Point2D.Double();
249       pt.y = Math.asin(Math.sin(phi1) * Math.cos(c) + Math.cos(phi1) * Math.sin(c) * Math.cos(az))
250           * RAD2DEG;
251       pt.x = (lam1 + Math.atan2(Math.sin(c) * Math.sin(az),
252           (Math.cos(phi1) * Math.cos(c) - Math.sin(phi1) * Math.sin(c) * Math.cos(az)))) * RAD2DEG;
253       pts[i] = pt;
254     }
255     return pts;
256   }

258   public void mapRect(FastProjector fast, double x1, double y1, double x2, double y2, int dx,
259       int dy, int dw, int dh, int[] src, int[] dest, int sw, int sh, int scan, GeoRange range) {
260     double w = range.getWest();
261     double n = range.getNorth();
262     double lonRange = range.getLonRange();
263     double latRange = range.getLatRange();
264     Point2D.Double pt = new Point2D.Double();
265     for (int i = 0; i < dw; i++)
266       for (int j = 0; j < dh; j++) {
267         pt.x = x1 + (double) i / (double) dw * (x2 - x1);
268         pt.y = y1 + (double) j / (double) dh * (y2 - y1);
269         fast.inverse(pt);
270         if (Double.isNaN(pt.x))
271           continue;
272         pt.x = pt.x % 360;
273         if (pt.x > 180)
274           pt.x -= 360;
275         if (pt.x < -180)
276           pt.x += 360;
277         // int sx = (int)((e - pt.x) / lonRange * sw);
278         // int sy = (int)((n - pt.y) / latRange * sh);
279         int sx = (int) ((pt.x - w) / lonRange * sw);
280         int sy = (int) (-(pt.y - n) / latRange * sh);
281         if (sx >= sw || sy >= sh || sx < 0 || sy < 0)
282           continue;
283         // dest[dx + i + (dy + j) * 1000] = src.getRGB(sx, sy);
284         dest[dx + i + (dy + dh - j - 1) * scan] = src[sx + sy * sw];
285       }
286   }

288   public void fastMapRect(FastProjector fast, double x1, double y1, double x2, double y2, int dx,
289       int dy, int dw, int dh, int[] src, int[] dest, int sw, int sh, int scan, GeoRange range) {
290     Point2D.Double ul = new Point2D.Double(x1, y1);
291     Point2D.Double ur = new Point2D.Double(x2, y1);
292     Point2D.Double lr = new Point2D.Double(x2, y2);
293     Point2D.Double ll = new Point2D.Double(x1, y2);
294     fast.inverse(ul);
295     fast.inverse(ur);
296     fast.inverse(lr);
297     fast.inverse(ll);
298     if (Double.isNaN(ul.x) && Double.isNaN(ur.x) && Double.isNaN(lr.x) && Double.isNaN(ll.x))
299       return;
300     if (Double.isNaN(ul.x) || Double.isNaN(ur.x) || Double.isNaN(lr.x) || Double.isNaN(ll.x))
301       mapRect(fast, x1, y1, x2, y2, dx, dy, dw, dh, src, dest, sw, sh, scan, range);
302     else {
303       double w = range.getWest();
304       double n = range.getNorth();
305       double lonRange = range.getLonRange();
306       double latRange = range.getLatRange();
307 
308       // System.out.println("lonRange: " +lonRange);
309       // System.out.println("w: " + w);
310       // System.out.println("ul: " + ul);
311       // System.out.println("ur: " + ur);
312       // System.out.println("u lr: " + GeoRange.getLonRange(ul.x, ur.x));
313       // System.out.println("ll: " + ll);
314       // System.out.println("lr: " + lr);
315 
316       // for each col
317       for (int i = 0; i < dw; i++) {
318         if (Math.abs(ur.x - ul.x) > 180) {
319           if (ur.x > ul.x)
320             ur.x -= 360;
321           else
322             ul.x -= 360;
323           // System.out.println("u adj: " + (ur.x - ul.x));
324         }
325         if (Math.abs(lr.x - ll.x) > 180) {
326           if (lr.x > ll.x)
327             lr.x -= 360;
328           else
329             ll.x -= 360;
330           // System.out.println("l adj:\n\tll: " + ll + "\n\tlr: " +
331           // lr + " r: " + (lr.x - ll.x));
332         }
333         double wx0 = ul.x + i * (ur.x - ul.x) / (double) dw;
334         double wy0 = ul.y + i * (ur.y - ul.y) / (double) dw;
335         double wx1 = ll.x + i * (lr.x - ll.x) / (double) dw;
336         double wy1 = ll.y + i * (lr.y - ll.y) / (double) dw;
337         if (Math.signum(wx0) != Math.signum(wx1) && Math.abs(wx1 - wx0) > 180) {
338           if (wx1 < 0)
339             wx1 += 360;
340           else if (wx0 < 0)
341             wx0 += 360;
342         }
343         double x = wx0;
344         double y = wy0;
345         double sdx = (wx1 - wx0) / (double) dh;
346         double sdy = (wy1 - wy0) / (double) dh;
347         // int di = dx + i + (dy + dh - 1) * scan;
348         int di = dx + i + (dy + dh) * scan;
349         int sx;
350         int sy;
351         // for each row
352         for (int j = 0; j < dh; j++) {
353           double adjx = x;
354           if (adjx < w && range.containsLongitude(adjx)) {
355             adjx += 360;
356           }
357           sx = (int) ((adjx - w) / lonRange * sw);
358           sy = (int) (-(y - n) / latRange * sh);
359           x += sdx;
360           y += sdy;
361           if (sx >= sw || sy >= sh || sx < 0 || sy < 0) {
362             di -= scan;
363             continue;
364           }
365           if (di < dest.length)
366             dest[di] = src[sx + sy * sw];
367           di -= scan;
368         }
369       }
370     }
371   }

381   public BufferedImage getProjectedImage(int n, int dw, int dh, int[] src, int sw, int sh,
382       GeoRange gr, double mw, double me, double ms, double mn) {
383     CodeTimer ct = new CodeTimer("getProjectedImage");
384     FastProjector fast = getFastProjector();
385     int[] buf = new int[dw * dh];
386     Arrays.fill(buf, 0xffffffff);
387     int stepX = dw / (n * 2);
388     int stepY = dh / (n * 2);
389     for (int i = -n; i < n; i++)
390       for (int j = -n; j < n; j++) {
391         double x0 = mw + (me - mw) * (double) (i + n) / (double) (n * 2);
392         double y0 = ms + (mn - ms) * (double) (j + n) / (double) (n * 2);
393         double x1 = mw + (me - mw) * (double) (i + n + 1) / (double) (n * 2);
394         double y1 = ms + (mn - ms) * (double) (j + n + 1) / (double) (n * 2);
395         fastMapRect(fast, x0, y0, x1, y1, (n + i) * stepX, (n - j - 1) * stepY, stepX, stepY, src,
396             buf, sw, sh, dw, gr);
397       }
398 
399     DataBuffer dbi = new DataBufferInt(buf, dw * dh);
400     int[] bandMasks = {0xff0000, 0xff00, 0xff, 0xff000000};
401     WritableRaster wr = Raster.createPackedRaster(dbi, dw, dh, dw, bandMasks, null);
402     ColorModel cm = ColorModel.getRGBdefault();
403     BufferedImage bi = new BufferedImage(cm, wr, false, null);
404     ct.stopAndReport();
405     return bi;
406   }
