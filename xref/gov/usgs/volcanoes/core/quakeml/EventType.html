7   package gov.usgs.volcanoes.core.quakeml;
8   
9   /**
10   * Possible event types.
11   * 
12   * @author Tom Parker
13   *
14   */
15  public enum EventType {
16    NOT_EXISTING("not existing"),
17    NOT_REPORTED("not reported"),
18    EARTHQUAKE("earthquake"),
19    ANTHROPOGENIC_EVENT("anthropogenic event"),
20    COLLAPSE("collapse"),
21    CAVITY_COLLAPSE("cavity collapse"),
22    MINE_COLLAPSE("mine collapse"),
23    BUILDING_COLLAPSE("building collapse"),
24    EXPLOSION("explosion"),
25    ACCIDENTAL_EXPLOSION("accidental explosion"),
26    CHEMICAL_EXPLOSION("chemical explosion"),
27    CONTROLLED_EXPLOSION("controlled explosion"),
28    EXPERIMENTAL_EXPLOSION("experimental explosion"),
29    INDUSTRIAL_EXPLOSION("industrial explosion"),
30    MINING_EXPLOSION("mining explosion"),
31    QUARRY_BLAST("quarry blast"),
32    ROAD_CUT("road cut"),
33    BLASTING_LEVEE("blasting levee"),
34    NUCLEAR_EXPLOSION("nuclear explosion"),
35    INDUCED_OR_TRIGGERED_EVENT("induced or triggered event"),
36    ROCK_BURST("rock burst"),
37    RESERVOIR_LOADING("reservoir loading"),
38    FLUID_INJECTION("fluid injection"),
39    FLUID_EXTRACTION("fluid extraction"),
40    CRASH("crash"),
41    PLANE_CRASH("plane crash"),
42    TRAIN_CRASH("train crash"),
43    BOAT_CRASH("boat crash"),
44    OTHER_EVENT("other event"),
45    ATMOSPHERIC_EVENT("atmospheric event"),
46    SONIC_BOOM("sonic boom"),
47    SONIC_BLAST("sonic blast"),
48    ACOUSTIC_NOISE("acoustic noise"),
49    THUNDER("thunder"),
50    AVALANCHE("avalanche"),
51    SNOW_AVALANCHE("snow avalanche"),
52    DEBRIS_AVALANCHE("debris avalanche"),
53    HYDROACOUSTIC_EVENT("hydroacoustic event"),
54    ICE_QUAKE("ice quake"),
55    SLIDE("slide"),
56    LANDSLIDE("landslide"),
57    ROCKSLIDE("rockslide"),
58    METEORITE("meteorite"),
59    VOLCANIC_ERUPTION("volcanic eruption");
60        
61    private final String description;
62  
63    private EventType(String description) {
64      this.description = description;
65    }
66    
67    public String toString() {
68      return description;
69    }
70  
71    /**
72     * Parse a string.
73     * 
74     * @param inString String to parse
75     * @return matching EventType
76     */
77    public static EventType parse(String inString) {
78      for (EventType type : EventType.values()) {
79        if (type.description.equals(inString.toLowerCase())) {
80          return type;
81        }
82      }
83      throw new IllegalArgumentException("Invalid type " + inString);
84  
85    }
