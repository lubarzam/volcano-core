38    /**
39     * Constructor for newly created origin.
40     * @param publicId resource identifier of origin
41     * @param time focal time
42     * @param longitude hypocenter longitude (WGS84)
43     * @param latitude hypocenter latitude (WGS84)
44     */
45    public Origin(String publicId, long time, double longitude, double latitude) {
46      this.publicId = publicId;
47      this.time = new TimeQuantity(time);
48      this.longitude = new RealQuantity(longitude);
49      this.latitude = new RealQuantity(latitude);
50    }
51  
52    /**
53     * Constructor.
54     *
55     * @param originElement XML element
56     * @param picks associated picks
57     */
58    public Origin(Element originElement, Map<String, Pick> picks) {
59      publicId = originElement.getAttribute("publicID");
60      arrivals = new HashMap<String, Arrival>();
61  
62      LOGGER.debug("new origin {}", publicId);
63  
64      final Element lonElement = (Element) originElement.getElementsByTagName("longitude").item(0);
65      longitude = new RealQuantity(lonElement);
66  
67      final Element latElement = (Element) originElement.getElementsByTagName("latitude").item(0);
68      latitude = new RealQuantity(latElement);
69  
70      final Element depthElement = (Element) originElement.getElementsByTagName("depth").item(0);
71      depth = new RealQuantity(depthElement);
72  
73      final Element timeElement = (Element) originElement.getElementsByTagName("time").item(0);
74      time = new TimeQuantity(timeElement);
75  
76      Element evaluationElement =
77          (Element) originElement.getElementsByTagName("evaluationStatus").item(0);
78      if (evaluationElement != null) {
79        evaluationStatus = EvaluationStatus.valueOf(evaluationElement.getTextContent().toUpperCase());
80      }
81  
82      evaluationElement = (Element) originElement.getElementsByTagName("evaluationMode").item(0);
83      if (evaluationElement != null) {
84        evaluationMode = EvaluationMode.valueOf(evaluationElement.getTextContent().toUpperCase());
85      }
86  
87      final Element qualityElement = (Element) originElement.getElementsByTagName("quality").item(0);
88      if (qualityElement != null) {
89        quality = new OriginQuality(qualityElement);
90      }
91  
92      parseArrivals(originElement.getElementsByTagName("arrival"), picks);
93    }

123   private void parseArrivals(NodeList arrivalElements, Map<String, Pick> picks) {
124     final int arrivalCount = arrivalElements.getLength();
125     for (int idx = 0; idx < arrivalCount; idx++) {
126       final Arrival arrival = new Arrival((Element) arrivalElements.item(idx), picks);
127       arrivals.put(arrival.publicId, arrival);
128     }
129   }

131   /**
132    * To XML Element.
133    * @param doc xml document
134    * @return xml element
135    */
136   public Element toElement(Document doc) {
137     Element origin = doc.createElement("origin");
138     origin.setAttribute("publicID", publicId);
139     origin.appendChild(time.toElement(doc));
140     origin.appendChild(longitude.toElement(doc, "longitude"));
141     origin.appendChild(latitude.toElement(doc, "latitude"));
142     if (depth != null) {
143       origin.appendChild(depth.toElement(doc, "depth"));
144     }
145     if (quality != null) {
146       origin.appendChild(quality.toElement(doc));
147     }
148     for (Arrival arrival : arrivals.values()) {
149       origin.appendChild(arrival.toElement(doc));
150     }
151     if (evaluationMode != null) {
152       Element evalModeElement = doc.createElement("evaluationMode");
153       evalModeElement.appendChild(doc.createTextNode(evaluationMode.toString().toLowerCase()));
154       origin.appendChild(evalModeElement);
155     }
156     if (evaluationStatus != null) {
157       Element evalStatusElement = doc.createElement("evaluationStatus");
158       evalStatusElement.appendChild(doc.createTextNode(evaluationStatus.toString().toLowerCase()));
159       origin.appendChild(evalStatusElement);
160     }
161     return origin;
162   }

164   @Override
165   public String toString() {
166     StringBuffer sb = new StringBuffer();
167     sb.append("PublicId: " + publicId + "\n");
168     sb.append("Time: " + time.toString() + "\n");
169     sb.append("Location: " + latitude + "°, " + longitude + "°\n");
170     sb.append("Depth: " + depth + "m\n");
171     sb.append(quality);
172     return sb.toString();
173   }
