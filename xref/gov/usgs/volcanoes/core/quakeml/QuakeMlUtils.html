8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;

11  import java.text.ParseException;
12  import java.text.SimpleDateFormat;
13  import java.util.Date;
14  import java.util.TimeZone;

16  /**
17   * A utility class for working with QuakeML files.
18   *
19   * @author Tom Parker
20   *
21   */
22  public class QuakeMlUtils {
23    private static final String DATE_FORMAT = "yyyy-MM-dd'T'HH:mm:ss.SSSX";
24    private static final Logger LOGGER = LoggerFactory.getLogger(QuakeMlUtils.class);
25  
26    /**
27     * Parse a QuakeML time string. Similar to ISO 8601, but not quite the same. Because yet another
28     * time standard is exactly what we need. When will we learn?
29     *
30     * @param timeString Time string
31     * @return time in typical epoch ms
32     */
33    public static long parseTime(String timeString) {
34      /*
35       * final String inString = timeString; timeString = timeString.replaceFirst("\\.(\\d)Z?",
36       * ".$100Z"); timeString = timeString.replaceFirst("\\.(\\d{2})Z?$", ".$10Z"); timeString =
37       * timeString.replaceFirst(":(\\d{2})Z?$", ":$1.000Z");
38       * 
39       * long time = Long.MIN_VALUE; final SimpleDateFormat dateF = new SimpleDateFormat(DATE_FORMAT);
40       * dateF.setTimeZone(TimeZone.getTimeZone("UTC")); try { time =
41       * dateF.parse(timeString).getTime(); } catch (final ParseException ex) {
42       * LOGGER.error("Cannot parse time String {}", inString); throw new
43       * RuntimeException("Cannot parse time string " + inString); } return time;
44       */
45      return parseDate(timeString).getTime();
46    }
47  
48    /**
49     * Parse a QuakeML time string. Similar to ISO 8601, but not quite the same. Because yet another
50     * time standard is exactly what we need. When will we learn?
51     * 
52     * @param timeString Time string
53     * @return date object
54     */
55    public static Date parseDate(String timeString) {
56      final String inString = timeString;
57      timeString = timeString.replaceFirst("\\.(\\d)Z?", ".$100Z");
58      timeString = timeString.replaceFirst("\\.(\\d{2})Z?$", ".$10Z");
59      timeString = timeString.replaceFirst(":(\\d{2})Z?$", ":$1.000Z");
60  
61      final SimpleDateFormat dateF = new SimpleDateFormat(DATE_FORMAT);
62      dateF.setTimeZone(TimeZone.getTimeZone("UTC"));
63      try {
64        Date dt = dateF.parse(timeString);
65        return dt;
66      } catch (final ParseException ex) {
67        LOGGER.error("Cannot parse time String {}", inString);
68        throw new RuntimeException("Cannot parse time string " + inString);
69      }
70    }
71  
72    /**
73     * Format date for QuakeML
74     * 
75     * @param millis milliseconds since 1/1/1970 00:00:00 GMT
76     * @return time string in yyyy-MM-dd'T'HH:mm:ss.SSSX format
77     */
78    public static String formatDate(long millis) {
79      SimpleDateFormat dateF = new SimpleDateFormat(DATE_FORMAT);
80      dateF.setTimeZone(TimeZone.getTimeZone("UTC"));
81      Date date = new Date(millis);
82      return dateF.format(date);
83    }
84  }
