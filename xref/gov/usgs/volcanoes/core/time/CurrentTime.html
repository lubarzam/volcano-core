27  /**
28   * A class for getting the current time using an NTP server. Once accurate
29   * time is retrieved from one of the servers, it is cached. By default,
30   * if a call is made for the current time after more than 10 minutes since the
31   * accurate time was last retrieved then the NTP server is queried again.
32   *
33   * <p>If a file called 'NTP.config' is in the current directory it is used to
34   * optionally specify the servers, timeout, or recalibration interval. Example:
35   *
36   * <p># servers, comma-separated list in order
37   * servers=time-a.nist.gov,132.163.4.101,129.6.15.28
38   *
39   * <p># timeout in milliseconds
40   * timeout=500
41   *
42   * <p># interval between recalibration in milliseconds
43   * recalibrationInterval=60000
44   *
45   * @author Dan Cervelli
46   * @author Tom Parker
47   */
48  public class CurrentTime {
49    private static class CurrentTimeHolder {
50      public static CurrentTime currentTime = new CurrentTime();
51    }
52  
53    private static final String CONFIG_FILENAME = "NTP.config";
54    private static final SimpleDateFormat DATE_FORMAT;
55    private static final String[] DEFAULT_NTP_SERVERS =
56        new String[] {"0.pool.ntp.org", "1.pool.ntp.org", "time.nist.gov"};
57  
58    private static final int DEFAULT_RECALIBRATION_INTERVAL = 10 * 60 * 1000; // 10 minutes
59    private static final int DEFAULT_TIMEOUT = 500;
60  
61    private static final Logger LOGGER = LoggerFactory.getLogger(CurrentTime.class);
62  
63    static {
64      DATE_FORMAT = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss.SSS");
65      DATE_FORMAT.setTimeZone(TimeZone.getTimeZone("UTC"));
66    }

69     * Realize singleton pattern, permit only 1 instance of class in the application.

72      return CurrentTimeHolder.currentTime;
73    }

75    /**
76     * Main method.
77     *
78     * @param args command line args
79     */
80    public static void main(String[] args) {
81      final long now = CurrentTime.getInstance().now();
82  
83      System.out.printf(
84          "        GMT Time\nMillis: %d\n   J2K: %f\n    EW: %f\n  Date: %s\nOffset: %d\n", now,
85          J2kSec.fromEpoch(now), Ew.fromEpoch(now), Time.toDateString(now),
86          CurrentTime.getInstance().getLastOffset());
87    }

89    private long lastOffset;
90  
91    private boolean netFailed = false;
92  
93    private long recalibrationInterval = DEFAULT_RECALIBRATION_INTERVAL;
94  
95    private String[] servers = DEFAULT_NTP_SERVERS;
96  
97    private boolean synchronizeDisabled = false;
98  
99    private int timeout = DEFAULT_TIMEOUT;
100 
101   /**
102    * Default constructor.
103    */
104   private CurrentTime() {
105     final List<String> canaditeNames = new LinkedList<String>();
106     canaditeNames.add(CONFIG_FILENAME);
107     canaditeNames.add(System.getProperty("user.home") + File.separatorChar + CONFIG_FILENAME);
108     final String configFile =
109         StringUtils.stringToString(ConfigFile.findConfig(canaditeNames), CONFIG_FILENAME);
110 
111     final ConfigFile cf = new ConfigFile(configFile);
112     if (cf.wasSuccessfullyRead()) {
113       final String svrs = cf.getString("servers");
114       if (svrs != null) {
115         servers = svrs.split(",");
116       }
117       timeout = StringUtils.stringToInt(cf.getString("timeout"), DEFAULT_TIMEOUT);
118       synchronizeDisabled = StringUtils.stringToBoolean(cf.getString("synchronizeDisabled"));
119       recalibrationInterval = StringUtils.stringToInt(cf.getString("recalibrationInterval"),
120           DEFAULT_RECALIBRATION_INTERVAL);
121     }
122 
123     new Thread(offsetUpdater()).start();
124 
125   }
126 
127   private Runnable offsetUpdater() {
128     Runnable run = new Runnable() {
129       public void run() {
130         while (true) {
131           getOffset();
132           try {
133             Thread.sleep(recalibrationInterval);
134           } catch (InterruptedException ignore) {
135             LOGGER.debug("offsetUpdater interrupted, I'll recalibrate early.");
136           }
137         }
138       }
139     };
140 
141     return run;
142   }

153   /**
154    * Query configured ntp servers.
155    *
156    * @return result
157    */
158   public synchronized long getOffset() {
159     if (synchronizeDisabled == true) {
160       return 0;
161     }
162 
163     final Retriable<Long> rt = new Retriable<Long>("getCurrentTime()", servers.length) {
164       private int attempt = 0;
165 
166       @Override
167       public boolean attempt() throws UtilException {
168         double localClockOffset = 0;
169         DatagramSocket socket = null;
170         try {
171           socket = new DatagramSocket();
172           socket.setSoTimeout(timeout);
173           final InetAddress address = InetAddress.getByName(servers[attempt]);
174           final byte[] buf = new NtpMessage().toByteArray();
175           DatagramPacket packet = new DatagramPacket(buf, buf.length, address, 123);
176 
177           NtpMessage.encodeTimestamp(packet.getData(), 40,
178               (System.currentTimeMillis() / 1000.0) + 2208988800.0);
179 
180           socket.send(packet);
181 
182           // Get response
183           packet = new DatagramPacket(buf, buf.length);
184           socket.receive(packet);
185 
186           // Immediately record the incoming timestamp
187           final double destinationTimestamp = (System.currentTimeMillis() / 1000.0) + 2208988800.0;
188 
189           // Process response
190           final NtpMessage msg = new NtpMessage(packet.getData());
191 
192           localClockOffset = ((msg.receiveTimestamp - msg.originateTimestamp)
193               + (msg.transmitTimestamp - destinationTimestamp)) / 2;
194           final long l = Math.round(localClockOffset * 1000);
195           result = new Long(l);
196           lastOffset = l;
197           LOGGER.debug("Successfully synchronized with NTP server: {}, offset: {}",
198               servers[attempt], lastOffset);
199           socket.close();
200           return true;
201         } catch (final Exception ex) {
202           LOGGER.debug("Could not synchronize with NTP server: " + servers[attempt]);
203         }
204 
205         if (socket != null) {
206           socket.close();
207         }
208 
209         return false;
210       }
211 
212       @Override
213       public void attemptFix() {
214         attempt++;
215       }
216     };
217     Long result = null;
218     try {
219       rt.setOutput(false);
220       result = rt.go();
221     } catch (final UtilException ex) {
222       // Do nothing
223     }
224     if (result == null) {
225       netFailed = true;
226       return 0;
227     } else {
228       return result.longValue();
229     }
230   }
231 
232   /**
233    * Get current time.
234    *
235    * @return calibrated time, as long
236    */
237   public long now() {
238     Long time = System.currentTimeMillis();
239     if (!netFailed) {
240       time += lastOffset;
241     }
242 
243     return time;
244   }
