3   /**
4    * Container for two objects, possible different types.
5    * 
6    * @author Dan Cervelli
7    */
8   public class Pair<O1, O2> {
9     public O1 item1;
10    public O2 item2;
11  
12    /**
13     * Constructor.
14     * 
15     * @param o1 first object
16     * @param o2 second object
17     */
18    public Pair(O1 o1, O2 o2) {
19      item1 = o1;
20      item2 = o2;
21    }
22  
23    /**
24     * Yields true if otherPair same as this.
25     * 
26     * @param otherPair pair being compared to this
27     * @return result of comparison
28     */
29    public boolean equals(Pair<O1, O2> otherPair) {
30      return item1.equals(otherPair.item1) && item2.equals(otherPair.item2);
31    }
32  
33    /**
34     * String representation of this.
35     * 
36     * @return String representation of this 
37     */
38    public String toString() {
39      return item1.toString() + "&" + item2.toString();
40    }
41  }
