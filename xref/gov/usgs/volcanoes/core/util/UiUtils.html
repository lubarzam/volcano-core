7   package gov.usgs.volcanoes.core.util;
8   
9   import java.awt.event.ActionEvent;
10  
11  import javax.swing.AbstractAction;
12  import javax.swing.AbstractButton;
13  import javax.swing.JComponent;
14  import javax.swing.KeyStroke;
15  
16  /**
17   * Utility methods for working with UI elements.
18   * 
19   * @author Dan Cervelli
20   * @author Tom Parker
21   *
22   */
23  public final class UiUtils {
24  
25    /**
26     * Adds to JComponent action and keystroke for it.
27     * 
28     * @param comp JComponent
29     * @param ks KeyStroke
30     * @param name name
31     * @param action action
32     */
33    public static void mapKeyStrokeToAction(final JComponent comp, final String ks, final String name,
34        final AbstractAction action) {
35      comp.getInputMap(JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT).put(KeyStroke.getKeyStroke(ks),
36          name);
37      comp.getActionMap().put(name, action);
38    }
39  
40    /**
41     * Adds to JComponent button and keystroke for it.
42     * 
43     * @param comp JComponent
44     * @param ks KeyStroke
45     * @param name name
46     * @param button action
47     */
48    public static void mapKeyStrokeToButton(final JComponent comp, final String ks, final String name,
49        final AbstractButton button) {
50      comp.getInputMap(JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT).put(KeyStroke.getKeyStroke(ks),
51          name);
52      comp.getActionMap().put(name, new AbstractAction() {
53        private static final long serialVersionUID = -1;
54  
55        public void actionPerformed(ActionEvent evt) {
56          button.doClick();
57        }
58      });
59    }
60  
61    /** uninstantiatable. */
62    private UiUtils() {}
63  
64  }
