8   /**
9    * Quick and Dirty xml parser. This parser is, like the SAX parser, an event
10   * based parser, but with much less functionality.
11   * 
12   * <p>Reformatted and Java 1.5-ized from article:
13   * http://www.javaworld.com/javatips/jw-javatip128_p.html
14   * 
15   * @author Steven R. Brandt, Dan Cervelli
16   */
17  public class SimpleXmlParser {
18    /**
19     * Return popped value from st if non-empty, else PRE.
20     * @param st Stack
21     * @return popped value or, if none, PRE
22     */
23    private static int popMode(Stack<Integer> st) {
24      if (!st.empty()) {
25        return ((Integer) st.pop()).intValue();
26      } else {
27        return PRE;
28      }
29    }
30  
31    private static final int TEXT = 1;
32    private static final int ENTITY = 2;
33    private static final int OPEN_TAG = 3;
34    private static final int CLOSE_TAG = 4;
35    private static final int START_TAG = 5;
36    private static final int ATTRIBUTE_LVALUE = 6;
37    private static final int ATTRIBUTE_EQUAL = 9;
38    private static final int ATTRIBUTE_RVALUE = 10;
39    private static final int QUOTE = 7;
40    private static final int IN_TAG = 8;
41    private static final int SINGLE_TAG = 12;
42    private static final int COMMENT = 13;
43    private static final int DONE = 11;
44    private static final int DOCTYPE = 14;
45    private static final int PRE = 15;
46    private static final int CDATA = 16;
47  
48    /**
49     * Parse character stream into xml document.
50     * @param doc event handler realizing actions to perform while event happens
51     * @param reader character-stream reader with processed xml text
52     * @throws Exception when things go wrong
53     */
54    public static void parse(XmlDocHandler doc, Reader reader) throws Exception {
55      Stack<Integer> st = new Stack<Integer>();
56      int depth = 0;
57      int mode = PRE;
58      int quotec = '"';
59      depth = 0;
60      StringBuffer sb = new StringBuffer();
61      StringBuffer etag = new StringBuffer();
62      String tagName = null;
63      String lvalue = null;
64      String rvalue = null;
65      Map<String, String> attrs = null;
66      st = new Stack<Integer>();
67      doc.startDocument();
68      int line = 1;
69      int col = 0;
70      boolean eol = false;
71      int ch = 0;
72      while ((ch = reader.read()) != -1) {
73  
74        // We need to map \r, \r\n, and \n to \n
75        // See XML spec section 2.11
76        if (ch == '\n' && eol) {
77          eol = false;
78          continue;
79        } else if (eol) {
80          eol = false;
81        } else if (ch == '\n') {
82          line++;
83          col = 0;
84        } else if (ch == '\r') {
85          eol = true;
86          ch = '\n';
87          line++;
88          col = 0;
89        } else {
90          col++;
91        }
92  
93        if (mode == DONE) {
94          doc.endDocument();
95          return;
96  
97          // We are between tags collecting text.
98        } else if (mode == TEXT) {
99          if (ch == '<') {
100           st.push(mode);
101           mode = START_TAG;
102           if (sb.length() > 0) {
103             doc.text(sb.toString());
104             sb.setLength(0);
105           }
106         } else if (ch == '&') {
107           st.push(mode);
108           mode = ENTITY;
109           etag.setLength(0);
110         } else {
111           sb.append((char) ch);
112         }
113         // we are processing a closing tag: e.g. </foo>
114       } else if (mode == CLOSE_TAG) {
115         if (ch == '>') {
116           mode = popMode(st);
117           tagName = sb.toString();
118           sb.setLength(0);
119           depth--;
120           if (depth == 0) {
121             mode = DONE;
122           }
123           doc.endElement(tagName);
124         } else {
125           sb.append((char) ch);
126         }
127 
128         // we are processing CDATA
129       } else if (mode == CDATA) {
130         if (ch == '>' && sb.toString().endsWith("]]")) {
131           sb.setLength(sb.length() - 2);
132           doc.text(sb.toString());
133           sb.setLength(0);
134           mode = popMode(st);
135         } else {
136           sb.append((char) ch);
137         }
138         // we are processing a comment. We are inside
139         // the <!-- .... --> looking for the -->.
140       } else if (mode == COMMENT) {
141         if (ch == '>' && sb.toString().endsWith("--")) {
142           sb.setLength(0);
143           mode = popMode(st);
144         } else {
145           sb.append((char) ch);
146         }
147         // We are outside the root tag element
148       } else if (mode == PRE) {
149         if (ch == '<') {
150           mode = TEXT;
151           st.push(mode);
152           mode = START_TAG;
153         }
154 
155         // We are inside one of these <? ... ?>
156         // or one of these <!DOCTYPE ... >
157       } else if (mode == DOCTYPE) {
158         if (ch == '>') {
159           mode = popMode(st);
160           if (mode == TEXT) {
161             mode = PRE;
162           }
163         }
164 
165         // we have just seen a < and
166         // are wondering what we are looking at
167         // <foo>, </foo>, <!-- ... --->, etc.
168       } else if (mode == START_TAG) {
169         mode = popMode(st);
170         if (ch == '/') {
171           st.push(mode);
172           mode = CLOSE_TAG;
173         } else if (ch == '?') {
174           mode = DOCTYPE;
175         } else {
176           st.push(mode);
177           mode = OPEN_TAG;
178           tagName = null;
179           attrs = new HashMap<String, String>();
180           sb.append((char) ch);
181         }
182 
183         // we are processing an entity, e.g. &lt;, &#187;, etc.
184       } else if (mode == ENTITY) {
185         if (ch == ';') {
186           mode = popMode(st);
187           String cent = etag.toString();
188           etag.setLength(0);
189           if (cent.equals("lt")) {
190             sb.append('<');
191           } else if (cent.equals("gt")) {
192             sb.append('>');
193           } else if (cent.equals("amp")) {
194             sb.append('&');
195           } else if (cent.equals("quot")) {
196             sb.append('"');
197           } else if (cent.equals("apos")) {
198             sb.append('\'');
199           } else if (cent.startsWith("#x")) {
200             sb.append((char) Integer.parseInt(cent.substring(2), 16));
201           } else if (cent.startsWith("#")) {
202             sb.append((char) Integer.parseInt(cent.substring(1)));
203             // Insert custom entity definitions here
204           } else {
205             exc("Unknown entity: &" + cent + ";", line, col);
206           }
207         } else {
208           etag.append((char) ch);
209         }
210 
211         // we have just seen something like this:
212         // <foo a="b"/
213         // and are looking for the final >.
214       } else if (mode == SINGLE_TAG) {
215         if (tagName == null) {
216           tagName = sb.toString();
217         }
218         if (ch != '>') {
219           exc("Expected > for tag: <" + tagName + "/>", line, col);
220         }
221         doc.startElement(tagName, attrs);
222         doc.endElement(tagName);
223         if (depth == 0) {
224           doc.endDocument();
225           return;
226         }
227         sb.setLength(0);
228         attrs = new HashMap<String, String>();
229         tagName = null;
230         mode = popMode(st);
231 
232         // we are processing something
233         // like this <foo ... >. It could
234         // still be a <!-- ... --> or something.
235       } else if (mode == OPEN_TAG) {
236         if (ch == '>') {
237           if (tagName == null) {
238             tagName = sb.toString();
239           }
240           sb.setLength(0);
241           depth++;
242           doc.startElement(tagName, attrs);
243           tagName = null;
244           attrs = new HashMap<String, String>();
245           mode = popMode(st);
246         } else if (ch == '/') {
247           mode = SINGLE_TAG;
248         } else if (ch == '-' && sb.toString().equals("!-")) {
249           mode = COMMENT;
250         } else if (ch == '[' && sb.toString().equals("![CDATA")) {
251           mode = CDATA;
252           sb.setLength(0);
253         } else if (ch == 'E' && sb.toString().equals("!DOCTYP")) {
254           sb.setLength(0);
255           mode = DOCTYPE;
256         } else if (Character.isWhitespace((char) ch)) {
257           tagName = sb.toString();
258           sb.setLength(0);
259           mode = IN_TAG;
260         } else {
261           sb.append((char) ch);
262         }
263 
264         // We are processing the quoted right-hand side
265         // of an element's attribute.
266       } else if (mode == QUOTE) {
267         if (ch == quotec) {
268           rvalue = sb.toString();
269           sb.setLength(0);
270           attrs.put(lvalue, rvalue);
271           mode = IN_TAG;
272           // See section the XML spec, section 3.3.3
273           // on normalization processing.
274         } else if (" \r\n\t".indexOf(ch) >= 0) {
275           sb.append(' ');
276         } else if (ch == '&') {
277           st.push(mode);
278           mode = ENTITY;
279           etag.setLength(0);
280         } else {
281           sb.append((char) ch);
282         }
283 
284       } else if (mode == ATTRIBUTE_RVALUE) {
285         if (ch == '"' || ch == '\'') {
286           quotec = ch;
287           mode = QUOTE;
288         } else if (Character.isWhitespace((char) ch)) {
289           ;
290         } else {
291           exc("Error in attribute processing", line, col);
292         }
293 
294       } else if (mode == ATTRIBUTE_LVALUE) {
295         if (Character.isWhitespace((char) ch)) {
296           lvalue = sb.toString();
297           sb.setLength(0);
298           mode = ATTRIBUTE_EQUAL;
299         } else if (ch == '=') {
300           lvalue = sb.toString();
301           sb.setLength(0);
302           mode = ATTRIBUTE_RVALUE;
303         } else {
304           sb.append((char) ch);
305         }
306 
307       } else if (mode == ATTRIBUTE_EQUAL) {
308         if (ch == '=') {
309           mode = ATTRIBUTE_RVALUE;
310         } else if (Character.isWhitespace((char) ch)) {
311           ;
312         } else {
313           exc("Error in attribute processing.", line, col);
314         }
315 
316       } else if (mode == IN_TAG) {
317         if (ch == '>') {
318           mode = popMode(st);
319           doc.startElement(tagName, attrs);
320           depth++;
321           tagName = null;
322           attrs = new HashMap<String, String>();
323         } else if (ch == '/') {
324           mode = SINGLE_TAG;
325         } else if (Character.isWhitespace((char) ch)) {
326           ;
327         } else {
328           mode = ATTRIBUTE_LVALUE;
329           sb.append((char) ch);
330         }
331       }
332     }
333     if (mode == DONE) {
334       doc.endDocument();
335     } else {
336       exc("missing end tag", line, col);
337     }
338   }

340   /**
341    * Generates error diagnostic message and throws exception with it.
342    * @param str text error description
343    * @param line line number
344    * @param col column number
345    * @throws Exception when things go wrong
346    */
347   private static void exc(String str, int line, int col) throws Exception {
348     throw new Exception(str + " near line " + line + ", column " + col);
349   }
