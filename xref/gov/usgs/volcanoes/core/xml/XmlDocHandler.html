3   import java.util.Map;

5   /**
6    * XML event handler for {@link SimpleXmlParser}. 
7    * User should extend this interface with concrete methods processing
8    * xml actions, and pass it to parser. Parser will call appropriate 
9    * method while event happens.
10   * 
11   * <P>Reformatted and Java 1.5-ized from article:
12   * http://www.javaworld.com/javatips/jw-javatip128_p.html
13   * 
14   * @author Steven R. Brandt, Dan Cervelli
15   */
16  public interface XmlDocHandler {
17    /**
18     * Element started.
19     * @param tag Tag name
20     * @param hh map of tag attributes and values
21     * @throws Exception when things go wrong
22     */
23    public void startElement(String tag, Map<String, String> hh) throws Exception;
24  
25    /**
26     * Element ended.
27     * @param tag Tag name
28     * @throws Exception when things go wrong
29     */
30    public void endElement(String tag) throws Exception;
31  
32    /**
33     * Document started.
34     * @throws Exception when things go wrong
35     */
36    public void startDocument() throws Exception;
37  
38    /**
39     * Document ended.
40     * @throws Exception when things go wrong
41     */
42    public void endDocument() throws Exception;
43  
44    /**
45     * Text or CDATA found.
46     * @param str input string
47     * @throws Exception when things go wrong
48     */
49    public void text(String str) throws Exception;
50  }
